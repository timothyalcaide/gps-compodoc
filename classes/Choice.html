import { Domain } from '@clients/adventure/bean/waypoint';
import { Kind, Sign } from '@features/map/models';
import { WaypointType } from '@models/waypoint/waypoint-type';

/**
 * Choice sign map element.
 * It is a specific type of sign that gives the alternative between job domains or diploma domains.
 * Its associated road is either followed by a road of the same type of domain or a crossroad
 * followed by roads of the same type of domain
 */
export class Choice extends Sign {
  public readonly kind: Kind = Kind.Choice;

  /**
   * Choice sign map element.
   * It is a specific type of sign that gives the alternative between job domains or diploma
   * domains. Its associated road is either followed by a road of the same type of domain or a
   * crossroad followed by roads of the same type of domain
   *
   * @param type The type (diploma or job)
   * @param label The label
   * @param userCount The popularity of the choice
   */
  constructor(type: WaypointType.diploma | WaypointType.job, label: string, userCount: number) {
    super(
      type === WaypointType.diploma
        ? ({
            type: WaypointType.diploma,
            nomenclature: WaypointType.domain,
            label,
            userCount
          } as Domain)
        : ({
            type: WaypointType.job,
            nomenclature: WaypointType.domain,
            label,
            userCount
          } as Domain)
    );
  }
}
