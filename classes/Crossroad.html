import { District, Kind, MapBowElement, Position, Road, Village } from '@features/map/models';
import { Utils, Vector } from '@features/map/utils';

/**
 * Crossroad map element.
 * It is a map representation of the choice between different domains
 */
export class Crossroad extends MapBowElement {
  /**
   * The children.
   * A crossroad has children which can only be roads
   */
  public readonly children: Road[] = [];

  public readonly start: Position;
  public readonly end: Position;
  public readonly orientation: Vector;
  public readonly kind = Kind.Crossroad;

  /**
   * The outer radius
   */
  public readonly radius: number;

  /**
   * The inner radius
   */
  public readonly innerRadius: number;

  /**
   * The center
   */
  public readonly center: Position;

  private readonly RADIUS = 50;
  private readonly INNER_RADIUS_FACTOR = 0.42;
  private readonly CROSSROAD_MAX_DISTANCE = 52;
  private readonly _stage: number = 0;

  public set child(child: Road) {
    this.children.push(child);
  }

  public get stage(): number {
    return this._stage;
  }

  public get selected(): boolean {
    return this.children.reduce((selected, childRoad) => {
      if (childRoad.selected) return true;

      return selected;
    }, false);
  }

  /**
   * Whether the crossroad is selected or not.
   * Set is only useful to unselect it
   */
  public set selected(selected: boolean) {
    if (!selected) {
      this.children.forEach(childRoad => {
        // eslint-disable-next-line no-param-reassign
        childRoad.selected = false;
      });
    }
  }

  /**
   * Get the user count depending on its parent
   */
  public get userCount(): number {
    if (this.parent.kind === Kind.District) {
      const districtSelectedVillage: Village = this.parent.selectedVillage;

      // The previous district may not be selected when the user unselect the village
      return (
        districtSelectedVillage?.waypoint.userCount || this.parent.villages[0].waypoint.userCount
      );
    }

    return this.parent.sign.waypoint.userCount;
  }

  /**
   * Crossroad map element.
   * It is a map representation of the choice between different domains
   *
   * @param parent The parent. It is either a road or a district
   */
  constructor(public readonly parent: Road | District) {
    super();

    this.parent.child = this;

    this.radius = this.RADIUS;
    this.innerRadius = this.RADIUS * this.INNER_RADIUS_FACTOR;
    this.start = this.parent.end;
    this.orientation = this.parent.orientation;

    this.end = Utils.orientedEndPoint(
      this.start,
      this.orientation,
      true,
      this.CROSSROAD_MAX_DISTANCE
    );

    this.center = new Position(this.end).add(this.orientation.multiply(this.radius));

    this._stage = this.calculateStage();
  }

  /**
   * Calculate the stage
   */
  private calculateStage(): number {
    return this.parent.kind === Kind.District ? this.parent.stage + 1 : this.parent.stage;
  }
}
