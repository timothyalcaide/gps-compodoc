import { Crossroad, Kind, MapBowElement, Position, Road, Village } from '@features/map/models';
import { Utils, Vector } from '@features/map/utils';

/**
 * District map element.
 * A district gathers multiple villages of the same domain
 */
export class District extends MapBowElement {
  /**
   * The children.
   * A district only has one child which is a road or a crossroad
   */
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  public readonly children: Array<any> = [];

  public readonly start: Position;
  public readonly end: Position;
  public readonly orientation: Vector;
  public readonly kind = Kind.District;

  /**
   * The village list of the district
   */
  public villages: Village[] = [];

  private _selectedVillage: Village = null;

  /**
   * Set the unique child of the district
   */
  public set child(child: Road | Crossroad) {
    if (this.children.length) this.children.shift();

    this.children.push(child);
  }

  public get stage(): number {
    return this.parent?.stage || 0;
  }

  public get selected(): boolean {
    return !!this.selectedVillage;
  }

  public set selected(selected: boolean) {
    if (!selected) {
      this.villages.forEach(village => {
        // eslint-disable-next-line no-param-reassign
        village.selected = false;
      });

      if (this.children.length) this.children[0].selected = false;
    }
  }

  /**
   * Whether the district is visible or not.
   * It is only displayed if it is visible and it has been discovered
   */
  public get visible(): boolean {
    // A district is visible if it is the first district or its parent road is selected
    if (!this.parent || this.parent.selected) return true;

    // Otherwise, it is visible if the parent of its parent road is not a crossroad and is selected
    if (this.parent.parent.kind !== Kind.Crossroad) return this.parent.parent.selected;

    /**
     * If the parent of the its parent road is a crossroad, it is visible if the parent of the
     * crossroad is selected
     */
    return this.parent.parent.parent.selected;
  }

  /**
   * The selected village in the district
   * null if there is not selected village
   */
  public get selectedVillage(): Village {
    return this._selectedVillage;
  }

  /**
   * The selected village in the district
   * null if there is not selected village
   */
  public set selectedVillage(village: Village) {
    this._selectedVillage = village;
  }

  public get villagesToDisplay(): Village[] {
    const { selectedVillage } = this;

    return selectedVillage ? [selectedVillage] : this.villages.slice(0, 3);
  }

  /**
   * Whether the district is disabled or not
   */
  public get disabled(): boolean {
    return !!this.parent && !this.parent.selected;
  }

  /**
   * District map element.
   * A district gathers multiple villages from the same domain
   *
   * @param parent The road parent
   * @param start The start position. Only used to initialize district for the first district
   * @param end The end position. Only used to initialize district for the first district
   * @param orientation The orientation. Only used to initialize district for the first district
   */
  constructor(
    public readonly parent: Road,
    start?: Position,
    end?: Position,
    orientation?: Vector
  ) {
    super();

    if (this.parent) {
      this.parent.child = this;

      this.start = this.parent.end;
      this.orientation = this.parent.orientation;
      this.end = Utils.orientedEndPoint(this.start, this.orientation, true);
    } else {
      this.start = start;
      this.end = end;
      this.orientation = orientation;
    }
  }
}
