import { DateFilter } from '@admin-models/filters/date-filter';
import dayjs, { Dayjs } from 'dayjs';

/**
 * Filters for the back-office
 */
export class Filters {
  /**
   * Filters for the back-office
   *
   * @param date Filter used for filtering by date
   * @param application Filter used for filtering by application
   * @param search Filter used for filtering by search text
   */
  constructor(public date?: DateFilter, public application?: string, public search?: string) {}

  /**
   * Merge properties from the provided filter into the actuel filter properties
   *
   * @param filters The filters to merge
   * @returns The new filters with properties merged
   */
  public merge(filters: Filters): Filters {
    return new Filters(
      filters.date || this.date,
      filters.application || this.application,
      filters.search || this.search
    );
  }

  /**
   * Get the start date from a date filter
   *
   * @returns The start date
   */
  public dateFilterToDayjs(): Dayjs {
    if (!this.date) return undefined;

    switch (this.date) {
      case DateFilter.LastWeek:
        return dayjs().subtract(7, 'day');
      case DateFilter.LastMonth:
        return dayjs().subtract(30, 'day');
      case DateFilter.Last3Month:
      default:
        return dayjs().subtract(3, 'month');
    }
  }
}
