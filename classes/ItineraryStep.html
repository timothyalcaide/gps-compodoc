import { Waypoint } from '@clients/adventure/bean/waypoint';
import { Step } from '@clients/poi/poi';
import { StepLocation } from '@models/step-location/step-location';
import { WaypointType } from '@models/waypoint/waypoint-type';

export class ItineraryStep {
  /**
   * Whether the itinerary step is selected in the itinerary
   */
  public selected: boolean;

  public get label(): string {
    return this._label || this.step?.name || this.waypoint?.label || '';
  }

  public get type(): WaypointType {
    return this._type || this.step?.type || this.waypoint?.nomenclature || WaypointType.diploma;
  }

  /**
   * The itinerary step location. Only used if the itinerary step is a human step
   */
  public get location(): StepLocation {
    return this.step?.location;
  }

  /**
   * The substeps. Only used if the itinerary step is a human step
   */
  public get substeps(): Step[] {
    return this.step?.substeps || [];
  }

  /**
   * Whether the itinerary step is a final step in the itinerary
   * It is different from being the last (the last step can be not final)
   */
  public get final(): boolean {
    if (this.step) return this.isLast;

    if (this.waypoint) return this.waypoint.final;

    return false;
  }

  public get pinIcon(): string {
    if (this._pinIcon) return this._pinIcon;

    if (this.final || this.isLast) return 'pin-village-destination';

    return this.selected ? 'pin-village-selected' : 'pin-village';
  }

  public get userCount(): number {
    return this.waypoint?.userCount || 0;
  }

  public get level(): number {
    return this.waypoint?.level || 0;
  }

  public cost = 0;

  /**
   * Itinerary step represents a step in an UI itinerary
   *
   * @param step The step defining the itinerary step. When provided, waypoint must be undefined
   * @param waypoint The waypoint defining the itinerary step. When provided, step must be undefined
   * @param isFirst Whether the itinerary step is the first in the itinerary
   * @param isLast Whether the itinerary step is the last in the itinerary
   * @param _label Label to override itinerary label
   * @param _type Type to override itinerary type
   * @param definition The definition associated to the instance from wh
   */
  constructor(
    private step?: Step,
    private waypoint?: Waypoint,
    public isFirst: boolean = false,
    private isLast: boolean = false,
    private _label?: string,
    private _type?: WaypointType,
    private _pinIcon?: string
  ) {
    this.isLast = this.isLast && !this.isFirst;
  }
}
