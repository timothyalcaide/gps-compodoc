import { Regions } from '@organisms/localisation-filter-dialog/regions';
import { City } from '../../city/city';
import { IFilters } from '../filters.interface';

/**
 * Localisation filters
 */
export class LocalisationFilters implements IFilters {
  public get activeFilters(): number {
    return +!!this.regional.length + +!!this.city + +!!this.userGeolocation;
  }

  /**
   * Localisation filters
   *
   * @param userGeolocation The user geolocation if results have to be around him. Is null if this
   *                        filter is disabled. It can not be enabled if city is enabled
   * @param city A selected city if results have to be nearby the selected city. Is null if this
   *             filter is disabled. It can not be enabled if user geolocation is enabled
   * @param regional A list of selected regions if results have to be in the selected regions. Is
   *                 empty if this filter is disabled.
   */
  constructor(
    public userGeolocation: GeolocationCoordinates = null,
    public city: City = null,
    public regional: Regions[] = []
  ) {
    // Disable city filter if user geolocation is defined
    if (userGeolocation && this.city) this.city = null;
  }

  public empty(): void {
    this.regional = [];
    this.city = null;
    this.userGeolocation = null;
  }

  public copy(filter: LocalisationFilters): void {
    this.regional = [...filter.regional];
    this.city = filter.city;
    this.userGeolocation = filter.userGeolocation;
  }
}
