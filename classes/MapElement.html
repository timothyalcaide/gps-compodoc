import { Position } from '@features/map/models';
import { Vector } from '@features/map/utils';

export enum Kind {
  Choice = 'Choice',
  Crossroad = 'Crossroad',
  District = 'District',
  MoreDomain = 'MoreDomain',
  Road = 'Road',
  Sign = 'Sign',
  Village = 'Village'
}

/**
 * Map element
 */
export abstract class MapElement {
  private static uniqueIdCounter = 0;

  /**
   * The unique Id that identifies the map element
   */
  public readonly uniqueId: number;

  /**
   * The start position (left middle)
   */
  public readonly start: Position;

  /**
   * The end position (right middle)
   */
  public readonly end: Position;

  /**
   * The orientation
   */
  public readonly orientation: Vector;

  /**
   * The kind
   */
  public readonly kind: Kind;

  /**
   * Whether the map element is selected or not
   */
  public abstract get selected(): boolean;
  public abstract set selected(selected: boolean);

  /**
   * The stage
   */
  public abstract get stage(): number;

  /**
   * Map element
   */
  constructor() {
    this.uniqueId = MapElement.uniqueIdCounter++;
  }
}
