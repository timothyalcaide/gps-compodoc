import { Domain } from '@clients/adventure/bean/waypoint';
import { Kind, Road, Sign } from '@features/map/models';

/**
 * More domain sign map element.
 * It is a specific type of sign that allows to select a domain different from the three most famous
 * ones
 */
export class MoreDomain extends Sign {
  public readonly kind: Kind = Kind.MoreDomain;

  /**
   * The waypoint
   * It is not read-only as it can be updated by selecting a new domain
   */
  public waypoint: Domain;

  private _originDomains: Domain[] = null;

  /**
   * Whether the sign is disabled or not
   */
  public get disabled(): boolean {
    return !this.road.parent.parent.selected;
  }

  /**
   * The three most famous domains displayed along the more domain sign
   */
  public get originDomains(): Domain[] {
    if (!this._originDomains) this._originDomains = this.calculateOriginDomains();

    return this._originDomains;
  }

  /**
   * More domain sign map element.
   * It is a specific type of sign that allows to select a domain different from the three most
   * famous ones
   *
   * @param domainCount The number of sub-domain available
   */
  constructor(public domainCount: number) {
    super(null);
  }

  /**
   * Calculate the origin domains
   */
  public calculateOriginDomains(): Domain[] {
    return (this.road.parent.children as Road[]).reduce(
      (originDomains: Domain[], siblingRoad: Road) => {
        if (siblingRoad === this.road) return originDomains;

        return [...originDomains, siblingRoad.sign.waypoint];
      },
      []
    );
  }
}
