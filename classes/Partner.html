import { PartnerId } from './partner-id';
import { ApiPartner } from './partner.api';

/**
 * An application partner
 */
export class Partner {
  /**
   * An application partner
   *
   * @param isOptIn Indicate if the user has checked the opt-in for the partner
   * @param hrUuid The partner id as defined on Hroads
   */
  constructor(public isOptIn: boolean, public hrUuid: PartnerId) {}

  /**
   * Build the Partner object from the Api Partner object
   *
   * @param apiPartner The Api Partner object
   * @returns The new Partner object
   */
  public static build(apiPartner: ApiPartner): Partner {
    return new Partner(apiPartner.is_optin, apiPartner.partner.hr_uuid);
  }

  /**
   * Get the Api Partner object from the Partner object
   *
   * @returns The corresponding Api Partner object
   */
  public toApi(): ApiPartner {
    return {
      is_optin: this.isOptIn,
      partner: {
        hr_uuid: this.hrUuid
      }
    } as ApiPartner;
  }
}
