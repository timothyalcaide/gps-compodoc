import { Crossroad, Kind, MapBowElement, Position, Sign } from '@features/map/models';
import { Utils, Vector } from '@features/map/utils';
import { WaypointType } from '@models/waypoint/waypoint-type';

/**
 * Road map element.
 * A road is a map representation of a diploma domain or a job domain. The domain label is displayed
 * on a sign
 */
export class Road extends MapBowElement {
  /**
   * The children.
   * A road only has one child which can be either another road, a district or a crossroad
   */
  public readonly children: Array<MapBowElement> = [];

  public readonly start: Position;
  public readonly end: Position;
  public readonly orientation: Vector;
  public readonly kind = Kind.Road;

  /**
   * Set the unique child of the road
   */
  public set child(child: MapBowElement) {
    if (this.children.length) this.children.shift();

    this.children.push(child);
  }

  public get stage(): number {
    return this.parent.kind === Kind.District ? this.parent.stage + 1 : this.parent.stage;
  }

  public get selected(): boolean {
    return this.sign.selected;
  }

  public set selected(selected: boolean) {
    if (!selected) {
      if (this.children.length) this.children[0].selected = false;
    } else {
      if (this.parent.kind === Kind.Crossroad) {
        this.parent.children.forEach(siblingRoad => {
          if (siblingRoad.uniqueId === this.uniqueId) return;

          // eslint-disable-next-line no-param-reassign
          siblingRoad.selected = false;
        });
      }

      this.children.forEach(child => {
        // eslint-disable-next-line no-param-reassign
        child.discovered = true;
      });
    }

    this.sign.selected = selected;
  }

  private readonly ROAD_MAX_DISTANCE = 400;
  private readonly ROAD_MAX_DISTANCE_SHIFT = 100;

  /**
   * Road map element.
   * A road is a map representation of a diploma domain or a job domain. The domain label is
   * displayed on a sign
   *
   * @param parent The road parent. It can be either another road, a district or a crossroad
   * @param sign The road sign
   * @param numberOfRoads The number of roads coming from the same parent
   * @param choice If the road is part of the choice between job and diploma
   */
  constructor(
    public readonly parent: MapBowElement,
    public readonly sign: Sign,
    numberOfRoads: number,
    choice?: boolean
  ) {
    super();

    this.sign.road = this;

    this.parent.child = this;

    /**
     * Initialize its orientation based on the index within the crossroad children or based on its
     * parent orientation
     */
    if (this.parent.kind === Kind.Crossroad) {
      let index = 0;

      if (sign.kind === Kind.Choice) {
        if (sign.waypoint.type === WaypointType.diploma) index = 0;
        else index = 2;
      } else {
        // As the road is always the last one when being initialized, its index is the last index
        index = this.parent.children.length - 1;
      }

      this.orientation = Utils.roadOrientation(this.parent, index, numberOfRoads, choice);
    } else this.orientation = parent.orientation;

    this.end = Utils.orientedEndPoint(
      parent.end,
      this.orientation,
      false,
      this.ROAD_MAX_DISTANCE + this.ROAD_MAX_DISTANCE_SHIFT * this.stage
    );

    // Initialize its start position based on its parent type (crossroad or district)
    if (parent.kind === Kind.Crossroad) {
      this.start = Utils.orientedEndPoint(
        (parent as Crossroad).center,
        this.orientation,
        true,
        (parent as Crossroad).radius
      );
    } else this.start = this.parent.end;
  }
}
