import { MapElement, Position } from '@features/map/models';
import { Vector } from '@features/map/utils';

export type PoiPosition = { p: 'l' | 'r'; x: number; y: number };

export class Utils {
  /* -- Orientations -- */

  /**
   * Road orientation giving by index and size of a map element list
   * And depending on the parent orientation
   */
  public static roadOrientation(
    parent: MapElement,
    index: number,
    size: number,
    choice?: boolean
  ): Vector {
    const parentOrientation = new Vector(parent.orientation);

    // Use this variable to have the first element from top to left and from top to right
    const displayDirection: number =
      Math.sign(parentOrientation.x) === 0 ? 1 : Math.sign(parentOrientation.x);

    let orientation: Vector;

    if (!size || size === 1) orientation = Utils.endPointOrientation(parentOrientation, -10, 10);
    else if (size === 2) {
      if (index === 0) orientation = Utils.endPointOrientation(parentOrientation, -50, -40);
      else if (index === 1) orientation = Utils.endPointOrientation(parentOrientation, 40, 50);
    } else if (size === 3) {
      if (index === 0) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * -45,
          displayDirection * -45
        );
      } else if (index === 1) orientation = Utils.endPointOrientation(parentOrientation, 0, 0);
      else if (index === 2) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * 45,
          displayDirection * 45
        );
      }
    } else if (size === 4 && !choice) {
      if (index === 0) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * -40,
          displayDirection * -50
        );
      } else if (index === 1) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * -5,
          displayDirection * 5
        );
      } else if (index === 2) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * 35,
          displayDirection * 45
        );
      } else if (index === 3) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * 80,
          displayDirection * 90
        );
      }
    } else if (size === 4 && choice) {
      if (index === 0) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * -130,
          displayDirection * -150
        );
      } else if (index === 1) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * -150,
          displayDirection * -170
        );
      } else if (index === 2) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * -70,
          displayDirection * -90
        );
      } else if (index === 3) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * -50,
          displayDirection * -70
        );
      }
    } else if (size === 5) {
      if (index === 0) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * -70,
          displayDirection * -70
        );
      } else if (index === 1) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * -30,
          displayDirection * -30
        );
      } else if (index === 2) {
        orientation = Utils.endPointOrientation(parentOrientation, 0, 0);
      } else if (index === 3) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * 30,
          displayDirection * 30
        );
      } else if (index === 4) {
        orientation = Utils.endPointOrientation(
          parentOrientation,
          displayDirection * 70,
          displayDirection * 70
        );
      }
    }

    return orientation;
  }

  /* -- Position -- */

  public static orientedEndPoint(
    startPos: Position,
    orientation: Vector,
    fixed?: boolean,
    maxDistance?: number
  ): Position {
    const multiplicator = fixed ? 1 : this.randomBetweenFloat(0.8, 1);

    return startPos.add(
      Vector.normalize(orientation).multiply(Math.floor(maxDistance || 500 * multiplicator))
    );
  }

  /* -- Utils -- */

  public static random(threshold: number, sign?: boolean, shift?: number): number {
    let randomSign = 1;

    if (sign) randomSign = Math.round(Math.random()) * 2 - 1; // Random -1 or 1

    // Random 1 to threshold
    return (Math.floor(Math.random() * Math.floor(threshold)) + (shift || 1)) * randomSign;
  }

  public static randomBetween(min: number, max: number): number {
    const roundMin = Math.ceil(min);
    const roundMax = Math.floor(max);

    return Math.floor(Math.random() * (roundMax - roundMin + 1)) + roundMin;
  }

  /**
   * https://riptutorial.com/javascript/example/8330/random--with-gaussian-distribution
   */
  public static normalRandom(sign?: boolean): number {
    let randomSign = 1;

    if (sign) randomSign = Math.round(Math.random()) * 2 - 1; // Random -1 or 1

    const v = 5;
    let r = 0;

    for (let i = v; i > 0; i--) r += Math.random();

    return (r / v) * randomSign;
  }

  public static angleToRadians(deg: number): number {
    return deg * (Math.PI / 180);
  }

  private static randomBetweenFloat(min: number, max: number): number {
    return Math.random() * (max - min) + min;
  }

  private static endPointOrientation(
    baseOrientation: Vector,
    minAngle: number,
    maxAngle: number
  ): Vector {
    const angle = this.angleToRadians(this.randomBetween(minAngle, maxAngle));
    const orientation = new Vector(baseOrientation.x, baseOrientation.y);

    orientation.rotate(angle);

    return orientation;
  }
}
