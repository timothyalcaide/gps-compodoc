export class Vector {
  private _x = 0;
  private _y = 0;

  public get x(): number {
    return this._x;
  }

  public set x(x: number) {
    this._x = x;
  }

  public get y(): number {
    return this._y;
  }

  public set y(y: number) {
    this._y = y;
  }

  constructor(x: number | Vector, y?: number) {
    if (x instanceof Vector) {
      this._x = x.x;
      this._y = x.y;
    } else if (typeof y === 'number') {
      this._x = x;
      this._y = y;
    }
  }

  /**
   * Normalize the vector (it's length will be 1)
   *
   * @param v
   */
  public static normalize(v: Vector): Vector {
    return v.divide(v.length());
  }

  public add(v): Vector {
    if (v instanceof Vector) return new Vector(this.x + v.x, this.y + v.y);

    return new Vector(this.x + v, this.y + v);
  }

  public multiply(v): Vector {
    if (v instanceof Vector) return new Vector(this.x * v.x, this.y * v.y);

    return new Vector(this.x * v, this.y * v);
  }

  public divide(v): Vector {
    if (v instanceof Vector) return new Vector(this.x / v.x, this.y / v.y);

    return new Vector(this.x / v, this.y / v);
  }

  /**
   * Produit scalaire de 2 vecteurs
   */
  public dot(v: Vector): number {
    return this.x * v.x + this.y * v.y;
  }

  /**
   * Longueur d'un vecteur
   */
  public length(): number {
    return Math.sqrt(this.dot(this));
  }

  public angleTo(v: Vector): number {
    return Math.acos(this.dot(v) / (this.length() * v.length()));
  }

  public rotate(teta: number): void {
    const cos = Math.cos(teta);
    const sin = Math.sin(teta);
    const x = this.x * cos - this.y * sin;
    const y = this.x * sin + this.y * cos;

    this.x = x;
    this.y = y;
  }
}
