import { Application } from '@clients/adventure/bean/application';
import { Domain, Waypoint } from '@clients/adventure/bean/waypoint';
import { IN_OFFER_CONFIG } from '@configs/in-offer.config';
import {
  District,
  Kind,
  MapBowElement,
  MapLeafElement,
  Position,
  Road
} from '@features/map/models';
import { Vector } from '@features/map/utils';
import { Product } from '@models/context/product';
import { HumanPoi } from '@models/poi/human-poi/human-poi';
import { PositionablePoi } from '@models/poi/positionable-poi/positionable-poi';
import { TeamPoi } from '@models/poi/team-poi/team-poi';

/**
 * Village map element.
 * A village is a map representation of a diploma or a job
 */
export class Village extends MapLeafElement {
  public readonly start: Position;
  public readonly end: Position;
  public readonly orientation: Vector;
  public readonly kind = Kind.Village;

  /**
   * The human POIs
   */
  public humanPois: HumanPoi[];

  /**
   * The parcours POIs
   */
  public positionablePois: PositionablePoi[];

  /**
   * The team POI
   * It exists if the village has human POIs
   */
  public teamPoi: TeamPoi;

  /**
   * The VideoAskPOI
   * It exists if the village has a human POI with a VideoAsk
   */
  public videoAskPoi: HumanPoi;

  /**
   * The horizon villages
   */
  public horizons: Waypoint[];

  /**
   * The backwards
   */
  public backwards: Waypoint[];

  private _selected = false;
  private readonly _expand: Waypoint[] = [];

  public get stage(): number {
    return this.district.stage;
  }

  public get selected(): boolean {
    return this._selected;
  }

  public set selected(selected: boolean) {
    if (this._selected === selected) return;

    this._selected = selected;

    if (selected) this.district.selectedVillage = this;
    else if (this.district.selectedVillage.uniqueId === this.uniqueId) {
      this.district.selectedVillage = null;

      if (this.district.children.length) this.district.children[0].selected = false;
    }
  }

  /**
   * The expand contains waypoints starting right after the previous village
   * The final waypoint is the current village
   */
  public get expand(): Waypoint[] {
    return this._expand;
  }

  /**
   * The village domain
   */
  public get domain(): Domain {
    return this.district.parent?.sign.waypoint;
  }

  /**
   * Village map element.
   * A village is a map representation of a diploma or a job
   *
   * @param waypoint The waypoint
   * @param district The parent district
   */
  constructor(
    private document: Document,
    public readonly waypoint: Waypoint,
    public readonly district: District
  ) {
    super();

    district.villages.push(this);

    this.start = this.district.start;
    this.end = this.district.start.add(
      new Vector(0, -1).multiply(this.document.defaultView.innerHeight)
    );

    this._expand = this.calculateExpand();
  }

  /**
   * Whether the village is in the school offer
   *
   * @param application The application
   * @returns Whether the village is in the school offer
   */
  public inOffer(application: Application, product: Product): boolean {
    return IN_OFFER_CONFIG(application, this.stage, product);
  }

  /**
   * Calculate the expand going from the previous village to this village
   */
  private calculateExpand(): Waypoint[] {
    const expand: Waypoint[] = [this.waypoint];
    let currentMapBowElement: MapBowElement = this.district.parent;

    if (!currentMapBowElement) return expand;

    while (currentMapBowElement.kind !== Kind.District) {
      if (
        currentMapBowElement.kind === Kind.Road &&
        (currentMapBowElement as Road).sign.kind !== Kind.Choice
      )
        expand.unshift((currentMapBowElement as Road).sign.waypoint);

      currentMapBowElement = currentMapBowElement.parent;
    }

    return expand;
  }
}
