import { Component, Inject, OnDestroy } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Status } from '@clients/adventure/bean/situation';
import { Tag } from '@clients/adventure/bean/tag';
import { ApplicationContext, DialogService } from '@core/services';
import { Definition, DefinitionType } from '@models/definition/definition';
import { ConfirmationDialogComponent } from '@organisms/confirmation-dialog/confirmation-dialog.component';
import { AdminTagService } from '@services/admin-tag.service';
import { Observable, Subject } from 'rxjs';
import { filter, map, mergeMap, takeUntil, tap } from 'rxjs/operators';

@Component({
  selector: 'app-admin-definition-dialog',
  templateUrl: './admin-definition-dialog.component.html'
})
export class AdminDefinitionDialogComponent implements OnDestroy {
  public form: FormGroup;
  public readonly statuses = Status;
  public tags$: Observable<Tag[]>;
  public filteredTags$: Observable<Tag[]>;

  private onDestroySubject$ = new Subject<void>();

  constructor(
    private dialog: MatDialog,
    private dialogRef: MatDialogRef<AdminDefinitionDialogComponent>,
    private dialogService: DialogService,
    fb: FormBuilder,
    private adminTagService: AdminTagService,
    private context: ApplicationContext,
    @Inject(MAT_DIALOG_DATA) public data: Definition
  ) {
    // FIXME: name unity
    this.form = fb.group({
      name: [this.data?.name, Validators.required],
      description: [this.data?.description],
      cost: [this.data?.cost],
      originId: [this.data?.originId],
      destinationId: [this.data?.destinationId],
      application: [this.data?.application ? this.data.application : this.context.applicationName],
      type: [this.data?.type ? this.data.type : DefinitionType.FROM_ADMIN],
      statuses: [this.data?.statuses, Validators.required],
      tagId: [this.data?.tagId],
      video: [this.data?.video],
      enabled: [this.data?.enabled]
    });
    this.tags$ = this.adminTagService.tags().pipe(
      tap(tags => {
        this.filteredTags$ = this.form.controls.tagId.valueChanges.pipe(
          map(value => (value ? tags.filter(tag => tag.label.includes(value)) : tags)),
          map(results => results.slice(0, 5)),
          takeUntil(this.onDestroySubject$)
        );
      })
    );
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public newTag(tags: Tag[]): boolean {
    return (
      !!this.form.getRawValue().tagId &&
      this.form.getRawValue().tagId !== '' &&
      !tags.map(t => t.id).includes(this.form.getRawValue().tagId)
    );
  }

  public onSave(tags: Tag[]): void {
    // Create tag if is a new one
    if (this.newTag(tags)) {
      this.dialog
        .open(
          ConfirmationDialogComponent,
          this.dialogService.simpleDialogConfig({
            titleKey: `Do you want to create the category: ${this.form
              .getRawValue()
              .tagId.toLowerCase()}?`
          })
        )
        .afterClosed()
        .pipe(
          filter(res => !!res),
          mergeMap(() =>
            this.adminTagService
              .save(this.createTag(this.form.getRawValue().tagId.toLowerCase()))
              .pipe(tap(tag => this.form.get('tagId').setValue(tag.id)))
          )
        )
        .subscribe(() => this.save());
    } else this.save();
  }

  public trackByTag(_: number, tag: Tag): string {
    return tag.id;
  }

  // -- PRIVATE

  private save(): void {
    this.dialogRef.close(
      Definition.build({
        ...this.data,
        ...(this.form.getRawValue() as Definition)
      })
    );
  }

  private createTag(tag: string): Tag {
    return {
      application: this.context.applicationName,
      label: tag
    } as Tag;
  }
}
