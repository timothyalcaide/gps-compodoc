import { ApplicationFilter } from '@admin-models/filters/application-filter';
import { DateFilter } from '@admin-models/filters/date-filter';
import { Filters } from '@admin-models/filters/filters';
import { AdminApplicationContextService } from '@admin/services/admin-application-context.service';
import { Component, EventEmitter, Input, OnDestroy, OnInit, Output } from '@angular/core';
import { FormBuilder, FormGroup } from '@angular/forms';
import { TranslateService } from '@ngx-translate/core';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

@Component({
  selector: 'app-admin-filters',
  templateUrl: './admin-filters.component.html',
  styleUrls: ['./admin-filters.component.scss']
})
export class AdminFiltersComponent implements OnInit, OnDestroy {
  @Input() public filters: Filters;

  /**
   * Whether the filters include application filter
   */
  @Input() public withApplicationFilter = true;

  @Output() public readonly filtersChange = new EventEmitter<Filters>();

  public readonly DateFilter = DateFilter;
  public readonly ApplicationFilter = ApplicationFilter;
  public filtersForm: FormGroup;

  private onDestroySubject$ = new Subject<void>();

  constructor(
    private adminApplicationContextService: AdminApplicationContextService,
    private formBuilder: FormBuilder,
    private translateService: TranslateService
  ) {}

  // -- LIFE CYCLE --

  public ngOnInit(): void {
    this.initFiltersForm();
    this.listenFiltersChanges();
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  // -- UTILS --

  public getDateFilterLabel(dateFilter: DateFilter): string {
    return this.translateService.instant(`admin.filters.date.${dateFilter}`);
  }

  public getApplicationFilterLabel(applicationFilter: ApplicationFilter): string {
    if (applicationFilter === ApplicationFilter.Recruter)
      return this.adminApplicationContextService.managedApplication.name;

    return this.translateService.instant(`admin.filters.application.${applicationFilter}`);
  }

  // -- PRIVATE --

  private initFiltersForm(): void {
    this.filtersForm = this.formBuilder.group({
      date: this.filters.date,
      application: this.filters.application
    });
  }

  private listenFiltersChanges(): void {
    this.filtersForm.valueChanges
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe((filters: Filters) => {
        this.filtersChange.emit(this.filters.merge(filters));
      });
  }
}
