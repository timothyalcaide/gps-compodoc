import {
  FileOrArrayFile,
  MaxSizeValidator,
  NgxMatFileInputComponent
} from '@angular-material-components/file-input';
import {
  Component,
  EventEmitter,
  Input,
  OnDestroy,
  OnInit,
  Output,
  ViewChild
} from '@angular/core';
import { FormBuilder, FormGroup } from '@angular/forms';
import { AuthService } from '@core/services';
import { AdminApplicationService } from '@services/admin-application.service';
import { BehaviorSubject, Subject } from 'rxjs';
import { filter, mergeMap, takeUntil, tap } from 'rxjs/operators';

@Component({
  selector: 'app-admin-image-form',
  templateUrl: './admin-image-form.component.html',
  styleUrls: ['./admin-image-form.component.scss']
})
export class AdminImageFormComponent implements OnInit, OnDestroy {
  @Input() public images: string[];
  @Output() public readonly delete = new EventEmitter<string>();
  @Output() public readonly imageUploaded = new EventEmitter<string>();
  @Output() public readonly imageUploadFailed = new EventEmitter<void>();
  @Output() public readonly imageDeleted = new EventEmitter<number>();
  @Output() public readonly imageDeleteFailed = new EventEmitter<void>();
  @ViewChild(NgxMatFileInputComponent) public fileInput: NgxMatFileInputComponent;

  public formImage: FormGroup;
  public readonly MAX_SIZE = 1048576; // = 1Mb

  public loading$ = new BehaviorSubject<boolean>(false);

  private onDestroySubject$ = new Subject<void>();

  constructor(
    private adminApplicationService: AdminApplicationService,
    private formBuilder: FormBuilder,
    private authService: AuthService
  ) {}

  public ngOnInit(): void {
    this.formImage = this.formBuilder.group({
      image: [null, MaxSizeValidator(this.MAX_SIZE)]
    });

    this.listenImageChange();
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public onAddImage(): void {
    this.fileInput.openFilePicker();
  }

  public onDeleteImage(image: string, index: number): void {
    this.adminApplicationService
      .deleteImage(this.authService.user, image)
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe({
        next: () => this.imageDeleted.emit(index),
        error: () => this.imageDeleteFailed.emit()
      });
  }

  private listenImageChange(): void {
    this.formImage.controls.image.valueChanges
      .pipe(
        filter(
          (selectedImage: FileOrArrayFile) => !!selectedImage && this.formImage.controls.image.valid
        ),
        tap(() => this.loading$.next(true)),
        mergeMap((selectedImage: FileOrArrayFile) =>
          this.adminApplicationService.uploadImages(this.authService.user, selectedImage as Blob)
        ),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe({
        next: (imagePath: string) => {
          this.formImage.controls.image.setValue(null);

          this.imageUploaded.emit(imagePath);

          this.loading$.next(false);
        },
        error: () => {
          this.imageUploadFailed.emit();

          this.loading$.next(false);
        }
      });
  }
}
