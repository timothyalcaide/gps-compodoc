import { Filters } from '@admin-models/filters/filters';
import { Component, Input, OnChanges, OnInit, ViewChild } from '@angular/core';
import { MatPaginator } from '@angular/material/paginator';
import { MatTableDataSource } from '@angular/material/table';
import { Status } from '@clients/adventure/bean/situation';
import { statusKeyLabel } from '@clients/adventure/bean/situation.helper';
import { AuthService } from '@core/services';
import { Group } from '@models/group/group';
import { LeadBackOffice } from '@models/lead/lead-back-office';
import { Situation } from '@models/situation/situation';
import { OriginAndDestination } from '@models/waypoint/origin-with-destination';
import { AdminLeadService } from '@services/admin-lead.service';
import { map, Observable } from 'rxjs';

@Component({
  selector: 'app-admin-leads-table',
  templateUrl: './admin-leads-table.component.html',
  styleUrls: ['./admin-leads-table.component.scss']
})
export class AdminLeadsTableComponent implements OnInit, OnChanges {
  @Input() public filters: Filters;
  @ViewChild(MatPaginator) public paginator: MatPaginator;

  public readonly statusKeyLabel = statusKeyLabel;

  public readonly displayedColumns: string[] = [
    'createdDate',
    'firstName',
    'lastName',
    'email',
    'status',
    'level',
    'specialities',
    'waypoint',
    'waypoints',
    'type'
  ];

  public filteredLeads$: Observable<MatTableDataSource<LeadBackOffice>>;

  // -- SECURITY --

  public get hasAcquisitionOrMyRoadAdvanced(): boolean {
    return (
      this.authService.user.hasAuthorizedGroup([Group.MyRoadAdvanced]) ||
      this.authService.user.hasAuthorizedGroup([Group.Acquisition])
    );
  }

  constructor(private adminLeadService: AdminLeadService, private authService: AuthService) {}

  // -- LIFE CYCLE --

  public ngOnInit(): void {
    this.getLeads();
  }

  public ngOnChanges(): void {
    this.getLeads();
  }

  // -- UTILS --

  public resolveBacStatus(situation: Situation): string {
    if (situation?.status?.startsWith(Status.Bac.toString())) return situation.status;

    return '';
  }

  public adventureSummary(adv: OriginAndDestination): string {
    let summary = '';

    if (adv.origin?.label) summary += adv.origin?.label || '';

    if (adv.origin && adv.destination) summary += ' → ';

    if (adv.destination?.label) summary += adv.destination?.label || '';

    return summary;
  }

  private getLeads(): void {
    this.filteredLeads$ = this.adminLeadService.getApplicationLeads(this.filters).pipe(
      map(leads => {
        const dataSource = new MatTableDataSource<LeadBackOffice>(leads);

        dataSource.paginator = this.paginator;

        return dataSource;
      })
    );
  }
}
