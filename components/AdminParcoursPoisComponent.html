import { Filters } from '@admin-models/filters/filters';
import { AdminPoiDialogComponent } from '@admin-organisms/admin-poi-dialog/admin-poi-dialog.component';
import { Component, OnDestroy, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { Router } from '@angular/router';
import { ApplicationContext, DialogService } from '@core/services';
import { ParcoursPoi } from '@models/poi/parcours-poi/parcours-poi';
import { PoiProvider } from '@models/poi/poi-provider';
import { ConfirmationDialogComponent } from '@organisms/confirmation-dialog/confirmation-dialog.component';
import { AdminPoiService } from '@services/admin-poi.service';
import { BehaviorSubject, Subject } from 'rxjs';
import { filter, mergeMap, takeUntil, tap } from 'rxjs/operators';
import { AdminBaseComponent } from '../../abstract/admin-base.component';

@Component({
  selector: 'app-admin-parcours-pois',
  templateUrl: './admin-parcours-pois.component.html'
})
export class AdminParcoursPoisComponent extends AdminBaseComponent implements OnInit, OnDestroy {
  public columns = [
    'name',
    'description',
    'type',
    'kind',
    'nodeLabel',
    'applicationId',
    'placement',
    'actions'
  ];

  public filters: Filters = new Filters();
  public parcoursPois$: BehaviorSubject<ParcoursPoi[]> = new BehaviorSubject<ParcoursPoi[]>([]);

  private onDestroySubject$ = new Subject<void>();

  constructor(
    private adminPoiService: AdminPoiService,
    private context: ApplicationContext,
    public dialog: MatDialog,
    private dialogService: DialogService,
    protected router: Router,
    protected snackBar: MatSnackBar
  ) {
    super(router, snackBar);
  }

  // -- LYFE CYCLE --

  public ngOnInit(): void {
    this.load();
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  // -- EVENTS --

  public onGenerate(): void {
    this.adminPoiService
      .createPoisParcours(this.context.application.nodeId)
      .pipe(tap(() => this.load()))
      .subscribe({
        next: () => this.success('Pois generated'),
        error: () => this.error('Failed to generate Pois')
      });
  }

  public onAdd(): void {
    this.save('add', null);
  }

  public onEdit(parcoursPoi: ParcoursPoi): void {
    this.save('edit', parcoursPoi);
  }

  public onDelete(parcoursPoi: ParcoursPoi): void {
    this.dialog
      .open(ConfirmationDialogComponent, this.dialogService.simpleDialogConfig())
      .afterClosed()
      .pipe(
        filter(res => !!res),
        mergeMap(() => this.adminPoiService.deleteParcoursPoi(parcoursPoi)),
        tap(() => this.load())
      )
      .subscribe({
        next: () => this.success('Poi deleted'),
        error: () => this.error('Failed to delete Poi')
      });
  }

  public onSearch(text: string): void {
    this.filters = new Filters(this.filters.date, this.filters.application, text);
  }

  /* -- PRIVATE -- */

  private load(): void {
    this.adminPoiService
      .parcoursPois(this.context.application.nodeId)
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe(pois => this.parcoursPois$.next(pois));
  }

  /**
   * Add or edit a parcours POI
   *
   * @param action The action type; 'add' or 'edit'
   * @param parcoursPoi The parcours POI to add or edit
   */
  // FIXME : Factorise this method (at least) with AdminMongoPoiComponent
  private save(action: string, parcoursPoi: ParcoursPoi): void {
    this.dialog
      .open(
        AdminPoiDialogComponent,
        this.dialogService.defaultDialogConfig({
          poi: parcoursPoi,
          provider: PoiProvider.parcours
        })
      )
      .afterClosed()
      .pipe(
        filter(res => !!res),
        mergeMap(res => this.adminPoiService.saveParcoursPoi(res)),
        tap(() => this.load())
      )
      .subscribe({
        next: () => this.success(`Poi ${action}ed`),
        error: () => this.error(`Failed to ${action} Poi`)
      });
  }
}
