import { DateFilter } from '@admin-models/filters/date-filter';
import { Filters } from '@admin-models/filters/filters';
import { ListMetric } from '@admin-models/list-metric/list-metric';
import { NumericMetric } from '@admin-models/numeric-metric/numeric-metric';
import { Component, OnInit } from '@angular/core';
import { ACQUISITION_METRICS_CONFIG } from '@configs/acquisition-metrics.config';
import { ApplicationContext } from '@core/services';
import { AdminApplicationContextService } from '@services/admin-application-context.service';
import { AdminPerformanceService } from '@services/admin-performance.service';
import { Dayjs } from 'dayjs';
import { Observable } from 'rxjs';

@Component({
  templateUrl: './admin-performance.component.html',
  styleUrls: ['./admin-performance.component.scss']
})
export class AdminPerformancePageComponent implements OnInit {
  public _filters: Filters = new Filters(DateFilter.Last3Month);

  public myRoadNumericMetrics$: Observable<NumericMetric>[];
  public acquisitionNumericMetrics$: Observable<NumericMetric>[];
  public acquisitionListMetrics$: Observable<ListMetric>[];

  // -- ACCESSORS --

  public get filters(): Filters {
    return this._filters;
  }

  public set filters(filters: Filters) {
    this._filters = filters;
    this.setMetrics();
  }

  constructor(
    private context: ApplicationContext,
    private adminApplicationContextService: AdminApplicationContextService,
    private adminPerformanceService: AdminPerformanceService
  ) {}

  // -- LYFE CYCLE --

  public ngOnInit(): void {
    this.setMetrics();
  }

  // -- PRIVATE --

  private setMetrics(): void {
    const startDate = this._filters.dateFilterToDayjs();

    this.setMyRoadNumericMetrics(startDate);

    if (ACQUISITION_METRICS_CONFIG(this.context.application)) {
      this.setAcquisitionNumericMetrics(startDate);
      this.setAcquisitionListMetrics(startDate);
    }
  }

  private setMyRoadNumericMetrics(dateFilter: Dayjs): void {
    this.myRoadNumericMetrics$ = [
      this.adminPerformanceService.countLeadsForSchool(
        this.adminApplicationContextService.managedApplication.nodeId,
        dateFilter
      )
    ];
  }

  private setAcquisitionNumericMetrics(dateFilter: Dayjs): void {
    this.acquisitionNumericMetrics$ = [
      this.adminPerformanceService.countLeadsForApplication(
        this.adminApplicationContextService.managedApplication.application,
        dateFilter
      ),
      this.adminPerformanceService.countStartedInstances(
        this.adminApplicationContextService.managedApplication.application,
        dateFilter
      ),
      this.adminPerformanceService.countCompletedInstances(
        this.adminApplicationContextService.managedApplication.application,
        dateFilter
      )
    ];
  }

  private setAcquisitionListMetrics(dateFilter: Dayjs): void {
    this.acquisitionListMetrics$ = [
      this.adminPerformanceService.countOrigins(
        this.adminApplicationContextService.managedApplication.application,
        dateFilter
      )
    ];
  }
}
