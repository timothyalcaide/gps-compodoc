import { Component, Inject } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { WITH_REWARD_LOGO_CONFIG } from '@configs/with-reward-logo.config';
import { ApplicationContext } from '@core/services';
import { Reward } from '@models/reward/reward';
import { RewardType } from '@models/reward/reward-type';

@Component({
  selector: 'app-admin-reward-dialog',
  templateUrl: './admin-reward-dialog.component.html'
})
export class AdminRewardDialogComponent {
  public form: FormGroup;

  public get hasLogoInput(): boolean {
    return WITH_REWARD_LOGO_CONFIG(this.context.application);
  }

  constructor(
    private context: ApplicationContext,
    private dialog: MatDialogRef<AdminRewardDialogComponent>,
    fb: FormBuilder,
    @Inject(MAT_DIALOG_DATA) public data: Reward
  ) {
    // FIXME: name unity
    this.form = fb.group({
      application: [this.data ? this.data.application : this.context.applicationName],
      title: [this.data?.title, Validators.required],
      description: [this.data?.description],
      points: [this.data?.points, [Validators.required, Validators.min(10), Validators.max(100)]],
      email: [this.data?.email, [Validators.required, Validators.email]],
      partnerLogo: [this.data?.partnerLogo],
      enabled: [this.data?.enabled],
      type: [this.data ? this.data.type : RewardType.OTHER]
    });

    if (this.data?.id) this.form.get('title');
  }

  public onSave(): void {
    this.dialog.close({ ...this.data, ...(this.form.getRawValue() as Reward) });
  }
}
