import { Filters } from '@admin-models/filters/filters';
import { AdminRewardDialogComponent } from '@admin-organisms/admin-rewards-dialog/admin-reward-dialog.component';
import { AdminBaseComponent } from '@admin/components/abstract/admin-base.component';
import { Component, OnDestroy, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { Router } from '@angular/router';
import { DialogService } from '@core/services';
import { Reward } from '@models/reward/reward';
import { ConfirmationDialogComponent } from '@organisms/confirmation-dialog/confirmation-dialog.component';
import { AdminRewardService } from '@services/admin-reward.service';
import { BehaviorSubject, Subject } from 'rxjs';
import { filter, mergeMap, takeUntil, tap } from 'rxjs/operators';

@Component({
  templateUrl: './admin-rewards.component.html'
})
export class AdminRewardsPageComponent extends AdminBaseComponent implements OnInit, OnDestroy {
  public columns = ['title', 'description', 'points', 'email', 'actions'];
  public filters: Filters = new Filters();
  public rewards$: BehaviorSubject<Reward[]> = new BehaviorSubject<Reward[]>([]);

  private onDestroySubject$ = new Subject<void>();

  constructor(
    private adminRewardService: AdminRewardService,
    private dialog: MatDialog,
    private dialogService: DialogService,
    protected router: Router,
    protected snackBar: MatSnackBar
  ) {
    super(router, snackBar);
  }

  // -- LYFE CYCLE --

  public ngOnInit(): void {
    this.load();
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  // -- EVENTS --

  public onAdd(): void {
    this.save('add', null);
  }

  public onEdit(reward: Reward): void {
    this.save('edit', reward);
  }

  public onEnable(reward: Reward): void {
    if (!reward.email) this.error('You need an email to enable this reward');
    else {
      this.adminRewardService.saveReward(reward).subscribe({
        next: () => {
          if (reward.enabled === true) this.success('Reward enabled');
          else if (reward.enabled === false) this.success('Reward disabled');
        },
        error: () => this.error('Failed to enable the Reward')
      });
    }
  }

  public onDelete(reward: Reward): void {
    this.dialog
      .open(ConfirmationDialogComponent, this.dialogService.simpleDialogConfig())
      .afterClosed()
      .pipe(
        filter(res => !!res),
        mergeMap(() => this.adminRewardService.deleteReward(reward)),
        tap(() => this.load())
      )
      .subscribe({
        next: () => this.success('Reward deleted'),
        error: () => this.error('Failed to delete Reward')
      });
  }

  public onSearch(text: string): void {
    this.filters = new Filters(this.filters.date, this.filters.application, text);
  }

  // -- PRIVATE --

  private load(): void {
    this.adminRewardService
      .rewards()
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe(rewards => this.rewards$.next(rewards));
  }

  private save(action: string, reward: Reward): void {
    this.dialog
      .open(AdminRewardDialogComponent, this.dialogService.defaultDialogConfig(reward))
      .afterClosed()
      .pipe(
        filter(res => !!res),
        mergeMap(res => this.adminRewardService.saveReward(res)),
        tap(() => this.load())
      )
      .subscribe({
        next: () => this.success(`Reward ${action}ed`),
        error: () => this.error(`Failed to ${action} Reward`)
      });
  }
}
