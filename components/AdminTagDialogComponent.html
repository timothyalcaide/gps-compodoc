import { COMMA, ENTER } from '@angular/cdk/keycodes';
import { Component, Inject } from '@angular/core';
import { FormArray, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatChipInputEvent } from '@angular/material/chips';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Tag } from '@clients/adventure/bean/tag';
import { API_TAG_FORM_CONFIG } from '@configs/api-tag-form.config';
import { ApplicationContext } from '@core/services';
import { UnicityLabel } from '@directives/custom-validators/unicity-label.directive';
import { AdminTagService } from '@services/admin-tag.service';
import { Observable } from 'rxjs';
import { tap } from 'rxjs/operators';

@Component({
  selector: 'app-admin-tag-dialog',
  templateUrl: './admin-tag-dialog.component.html'
})
export class AdminTagDialogComponent {
  public readonly CHIP_SEPARATORS = [ENTER, COMMA];
  public form: FormGroup;
  public tags$: Observable<Tag[]>;
  public tagApis$: Observable<string[]>;

  public get hasTagForm(): boolean {
    const { application, product } = this.context;

    return API_TAG_FORM_CONFIG(application, product);
  }

  constructor(
    private dialogRef: MatDialogRef<AdminTagDialogComponent>,
    private fb: FormBuilder,
    private context: ApplicationContext,
    private adminTagService: AdminTagService,
    @Inject(MAT_DIALOG_DATA) public data: Tag
  ) {
    this.tags$ = this.adminTagService.tags().pipe(
      tap(tags => {
        this.form = this.fb.group(
          {
            application: [
              this.data?.application ? this.data.application : this.context.applicationName
            ],
            label: [this.data?.label, Validators.required],
            api: [this.data?.api],
            config: this.fb.array([]),
            weight: [this.data?.weight]
          },
          { validators: [UnicityLabel(tags, 'label', this.data?.id)] }
        );
        this.resolveConfig(this.form);
      })
    );
    this.tagApis$ = this.adminTagService.tagApis();
  }

  /**
   * Save tag with existing data and property config (special case)
   */
  public onSave(): void {
    const config = {};

    this.controlsCfg.forEach(ctl => {
      config[ctl.value.key] = ctl.value.value;
    });
    this.form.removeControl('config');
    this.dialogRef.close({
      ...this.data,
      ...(this.form.getRawValue() as Tag),
      ...({ config } as Tag)
    });
  }

  /* -- CONFIG -- */

  public get controlsCfg(): FormGroup[] {
    return (this.form.get('config') as FormArray)?.controls as FormGroup[];
  }

  /**
   * Add a key/value form for the config property
   * With an array control so add a simple or multiple input
   */
  public onAddRow(array: boolean): void {
    const details = this.form.get('config') as FormArray;

    details.push(
      this.fb.group({
        key: '',
        value: array ? [[]] : '',
        array
      })
    );
  }

  /**
   * Delete a key/value form for the config property
   */
  public onDeleteRow(index): void {
    const details = this.form.get('config') as FormArray;

    details.removeAt(index);
  }

  public onAddValue(form: FormGroup, e: MatChipInputEvent): void {
    if (e.value) form.controls.value.setValue([...form.controls.value.value, e.value]);

    e.chipInput.clear();
  }

  public onRemoveValue(form: FormGroup, value: string): void {
    form.controls.value.setValue(form.controls.value.value.filter((v: string) => v !== value));
  }

  public trackByConfigValue(_: number, value: string): string {
    return value;
  }

  /**
   * Resolve config property to create the form
   * The config property can be anything,
   * so we create an input form if it's a simple property (string, number)
   * or an multiple input form (array) if t's an array property
   */
  private resolveConfig(form: FormGroup): void {
    if (!this.data) return;

    const details = form.get('config') as FormArray;

    Object.entries(this.data.config || {}).forEach(v => {
      if (Array.isArray(v[1]))
        details.push(this.fb.group({ key: v[0], value: this.fb.array(v[1]), array: true }));
      else details.push(this.fb.group({ key: v[0], value: v[1], array: false }));
    });
  }
}
