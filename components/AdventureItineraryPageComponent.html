import { AfterViewInit, Component, OnInit, ViewChild } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { GenerateRoadmapDirective } from '@directives/generate-roadmap/generate-roadmap.directive';
import { NetflixLoaderDirective } from '@directives/netflix-loader/netflix-loader.directive';
import { Instance } from '@models/instance/instance';
import { ItineraryStep } from '@models/itinerary-step/itinerary-step';
import { ItineraryService } from '@services/itinerary.service';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';

@Component({
  selector: 'app-page-adventure-itinerary',
  templateUrl: './adventure-itinerary.component.html',
  styleUrls: ['./adventure-itinerary.component.scss']
})
export class AdventureItineraryPageComponent implements OnInit, AfterViewInit {
  @ViewChild(NetflixLoaderDirective)
  public generateRoadmapButtonNetflixLoader: NetflixLoaderDirective;

  @ViewChild(GenerateRoadmapDirective) public generateRoadmapDirective: GenerateRoadmapDirective;

  public readonly AppRouteUrls = AppRouteUrls;
  public instance$: Observable<Instance>;

  constructor(public itinerary: ItineraryService, private route: ActivatedRoute) {}

  public ngOnInit(): void {
    // Load instance from AdventureResolver data
    this.instance$ = this.route.data.pipe(map(data => data.instance as Instance));
  }

  public ngAfterViewInit(): void {
    this.generateRoadmapButtonNetflixLoader.start();
  }

  public onGenerateRoadmap(): void {
    this.generateRoadmapDirective.onGenerate();
  }

  public getItinerarySteps(instance: Instance): ItineraryStep[] {
    return this.itinerary.getItinerarySteps(
      instance.jobOrDiplomaWps,
      true,
      true,
      instance.situation
    );
  }
}

<ng-container *ngIf="instance$ | async as instance">
  <app-simple-template
