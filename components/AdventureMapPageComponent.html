import { DOCUMENT } from '@angular/common';
import { AfterViewInit, Component, Inject, OnDestroy, OnInit, ViewChild } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { Router, UrlTree } from '@angular/router';
import { Waypoint } from '@clients/adventure/bean/waypoint';
import { Poi } from '@clients/poi/poi';
import { TRENDS_REGION_CONFIG } from '@configs/trends-region.config';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import {
  AdventureService,
  AnalyticsService,
  ApplicationContext,
  BreakpointService,
  DialogService,
  SituationContext
} from '@core/services';
import { MapSvgComponent } from '@features/map/components/map-svg/map-svg.component';
import { District, Village } from '@features/map/models';
import { MapService } from '@features/map/services/map.service';
import { AdventureType } from '@models/adventure-type/adventure-type';
import { ExitMapAnalyticsEvent } from '@models/analytics-event/exit-map.analytics-event';
import { MapVillageOpenAnalyticsEvent } from '@models/analytics-event/map-village-open.analytics-event';
import { Product } from '@models/context/product';
import { DefinitionType } from '@models/definition/definition';
import { Filters } from '@models/filters/filters';
import { WaypointType } from '@models/waypoint/waypoint-type';
import { TranslateService } from '@ngx-translate/core';
import { ConfirmationDialogComponent } from '@organisms/confirmation-dialog/confirmation-dialog.component';
import { InstanceService } from '@services/instance.service';
import { MapTemplateComponent } from '@shared/components/templates/map-template/map-template.component';
import { environment } from 'environments/environment';
import { BehaviorSubject, Observable, Subject } from 'rxjs';
import { filter, map, takeUntil, tap } from 'rxjs/operators';

@Component({
  selector: 'app-page-adventure-map',
  templateUrl: './adventure-map.component.html',
  styleUrls: ['./adventure-map.component.scss']
})
export class AdventureMapPageComponent implements OnInit, AfterViewInit, OnDestroy {
  @ViewChild(MapTemplateComponent, { static: true })
  public mapTemplateComponent: MapTemplateComponent;

  @ViewChild(MapSvgComponent, { static: true }) public mapSvgComponent: MapSvgComponent;

  /**
   * The root district of the global map
   */
  public initialDistrict: District;

  // Village map
  public village: Village;

  /**
   * The village index within its sibling list
   */
  public villageIndex = 0;

  public focusedPoi$ = new BehaviorSubject<Poi>(null);
  public siblings: Waypoint[] = [];
  public readonly filters = new Filters();
  public adventureType: AdventureType;

  private onDestroySubject$: Subject<void> = new Subject<void>();

  private get exitUrlTree(): UrlTree {
    const { definition } = this.adventureService.instance;

    if (this.adventureType === AdventureType.Internship)
      return this.router.parseUrl(`/${AppRouteUrls.INTENTIONS}`);

    if (definition.type === DefinitionType.FROM_USER && definition.destination) {
      return this.router.parseUrl(
        `/${AppRouteUrls.INTENTIONS_JOB_ORIGINS(definition.destination.id)}`
      );
    }

    if (definition.type === DefinitionType.FROM_USER && !definition.destination) {
      return this.router.parseUrl(
        `/${AppRouteUrls.INTENTIONS_WAYPOINT_DETAILS(WaypointType.diploma, definition.origin.id)}`
      );
    }

    if (definition.type !== DefinitionType.FROM_USER && definition.destination) {
      return this.router.parseUrl(
        `/${AppRouteUrls.INTENTIONS_WAYPOINT_DEFINITIONS(AdventureType.Job)}`
      );
    }

    return this.router.parseUrl(
      `/${AppRouteUrls.INTENTIONS_WAYPOINT_DEFINITIONS(AdventureType.Diploma)}`
    );
  }

  constructor(
    @Inject(DOCUMENT) private document: Document,
    private adventureService: AdventureService,
    private analyticsService: AnalyticsService,
    private breakpointService: BreakpointService,
    private context: ApplicationContext,
    private dialog: MatDialog,
    private dialogService: DialogService,
    private instanceService: InstanceService,
    private mapService: MapService,
    private router: Router,
    private translateService: TranslateService,
    private situationContext: SituationContext
  ) {}

  public ngOnInit(): void {
    this.adventureType = this.adventureService.instance.adventureType;
    this.initialDistrict = this.mapService.firstDistrict(this.mapSvgComponent.svg.nativeElement);

    const lastSelectedVillage: Village = this.mapService.initMapElements(
      this.adventureService.instance?.waypoints,
      this.initialDistrict
    );

    this.filters.trendsRegion = TRENDS_REGION_CONFIG(this.context.application);

    if (this.adventureType === AdventureType.Internship) this.filters.isInternship = true;

    if (this.adventureType === AdventureType.Internship && !lastSelectedVillage.district.parent) {
      this.village = lastSelectedVillage;
      this.onVillageMove();
    } else {
      this.mapSvgComponent.onVillageIn(lastSelectedVillage);
    }
  }

  public ngAfterViewInit(): void {
    // Listen focused poi changes to scroll back to the top of the card list in desktop
    this.focusedPoi$.pipe(takeUntil(this.onDestroySubject$)).subscribe(() => {
      if (this.breakpointService.snapshot.isDesktop)
        this.mapTemplateComponent.scrollToTopOfCardList();
    });
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public canDeactivate(): Observable<boolean | UrlTree> {
    return this.openExitConfirmationDialog().pipe(
      map(res => {
        if (!res) return false;

        return this.exitUrlTree;
      })
    );
  }

  /**
   * Get into a village
   */
  public onVillageIn(village: Village): void {
    this.village = village;
    this.village.selected = false;

    if (this.adventureType === AdventureType.Internship)
      this.mapTemplateComponent.onOpenVillageDetails();

    if (this.adventureType !== AdventureType.Internship) this.loadSiblings();

    // Preload the expand following the new focused village so the request is cached on the API
    this.instanceService
      .expand(
        this.village.expand,
        this.village.stage,
        this.adventureService.instance.situation.gender,
        this.filters
      )
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe();
  }

  public onCloseVillageDetails(): void {
    if (this.adventureType === AdventureType.Internship && this.village?.district.parent) {
      this.mapSvgComponent.onVillageOut();
    }
  }

  /**
   * Move to the next step
   */
  public onVillageMove(): void {
    if (!this.village.waypoint.final) {
      // adds the new expansion

      this.instanceService
        .expand(
          this.village.expand,
          this.village.stage,
          this.adventureService.instance.situation.gender,
          this.filters
        )
        .pipe(
          map(expand =>
            this.mapService.expansion(
              expand,
              this.village,
              this.translateService.instant('sign.diplomas'),
              this.translateService.instant('sign.jobs')
            )
          ),
          takeUntil(this.onDestroySubject$)
        )
        .subscribe(followingElement => {
          if (!followingElement) {
            if (!environment.production) console.error('No more records');

            this.onRoadmap();
          } else {
            this.village = null;
            this.villageIndex = null;
            this.siblings = [];
            this.mapSvgComponent.moveToNextStep(followingElement);
          }
        });
    }
  }

  /**
   * Get out of the current village
   */
  public onVillageOut(): void {
    this.adventureService.popFromInstance();

    this.village = null;
    this.siblings = [];
    this.focusedPoi$.next(null);
  }

  public onRoadmap(): void {
    if (this.context.product === Product.Show) this.situationContext.empty();

    this.router.navigateByUrl(
      `/${AppRouteUrls.ADVENTURE_ITINERARY(this.adventureService.instance.id)}`
    );
  }

  public onExit(): void {
    if (this.context.product === Product.Show) this.situationContext.empty();

    this.openExitConfirmationDialog()
      .pipe(
        filter(res => !!res),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe(() => {
        this.analyticsService.pushTag(
          new ExitMapAnalyticsEvent(
            this.village?.waypoint.type || 'global-map',
            !!this.village?.waypoint.final
          )
        );

        this.router.navigateByUrl(this.exitUrlTree).then(() => {
          this.adventureService.instance = null;
        });
      });
  }

  /**
   * Pick a sibling village
   * Scroll the sibling list to the top
   *
   * @param waypoint The waypoint of the village to pick
   */
  public onPickVillage(waypoint: Waypoint): void {
    if (this.breakpointService.snapshot.isDesktop)
      this.mapTemplateComponent.scrollToTopOfCardList();

    this.villageIndex = this.siblings.findIndex(wp => waypoint.id === wp.id);
    this.onVillageSwap(waypoint);
  }

  public onChooseVillage(): void {
    this.village.selected = true;
  }

  /**
   * Initialize current village siblings
   */
  private loadSiblings(): void {
    this.villageIndex = 0;

    if (this.adventureService.instance.definition.destination || this.village.stage) {
      this.instanceService
        .siblings(
          this.village.expand,
          this.village.stage,
          this.adventureService.instance.situation.gender,
          this.filters
        )
        .pipe(
          tap(wps => {
            this.villageIndex = wps.map(waypoint => waypoint.id).indexOf(this.village?.waypoint.id);
          }),
          takeUntil(this.onDestroySubject$)
        )
        .subscribe(siblings => {
          this.siblings = siblings;
        });
    } else this.siblings = [];
  }

  /**
   * Move to a sibling village
   * If the village does not exist yet, create it
   *
   * @param waypoint The sibling village waypoint
   */
  private onVillageSwap(waypoint: Waypoint): void {
    this.analyticsService.pushTag(new MapVillageOpenAnalyticsEvent(waypoint.label, waypoint.type));

    this.focusedPoi$.next(null);

    this.mapTemplateComponent.onCloseVillageDetails();
    this.village.selected = false;

    this.adventureService.popFromInstance();
    this.adventureService.pushToInstance(waypoint, this.village.stage);

    const village: Village = this.village.district.villages.find(
      v => v.waypoint.id === waypoint.id
    );

    this.village = village || new Village(this.document, waypoint, this.village.district);

    // Preload the expand following the new focused village so the request is cached on the API
    this.instanceService
      .expand(
        this.village.expand,
        this.village.stage,
        this.adventureService.instance.situation.gender,
        this.filters
      )
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe();
  }

  private openExitConfirmationDialog(): Observable<boolean | undefined> {
    return this.dialog
      .open(
        ConfirmationDialogComponent,
        this.dialogService.simpleDialogConfig({
          titleKey: 'adventure.exit',
          confirmButtonLabelKey: 'exit'
        })
      )
      .afterClosed();
  }
}
