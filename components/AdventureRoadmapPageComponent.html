import { Component, OnDestroy, OnInit } from '@angular/core';

import {
  AdventureService,
  AnalyticsService,
  ApplicationContext,

  PopupApiService,

  WaypointService
} from '@core/services';

@Component({
  selector: 'app-page-adventure-roadmap',
  templateUrl: './adventure-roadmap.component.html',
  styleUrls: ['./adventure-roadmap.component.scss']
})
export class AdventureRoadmapPageComponent implements OnInit, OnDestroy {
  public readonly TabId = TabId;
  public readonly AppRouteUrls = AppRouteUrls;
  public readonly isDesktop$: Observable<boolean>;
  public readonly isMobile$: Observable<boolean>;
  public WITH_SAVE_ROADMAP_BUTTON: boolean;
  public IS_PAYS_DE_LA_LOIRE: boolean;
  public instance: Instance;
  public isLastWpJob = false;
  public isLastWpInOffer = false;
  public IS_REGION_SUD: boolean;
  public applicationPreviewType: string;

  // Datas
  public skillsAndWaypointMoocs$: Observable<{ skills: Skill[]; waypointMoocs: Mooc[] }>;
  public eventPois$: Observable<EventPoi[]>;
  public alumniPois$: Observable<HumanPoi[]>;
  public studentPois$: Observable<HumanPoi[]>;
  public schools$: Observable<School[]>;
  public structures$: Observable<Structure[]>;
  public companies$: Observable<Company[]>;
  public recommendedBacSpecialties$: Observable<BacSpecialtiesGroup[]>;
  public jobKpis$: Observable<Kpi[]>;
  public jobVideoUrls$: Observable<string[]>;
  public scoreDetails$: Observable<Map<string, Poi[]>>;
  public rewardsFuture$: Observable<Reward[]>;
  public rewardsWon$: Observable<Reward[]>;
  public similarDefinitions$: Observable<Definition[]>;

  private onDestroySubject$ = new Subject<void>();

  public get application(): Application {
    return this.context.application;
  }

  public get product(): Product {
    return this.context.product;
  }

  public get hasDiscoveries(): boolean {
    return GAMIFICATION_ENABLED_CONFIG(this.context.product, this.context.applicationName);
  }

  public get firstWaypointInOffer(): boolean {
    return IN_OFFER_CONFIG(this.context.application, 0, this.context.product);
  }

  constructor(
    private context: ApplicationContext,
    private adventureService: AdventureService,
    private analyticsService: AnalyticsService,
    private breakpointService: BreakpointService,
    private definitionService: DefinitionService,
    private dialog: MatDialog,
    private dialogService: DialogService,
    private hroadsService: HroadsService,
    private instanceService: InstanceService,
    private loaderService: LoaderService,
    private poiService: PoiService,
    private popupApiService: PopupApiService,
    private rewardService: RewardService,
    private route: ActivatedRoute,
    private router: Router,
    private situationService: SituationService,
    private videoAskService: VideoAskService,
    private waypointService: WaypointService
  ) {
    this.isDesktop$ = this.breakpointService.isDesktop$;
    this.isMobile$ = this.breakpointService.isMobile$;
  }

  public ngOnInit(): void {
    // Remove instance from context as the adventure is over so the user can start again an instance from the Discover menu entry
    this.adventureService.instance = null;
    this.WITH_SAVE_ROADMAP_BUTTON = SAVE_ROADMAP_ENABLED_CONFIG(this.application, this.product);
    this.IS_PAYS_DE_LA_LOIRE = this.context.applicationName === ApplicationName.PaysDeLaLoire;
    this.IS_REGION_SUD = this.context.applicationName === ApplicationName.RegionSud;
    this.applicationPreviewType = APPLICATION_PREVIEW_TYPE_CONFIG(this.application);
    const filters: Filters = new Filters().context(this.context.application);

    if (STRUCTURES_ENABLED_CONFIG(this.context.application)) {
      this.structures$ = this.loaderService.globalObservableLoadingSession<Structure[]>(
        this.waypointService.structures(this.context.application.nodeId).pipe(share())
      );
    }

    this.route.data
      .pipe(
        map(data => data.instance),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe((instance: Instance) => {
        this.instance = instance;
        this.isLastWpJob = isJob(instance.lastWp.type);
        this.isLastWpInOffer = IN_OFFER_CONFIG(
          this.context.application,
          instance.waypoints.length - 1,
          this.context.product
        );

        // Load datas
        this.skillsAndWaypointMoocs$ = this.loaderService.globalObservableLoadingSession<{
          skills: Skill[];
          waypointMoocs: Mooc[];
        }>(
          forkJoin({
            skills: this.poiService.skills(instance.lastWp).pipe(share()),
            waypointMoocs: this.poiService.moocs(instance.lastWp).pipe(share())
          })
        );

        this.eventPois$ = this.loaderService.globalObservableLoadingSession<EventPoi[]>(
          this.poiService.events(instance.firstWp, filters).pipe(share())
        );

        this.alumniPois$ = this.loaderService.globalObservableLoadingSession<HumanPoi[]>(
          this.instanceService.alumniPois(null, null, instance).pipe(
            map(alumniPois =>
              // FIXME : Remove when VideoAsk is ready
              this.videoAskService.getMockedHumanPoisWithVideoAsks(
                alumniPois,
                instance.lastWp.id,
                true
              )
            ),
            share()
          )
        );

        // FIXME: Hide until this project is started -> https://www.notion.so/millionroads/Bug-Dans-l-quipe-Anciens-l-ves-dans-les-tudiants-2189d68535ec409eaf7937a5c636a4c3
        // if (isDiploma(instance.lastWp.type)) {
        //   this.studentPois$ = this.loaderService.globalObservableLoadingSession<HumanPoi[]>(
        //     this.instanceService.studentPois(null, null, instance).pipe(share())
        //   );
        // }

        this.companies$ = this.loaderService.globalObservableLoadingSession<Company[]>(
          this.waypointService.companies(instance.lastWp, this.application).pipe(share())
        );

        if (SCHOOLS_ENABLED_CONFIG(this.application, this.product)) {
          this.schools$ = this.loaderService.globalObservableLoadingSession<School[]>(
            this.waypointService.schools(instance.firstWp, this.context.application).pipe(share())
          );
        }

        if (ON_BOARDING_ENABLED_CONFIG(this.context.applicationName)) {
          this.similarDefinitions$ = this.loaderService.globalObservableLoadingSession<
            Definition[]
          >(
            this.definitionService.similarDefinitions(
              this.instance.definition,
              this.instance.situation.gender
            )
          );
        }

        if (ApplicationName.Popup === this.context.applicationName) this.sendInformationToPopup();

        this.recommendedBacSpecialties$ = this.loaderService.globalObservableLoadingSession<
          BacSpecialtiesGroup[]
        >(this.situationService.recommendedBacSpecialties(instance.firstWp.id));

        if (this.isLastWpJob) {
          this.jobKpis$ = forkJoin(
            [
              HroadsService.LAST_STUDY_BEFORE,
              HroadsService.JOB_COMPANY,
              HroadsService.COMPANY_DOMAIN
            ].map(kpi =>
              this.hroadsService
                .kpi(this.instance.lastWp, kpi.id, HroadsService.KPI_LIMIT)
                .pipe(map(kpiItem => new Kpi(kpi.id, kpiItem, kpi.icon)))
            )
          );

          this.jobVideoUrls$ = EXTERNAL_LINKS_CONFIG(this.context.product)
            ? this.loaderService.globalObservableLoadingSession<string[]>(
                this.poiService.videos(this.instance.lastWp)
              )
            : of([]);
        }

        this.rewardsFuture$ = this.rewardService.rewardsFuture();
        this.rewardsWon$ = this.rewardService.rewardsWon();
        this.scoreDetails$ = this.loaderService.globalObservableLoadingSession<Map<string, Poi[]>>(
          this.poiService
            .computeScoreDetails(this.rewardService.instanceScore.winners, this.instance.id)
            .pipe(share())
        );
      });
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public onBackToMap(): void {
    this.dialog
      .open(
        ConfirmationDialogComponent,
        this.dialogService.simpleDialogConfig({
          titleKey: 'roadmap.exit',
          subtitleKey: 'roadmap.exit-subtitle',
          confirmButtonLabelKey: 'roadmap.navigate'
        })
      )
      .afterClosed()
      .pipe(
        filter((confirmed: boolean) => confirmed),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe(() => {
        this.router.navigateByUrl(`/${AppRouteUrls.ADVENTURE_MAP(this.instance.id)}`);
      });
  }

  /**
   * Track when the user changes roadmap selected tab
   *
   * @param event The tab change event
   */
  public onSelectedTabChange(event: MatTabChangeEvent): void {
    this.analyticsService.pushTag(
      new TabChangeAnalyticsEvent(
        AnalyticsEventType.RoadmapTabChange,
        `Roadmap - ${event.tab.textLabel}`
      )
    );
  }

  // -- PRIVATE --

  /**
   * Send user information to popup external API
   */
  private sendInformationToPopup(): void {
    this.popupApiService
      .sendInformation(this.instance.lastWp)
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe();
  }
}
