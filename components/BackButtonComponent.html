import { AfterViewInit, Component, EventEmitter, Input, Output } from '@angular/core';
import { MapElementComponent } from '../map-element/map-element.component';

@Component({
  // eslint-disable-next-line @angular-eslint/component-selector
  selector: '[svg-back-button]',
  templateUrl: './back-button.component.html',
  styleUrls: ['./back-button.component.scss']
})
export class BackButtonComponent extends MapElementComponent implements AfterViewInit {
  @Output() public readonly back = new EventEmitter<void>();
  @Input() public villageSvgPath: SVGPathElement = null;

  private readonly BACK_BUTTON_MOBILE_OFFSET: number = 220;
  private readonly BACK_BUTTON_TABLET_OFFSET: number = 250;
  private readonly BACK_BUTTON_DESKTOP_OFFSET: number = 55;

  public ngAfterViewInit(): void {
    // As position is depending on dimensions, we initialize it
    super.initDimensions();
    super.initPosition();
  }

  public onBack(): void {
    this.back.emit();
  }

  protected getPosition(): DOMPoint {
    let backButtonOffset: number;

    if (this.breakpointService.snapshot.isMobile) backButtonOffset = this.BACK_BUTTON_MOBILE_OFFSET;
    else if (this.breakpointService.snapshot.isTablet)
      backButtonOffset = this.BACK_BUTTON_TABLET_OFFSET;
    else backButtonOffset = this.BACK_BUTTON_DESKTOP_OFFSET;

    const villageRelativePosition: DOMPoint =
      this.villageSvgPath.getPointAtLength(backButtonOffset);

    return new DOMPoint(villageRelativePosition.x - this.width / 2, villageRelativePosition.y);
  }
}
