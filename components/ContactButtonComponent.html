import { Component, Input, OnDestroy, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { Waypoint } from '@clients/adventure/bean/waypoint';
import {
  AdventureService,
  AnalyticsService,
  ApplicationContext,
  AuthService,
  DialogService
} from '@core/services';
import { SchoolContactAnalyticsEvent } from '@models/analytics-event/school-contact.analytics-event';
import { LeadContact } from '@models/lead/lead-contact';
import { LeadContactSchool } from '@models/lead/lead-contact-school';
import { School } from '@models/school/school';
import { User } from '@models/user/user';
import { ContactDialogComponent } from '@organisms/contact-dialog/contact-dialog.component';
import { RegistrationDialogComponent } from '@organisms/registration-dialog/registration-dialog.component';
import { ApplicationService } from '@services/application.service';
import * as sha1 from 'js-sha1';
import { BehaviorSubject, Subject } from 'rxjs';
import { filter, map, takeUntil, tap } from 'rxjs/operators';

@Component({
  selector: 'app-contact-button',
  templateUrl: './contact-button.component.html',
  styleUrls: ['./contact-button.component.scss']
})
export class ContactButtonComponent implements OnInit, OnDestroy {
  /**
   * The school that the button contacts
   * If no school is provided, the button contact the admission service of the school associated to
   * the current application
   */
  @Input() public school: School = null;

  /**
   * The current waypoint from which the contact is done
   */
  @Input() public waypoint: Waypoint = null;

  /**
   * The button color
   */
  @Input() public color = '';

  /**
   * Whether the success text displayed when the school is contacted has to be centered or not
   */
  @Input() public centerSuccessText = false;

  public hasContacted$ = new BehaviorSubject<boolean>(false);

  private onDestroySubject$: Subject<void> = new Subject<void>();

  constructor(
    private applicationService: ApplicationService,
    private adventureService: AdventureService,
    private analyticsService: AnalyticsService,
    private authService: AuthService,
    private context: ApplicationContext,
    private dialog: MatDialog,
    private dialogService: DialogService
  ) {}

  public ngOnInit(): void {
    if (this.authService.hasEmail) {
      if (this.school?.id) {
        this.applicationService
          .hasContactSchool(this.authService.email, this.school?.id)
          .pipe(
            tap(hasContacted => this.hasContacted$.next(hasContacted)),
            takeUntil(this.onDestroySubject$)
          )
          .subscribe();
      } else {
        this.applicationService
          .hasContactAdmission(this.authService.email)
          .pipe(
            tap(hasContacted => this.hasContacted$.next(hasContacted)),
            takeUntil(this.onDestroySubject$)
          )
          .subscribe();
      }
    }
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  /**
   * Start the lead contact saving process
   * If the user is authenticated, it is directly achieved
   * Otherwise, ask the user for registration (or login)
   */
  public onContact(): void {
    if (this.school) {
      this.dialog
        .open(
          ContactDialogComponent,
          this.dialogService.defaultDialogConfig({
            school: this.school,
            waypointLabel: this.waypoint.label
          })
        )
        .afterClosed()
        .pipe(
          filter(dialogMessage => !!dialogMessage),
          takeUntil(this.onDestroySubject$)
        )
        .subscribe(
          ({
            user,
            messageSubject,
            messageBody,
            campusId
          }: {
            user: User;
            messageSubject: string;
            messageBody: string;
            campusId?: string;
          }) => {
            /**
             * Dialog closing does not happen if the user chooses to login
             */
            this.contactSchool(user, messageSubject, messageBody, campusId);
          }
        );
    } else {
      this.dialog
        .open(
          RegistrationDialogComponent,
          this.dialogService.defaultDialogConfig({
            titleKey: 'dialog.contact.title',
            emojis: 'üè´'
          })
        )
        .afterClosed()
        .pipe(
          filter(dialogMessage => !!dialogMessage),
          takeUntil(this.onDestroySubject$)
        )
        .subscribe(() => {
          /**
           * Dialog closing does not happen if the user chooses to login
           */
          this.contactAdmission(this.authService.email);
        });
    }
  }

  /**
   * Contact the school or one of its campus
   *
   * @param user The user who contacts the school
   * @param messageSubject The subject of the message
   * @param campusId The campus to contact
   * @param messageBody The subject of the body
   */
  private contactSchool(
    user: User,
    messageSubject: string,
    messageBody: string,
    campusId?: string
  ): void {
    this.applicationService
      .contactSchool(
        this.context.applicationName,
        user.email,
        user,
        this.waypoint.id,
        this.adventureService.instance.id,
        this.school.id,
        campusId,
        messageSubject,
        messageBody
      )
      .pipe(
        map((leadContactSchool: LeadContactSchool) => !!leadContactSchool),
        tap(hasContacted => {
          this.hasContacted$.next(hasContacted);

          if (hasContacted) {
            this.analyticsService.pushTag(
              new SchoolContactAnalyticsEvent(this.school.name, sha1(user.email))
            );
          }
        }),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe();
  }

  /**
   * Contact the admission service of the school
   *
   * @param email The user email send to the school
   */
  private contactAdmission(email: string): void {
    this.applicationService
      .contactAdmission(
        this.context.applicationName,
        email,
        this.waypoint.id,
        this.adventureService.instance?.id || ''
      )
      .pipe(
        map((leadContact: LeadContact) => !!leadContact),
        tap(hasContacted => this.hasContacted$.next(hasContacted)),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe();
  }
}
