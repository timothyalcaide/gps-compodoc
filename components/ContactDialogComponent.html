import { Component, Inject, OnDestroy, OnInit } from '@angular/core';
import { FormBuilder, FormControl, FormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Router } from '@angular/router';
import { AUTHENTICATION_ENABLED_CONFIG } from '@configs/authentication.config';
import { ApplicationContext, AuthService, LoaderService } from '@core/services';
import { NameValidator } from '@directives/custom-validators/name-validator.directive';
import { ApplicationName } from '@models/context/application-name';
import { Product } from '@models/context/product';
import { School } from '@models/school/school';
import { TranslateService } from '@ngx-translate/core';
import { UaaService } from '@services/uaa.service';
import { BehaviorSubject, Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

@Component({
  selector: 'app-contact-dialog',
  templateUrl: './contact-dialog.component.html',
  styleUrls: ['./contact-dialog.component.scss']
})
export class ContactDialogComponent implements OnInit, OnDestroy {
  public readonly MESSAGE_SUBJECTS: string[] = [
    'information',
    'inscription',
    'documentation',
    'meeting',
    'event',
    'custom'
  ];

  public readonly IS_SHOW_PRODUCT: boolean;
  public readonly AUTHENTICATION_ENABLED: boolean;
  public readonly IS_MY_ROAD: boolean;
  public readonly isLoading$ = new BehaviorSubject<boolean>(false);
  public readonly isAuthenticated = this.authService.isAuthenticated;
  public contactForm: FormGroup;
  public applicationName: string;
  public returnUrl: string;
  public isMessageSubjectSelected = false;

  private readonly onDestroySubject$: Subject<void> = new Subject<void>();

  private get signature(): string {
    if (!this.contactForm.controls.firstName.value && !this.contactForm.controls.lastName.value)
      return 'Pr√©nom Nom';

    return `${this.contactForm.controls.firstName.value} ${this.contactForm.controls.lastName.value}`;
  }

  constructor(
    @Inject(MAT_DIALOG_DATA) public readonly data: { school: School; waypointLabel: string },
    private context: ApplicationContext,
    private authService: AuthService,
    private dialog: MatDialogRef<ContactDialogComponent>,
    private formBuilder: FormBuilder,
    private loading: LoaderService,
    private router: Router,
    private translate: TranslateService,
    private uaa: UaaService
  ) {
    this.IS_SHOW_PRODUCT = this.context.product === Product.Show;
    this.IS_MY_ROAD = this.context.applicationName === ApplicationName.MyRoad;
    this.AUTHENTICATION_ENABLED = AUTHENTICATION_ENABLED_CONFIG(
      this.context.product,
      this.context.applicationName
    );
  }

  public ngOnInit(): void {
    this.applicationName = this.context.applicationName;
    this.returnUrl = this.router.routerState.snapshot.url;

    const user = this.authService.userSessionStorage;

    this.contactForm = this.formBuilder.group({
      email: [user?.email || '', [Validators.required, Validators.email]],
      firstName: [user?.firstName || '', [Validators.required, NameValidator()]],
      lastName: [user?.lastName || '', [Validators.required, NameValidator()]],
      messageSubject: ['', Validators.required],
      messageBody: ['', Validators.required],
      lEtudiantOptIn: ''
    });

    if (this.data.school.campuses.length)
      this.contactForm.addControl(
        'campus',
        new FormControl(
          this.data.school.campuses.length === 1 ? this.data.school.campuses[0].id : '',
          Validators.required
        )
      );

    this.contactForm.controls.messageSubject.valueChanges
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe(newMessageSubject => {
        this.isMessageSubjectSelected = !!newMessageSubject;

        this.setDefaultText(newMessageSubject);
      });
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public onSubmit(): void {
    const { email, firstName, lastName, lEtudiantOptIn, messageSubject, messageBody, campus } =
      this.contactForm.value;

    this.loading
      .loadingObservable(
        this.uaa.checkIfUserExistsAndGetOrCreate(email, firstName, lastName, !!lEtudiantOptIn),
        this.isLoading$
      )
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe(user => {
        if (user) {
          /**
           * When the user is successfully registered, finalize the contact request even if he is
           * not authenticated
           */
          this.authService.userSessionStorage = user;

          this.dialog.close({
            user,
            messageSubject: this.translate.instant(
              `dialog.contact.subjects.${messageSubject}.subject`
            ),
            messageBody: messageBody.replaceAll('\n', '<br/>'),
            campusId: campus
          });
        }
      });
  }

  /**
   * Set the message body to the default text corresponding to the message subject
   *
   * @param messageSubject The message subject
   */
  private setDefaultText(messageSubject: string): void {
    this.contactForm.controls.messageBody.setValue(
      this.translate.instant(`dialog.contact.subjects.${messageSubject}.default-text`, {
        waypointLabel: this.data.waypointLabel,
        signature: this.signature
      })
    );
  }
}
