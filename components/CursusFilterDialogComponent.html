import { Component, Inject, OnDestroy, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { OFFER_COST_CONFIG } from '@configs/offer-cost.config';
import { AnalyticsService, ApplicationContext } from '@core/services';
import { CostFilterApplyAnalyticsEvent } from '@models/analytics-event/cost-filter-apply.analytics-event';
import { CursusFilterApplyAnalyticsEvent } from '@models/analytics-event/cursus-filter-apply.analytics-event';
import { Exam } from '@models/exam/exam';
import { CursusFilters } from '@models/filters/cursus-filters/cursus-filters';
import { ExamService } from '@services/exam.service';
import { filter, Observable, Subject, takeUntil } from 'rxjs';

@Component({
  selector: 'app-cursus-filter-dialog',
  templateUrl: './cursus-filter-dialog.component.html'
})
export class CursusFilterDialogComponent implements OnInit, OnDestroy {
  public readonly NO_EXAMS = CursusFilters.NO_EXAMS;
  public cursusFilterForm: FormGroup;
  public exams$: Observable<Exam[]>;
  public withCostFilter: boolean;

  private destroy$: Subject<void> = new Subject<void>();

  constructor(
    @Inject(MAT_DIALOG_DATA)
    public data: {
      initialState: CursusFilters;
      withExamsFilter: boolean;
    },
    private analyticsService: AnalyticsService,
    private ctx: ApplicationContext,
    private dialog: MatDialogRef<CursusFilterDialogComponent>,
    private formBuilder: FormBuilder,
    private examService: ExamService
  ) {}

  public ngOnInit(): void {
    this.cursusFilterForm = this.formBuilder.group({
      free: this.data.initialState.free,
      maxCost: [this.data.initialState.maxCost || '', Validators.min(1)],
      isPublic: this.data.initialState.isPublic,
      requiredExams: [this.data.initialState.requiredExams]
    });

    this.withCostFilter = OFFER_COST_CONFIG(this.ctx.application);

    if (this.data.withExamsFilter) this.initExams();

    this.initValueChangeObservable();
    this.listenFreeFilterSelections();
    this.listenMaxCostFilterChanges();
  }

  public initExams(): void {
    this.exams$ = this.examService.exams();
  }

  public initValueChangeObservable(): void {
    this.cursusFilterForm.controls.requiredExams.valueChanges.subscribe(newValue => {
      const oldValue = this.cursusFilterForm.value.requiredExams;

      // If present, the no-exams value is always the first value of the index
      if (newValue.length > 1 && newValue[0] === CursusFilters.NO_EXAMS) {
        if (oldValue[0] !== CursusFilters.NO_EXAMS)
          this.cursusFilterForm.controls.requiredExams.setValue([CursusFilters.NO_EXAMS]);
        else {
          this.cursusFilterForm.controls.requiredExams.setValue(newValue.slice(1));
        }
      }
    });
  }

  public onErase(): void {
    this.cursusFilterForm.controls.free.setValue(false);
    this.cursusFilterForm.controls.maxCost.setValue('');
    this.cursusFilterForm.controls.isPublic.setValue(false);
    this.cursusFilterForm.controls.hasSandwichTrainings.setValue(false);
    this.cursusFilterForm.controls.requiredExams.setValue([]);
  }

  public onSeeResults(): void {
    const cursusFilter = new CursusFilters(
      this.cursusFilterForm.value.free,
      this.cursusFilterForm.value.maxCost,
      this.cursusFilterForm.value.isPublic,
      this.cursusFilterForm.value.requiredExams
    );

    if (cursusFilter.isPublic) {
      const cursusFilterApplyAnalyticsEvent = new CursusFilterApplyAnalyticsEvent();

      if (cursusFilter.isPublic) cursusFilterApplyAnalyticsEvent.filterIsPublic = true;

      this.analyticsService.pushTag(cursusFilterApplyAnalyticsEvent);
    }

    if (cursusFilter.free || cursusFilter.maxCost) {
      const costFilterApplyAnalyticsEvent = new CostFilterApplyAnalyticsEvent();

      costFilterApplyAnalyticsEvent.filterMaxCost = cursusFilter.free ? 0 : cursusFilter.maxCost;

      this.analyticsService.pushTag(costFilterApplyAnalyticsEvent);
    }

    this.dialog.close(cursusFilter);
  }

  public ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  /**
   * Listen to free filter selections to empty max cost
   */
  private listenFreeFilterSelections(): void {
    this.cursusFilterForm.controls.free.valueChanges
      .pipe(
        filter((selected: boolean) => selected),
        takeUntil(this.destroy$)
      )
      .subscribe(() => {
        this.cursusFilterForm.controls.maxCost.setValue('');
      });
  }

  /**
   * Listen to max cost filter changes to unselect free filter
   */
  private listenMaxCostFilterChanges(): void {
    this.cursusFilterForm.controls.maxCost.valueChanges
      .pipe(
        filter((maxCost: number | '') => !!maxCost),
        takeUntil(this.destroy$)
      )
      .subscribe(() => {
        this.cursusFilterForm.controls.free.setValue(false);
      });
  }
}
