import { Component, Input, OnInit } from '@angular/core';
import { isJob, Waypoint } from '@clients/adventure/bean/waypoint';
import { Placement } from '@clients/poi/poi';
import { EXTERNAL_LINKS_CONFIG } from '@configs/external-links.config';
import { SCHOOLS_ENABLED_CONFIG } from '@configs/schools-enabled.config';
import { AdventureService, ApplicationContext } from '@core/services';
import { Instance } from '@models/instance/instance';
import { ParcoursPoi } from '@models/poi/parcours-poi/parcours-poi';
import { PoiType } from '@models/poi/poi-type';
import { TabId } from '@models/tab-ids/tab-ids';

@Component({
  selector: 'app-cursus-full',
  templateUrl: './cursus.full.component.html',
  styleUrls: ['./cursus.full.component.scss']
})
export class CursusFullComponent implements OnInit {
  @Input() public parcoursPois: ParcoursPoi[];
  @Input() public waypoint: Waypoint;
  @Input() public inOffer: boolean;
  @Input() public stage: number;
  @Input() public smallContainer = false;

  public externalLinks: boolean;

  public get isJob(): boolean {
    return isJob(this.waypoint.type);
  }

  public get instance(): Instance {
    return this.adventureService.instance;
  }

  constructor(private adventureService: AdventureService, private context: ApplicationContext) {}

  public ngOnInit(): void {
    this.externalLinks = EXTERNAL_LINKS_CONFIG(this.context.product);

    if (this.parcoursPois) {
      this.parcoursPois = this.parcoursPois.sort((a, b) => {
        switch (a.placement) {
          case Placement.parallel:
            if (b.placement !== Placement.end) {
              return 1;
            }

            return -1;
          case Placement.part_of:
            if (b.placement === Placement.start) {
              return 1;
            }

            return -1;
          case Placement.end:
            return 1;
          default:
            return -1;
        }
      });
    }
  }

  public hasLink(poiType: PoiType): boolean {
    switch (poiType) {
      case PoiType.SandwichTrainingJob:
      case PoiType.InternshipJob:
        return true;
      case PoiType.Parcoursup:
        return SCHOOLS_ENABLED_CONFIG(this.context.application, this.context.product);
      default:
        return false;
    }
  }

  public parcoursPoiQueryParams(poiType: PoiType): { tabId: TabId } {
    switch (poiType) {
      case PoiType.Parcoursup:
        return this.stage ? { tabId: TabId.Data } : { tabId: TabId.Schools };
      case PoiType.SandwichTrainingJob:
        return { tabId: TabId.Data };
      case PoiType.InternshipJob:
        return { tabId: TabId.Data };
      default:
        return null;
    }
  }

  public parcoursPoiFragment(poiType: PoiType): string {
    switch (poiType) {
      case PoiType.Parcoursup:
        return this.stage ? 'diploma-school' : 'schools-tab-top';
      case PoiType.SandwichTrainingJob:
        return 'internship-sandwich-training';
      case PoiType.InternshipJob:
        return 'internship-sandwich-training';
      default:
        return null;
    }
  }
}
