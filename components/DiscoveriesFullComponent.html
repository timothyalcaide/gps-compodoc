import { Component, Input, OnDestroy, OnInit } from '@angular/core';
import { computeMaxScore, computeScore, ScoreType } from '@clients/adventure/bean/instance-score';
import { UserScore } from '@clients/adventure/bean/user-score';
import { Poi } from '@clients/poi/poi';
import { RewardService } from '@core/services';
import { Reward } from '@models/reward/reward';
import { Subject } from 'rxjs';

@Component({
  selector: 'app-discoveries-full',
  templateUrl: './discoveries.full.component.html',
  styleUrls: ['./discoveries.full.component.scss']
})
export class DiscoveriesFullComponent implements OnInit, OnDestroy {
  @Input() public instanceId: string;
  @Input() public scoreDetails: Map<string, Poi[]>;
  @Input() public rewardsFuture: Reward[];
  @Input() public rewardsWon: Reward[];

  public score: number;
  public maxScore: number;
  public userScore: UserScore;

  private onDestroySubject$: Subject<void> = new Subject<void>();

  constructor(private rewardService: RewardService) {}

  public ngOnInit(): void {
    this.score = computeScore(this.rewardService.instanceScore);
    this.maxScore = computeMaxScore(this.rewardService.instanceScore);
    this.userScore = this.rewardService.userScore;
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public points(nb: number): number {
    return this.rewardService.instanceScore.configuration.points[ScoreType.POI] * nb;
  }
}
