import { ChangeDetectionStrategy, Component, OnInit } from '@angular/core';
import { loading } from '@core/decorators';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { BreakpointService } from '@core/services';
import { Observable } from 'rxjs';
import { ExpeditionService } from './services';

@Component({
  templateUrl: './expedition.component.html',
  styleUrls: ['./expedition.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class ExpeditionPageComponent implements OnInit {
  @loading() public readonly loading$: Observable<boolean>;

  public readonly AppRouteUrls = AppRouteUrls;
  public readonly expeditionInstances$ = this.expeditionService.expeditionInstances$;

  constructor(public breakpoint: BreakpointService, private expeditionService: ExpeditionService) {}

  public ngOnInit(): void {
    this.expeditionService.getAllExpeditionInstances();
  }
}
