import { Component, EventEmitter, Input, OnDestroy, OnInit, Output } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { handleBacStatus } from '@clients/adventure/bean/situation.helper';
import { AnalyticsService, ApplicationContext, DialogService } from '@core/services';
import { CursusFilterApplyAnalyticsEvent } from '@models/analytics-event/cursus-filter-apply.analytics-event';
import { CursusFilters } from '@models/filters/cursus-filters/cursus-filters';
import { Filters } from '@models/filters/filters';
import { LocalisationFilters } from '@models/filters/localisation-filters/localisation-filters';
import { CursusFilterDialogComponent } from '@organisms/cursus-filter-dialog/cursus-filter-dialog.component';
import { LocalisationFilterDialogComponent } from '@organisms/localisation-filter-dialog/localisation-filter-dialog.component';
import { Subject } from 'rxjs';
import { takeUntil, tap } from 'rxjs/operators';

@Component({
  selector: 'app-filters',
  templateUrl: './filters.component.html',
  styleUrls: ['./filters.component.scss']
})
export class FiltersComponent implements OnDestroy, OnInit {
  @Input() public withSchoolFilter = true;
  @Input() public withExamsFilter = false;
  @Input() public withSituationFilter = false;
  @Input() public filters: Filters;
  @Output() public readonly filter = new EventEmitter<Filters>();
  public icon: string;

  private onDestroySubject$: Subject<void> = new Subject<void>();

  constructor(
    private analytics: AnalyticsService,
    private context: ApplicationContext,
    private dialog: MatDialog,
    private dialogService: DialogService
  ) {}

  public ngOnInit(): void {
    this.filters = this.filters.context(this.context.application);

    if (this.withSituationFilter) {
      const { gender, status } = this.filters.situationFilters.situation;

      this.icon = `${handleBacStatus(status)}-${gender.toString()}`;
    }

    this.filter.emit(this.filters);
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public onSandwichFilter(): void {
    this.filters.sandwichFilters.hasSandwichTrainings =
      !this.filters.sandwichFilters.hasSandwichTrainings;

    if (this.filters.sandwichFilters.hasSandwichTrainings) {
      const cursusFilterApplyAnalyticsEvent = new CursusFilterApplyAnalyticsEvent();

      cursusFilterApplyAnalyticsEvent.filterHasSandwichTraining = true;

      this.analytics.pushTag(cursusFilterApplyAnalyticsEvent);
    }

    this.filter.emit(this.filters);
  }

  public onCursusFilter(): void {
    this.dialog
      .open(
        CursusFilterDialogComponent,
        this.dialogService.defaultDialogConfig({
          initialState: this.filters.cursusFilters,
          withExamsFilter: this.withExamsFilter
        })
      )
      .afterClosed()
      .pipe(
        tap((cursusFilters: CursusFilters | '' | undefined) => {
          /**
           * On dialog close from close button, it is an empty string that is received. It is ignored
           * On dialog close from click out, it is an undefined value that is received. It is ignored
           */
          if (cursusFilters !== '' && cursusFilters !== undefined) {
            this.filters.cursusFilters.copy(cursusFilters);

            this.filter.emit(this.filters);
          }
        }),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe();
  }

  public onLocalisationFilter(): void {
    this.dialog
      .open(
        LocalisationFilterDialogComponent,
        this.dialogService.defaultDialogConfig(this.filters.localisationFilters)
      )
      .afterClosed()
      .pipe(
        tap((localisationFilters: LocalisationFilters | '') => {
          /**
           * On dialog close from close button, it is an empty string that is received. It is ignored
           * On dialog close from click out, it is an undefined value that is received. It is ignored
           */
          if (localisationFilters !== '' && localisationFilters !== undefined) {
            this.filters.localisationFilters.copy(localisationFilters);

            this.filter.emit(this.filters);
          }
        }),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe();
  }

  public onSituationFilter(): void {
    this.filters.situationFilters.useSituation = !this.filters.situationFilters.useSituation;
    this.filter.emit(this.filters);
  }
}
