import { Component, Input, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { EXTERNAL_LINKS_CONFIG } from '@configs/external-links.config';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { RouteIds } from '@core/routing/route-ids';
import { ApplicationContext } from '@core/services';
import { ApplicationName } from '@models/context/application-name';

@Component({
  selector: 'app-footer',
  templateUrl: './footer.component.html',
  styleUrls: ['./footer.component.scss']
})
export class FooterComponent implements OnInit {
  @Input() public simpleLayout = true;

  public readonly AppRouteUrls = AppRouteUrls;
  public IN_NATIVE_APP: boolean;
  public STORE_LINKS: boolean;
  public EXTERNAL_LINKS: boolean;

  public get isHelpActive(): boolean {
    return this.route.snapshot.data.routeId === RouteIds.HELP;
  }

  constructor(private context: ApplicationContext, private route: ActivatedRoute) {}

  public ngOnInit(): void {
    this.IN_NATIVE_APP = this.context.inNativeApp;
    this.EXTERNAL_LINKS = EXTERNAL_LINKS_CONFIG(this.context.product);
    this.STORE_LINKS =
      this.EXTERNAL_LINKS && this.context.applicationName === ApplicationName.MyRoad;
  }
}
