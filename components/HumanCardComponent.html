import { ChangeDetectionStrategy, Component, Input, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { Step } from '@clients/poi/poi';
import { DialogService } from '@core/services';
import { IHuman, MediaTypeEnum } from '@models/company/company';
import { Gender } from '@models/gender/gender';
import { HumanDetailsDialogComponent } from '@organisms/human-details-dialog/human-details-dialog.component';
import { Utils } from '@features/map/utils';

export type GenderIconType =
  | 'man-poi-avatar-1'
  | 'man-poi-avatar-2'
  | 'man-poi-avatar-3'
  | 'man-poi-avatar-4'
  | 'man-poi-avatar-5'
  | 'lady-poi-avatar-1'
  | 'lady-poi-avatar-2'
  | 'lady-poi-avatar-3'
  | 'lady-poi-avatar-4'
  | 'lady-poi-avatar-5';

@Component({
  selector: 'app-human-card',
  templateUrl: './human-card.component.html',
  styleUrls: ['./human-card.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class HumanCardComponent implements OnInit {
  @Input() public human: IHuman;

  // TODO : companyIcon

  public currentStep: Step;
  public genderedIcon: GenderIconType;
  public videoAskId: string;

  constructor(private dialog: MatDialog, private dialogService: DialogService) {}

  public ngOnInit(): void {
    const { steps, medias } = this.human;

    this.currentStep = steps[steps.length - 1];

    if (medias[0].type === MediaTypeEnum.VIDEOASK) {
      this.videoAskId = medias[0].urlId;
    }

    if (!this.videoAskId) {
      this.genderedIcon = `${
        this.human.gender === Gender.lady ? Gender.lady : Gender.man
      }-poi-avatar-${Utils.randomBetween(1, 5)}` as GenderIconType;
    }
  }

  public onOpenUserDetails(): void {
    this.dialog.open(
      HumanDetailsDialogComponent,
      this.dialogService.defaultDialogConfig({
        human: this.human
      })
    );
  }
}
