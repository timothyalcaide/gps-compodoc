import { Component, ElementRef, Inject, OnInit, ViewChild } from '@angular/core';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { DomSanitizer, SafeResourceUrl } from '@angular/platform-browser';
import { Status } from '@clients/adventure/bean/situation';
import { situationIcon } from '@clients/adventure/bean/situation.helper';
import { isDiploma, isJob } from '@clients/adventure/bean/waypoint';
import { Step } from '@clients/poi/poi';
import { EXTERNAL_LINKS_CONFIG } from '@configs/external-links.config';
import { NO_PROFILE_SCHOOL_CONFIG } from '@configs/no-profile-school.config';
import { SCHOOLS_ENABLED_CONFIG } from '@configs/schools-enabled.config';
import { WITH_SOCIAL_NETWORK_LINK_CONFIG } from '@configs/with-social-network-link.config';
import {
  AnalyticsService,
  ApplicationContext,
  DialogService,
  WaypointService
} from '@core/services';
import { LinkedInLinkClickAnalyticsEvent } from '@models/analytics-event/linked-in-link-click.analytics-event';
import { Gender } from '@models/gender/gender';
import { Instance } from '@models/instance/instance';
import { HumanPoi } from '@models/poi/human-poi/human-poi';
import { School } from '@models/school/school';
import { TranslateService } from '@ngx-translate/core';
import { TestimonyDialogComponent } from '@organisms/testimony-dialog/testimony-dialog.component';
import { InstanceService } from '@services/instance.service';
import { VideoAskService } from '@services/video-ask/video-ask.service';
import { Observable } from 'rxjs';
import { map, tap } from 'rxjs/operators';

@Component({
  selector: 'app-human-profile-dialog',
  templateUrl: './human-profile-dialog.component.html',
  styleUrls: ['./human-profile-dialog.component.scss']
})
export class HumanProfileDialogComponent implements OnInit {
  @ViewChild('dialogContent') private dialogContent: ElementRef<HTMLDivElement>;

  public readonly isDiploma = isDiploma;
  public readonly isJob = isJob;
  public externalLinks: boolean;
  public withSocialNetworkLink: boolean;
  public videoAskUrl: SafeResourceUrl;
  public humanPoi$: Observable<HumanPoi>;
  public schools$: Observable<School[]>;
  public currentSchool$: Observable<School>;
  public currentStep: Step;
  public currentStepId: number;

  constructor(
    @Inject(MAT_DIALOG_DATA)
    public readonly data: {
      instance: Instance;
      humanPoiId: string;
      waypointId: string;
      currentUrl: string;
      stage: number;
    },
    private analyticsService: AnalyticsService,
    private context: ApplicationContext,
    private dialog: MatDialog,
    private dialogRef: MatDialogRef<HumanProfileDialogComponent>,
    private dialogService: DialogService,
    private instanceService: InstanceService,
    private sanitizer: DomSanitizer,
    private translateService: TranslateService,
    private videoAskService: VideoAskService,
    private waypointService: WaypointService
  ) {}

  public ngOnInit(): void {
    this.externalLinks = EXTERNAL_LINKS_CONFIG(this.context.product);
    this.withSocialNetworkLink =
      this.externalLinks && WITH_SOCIAL_NETWORK_LINK_CONFIG(this.context.application);

    this.humanPoi$ = this.instanceService
      .humanPoi(this.data.instance.id, this.data.humanPoiId)
      .pipe(
        // FIXME : Remove when VideoAsk is ready
        map(humanPoi => {
          if (this.data.humanPoiId === this.videoAskService.LIONEL_ID) {
            this.videoAskService.LIONEL_HUMAN_POI.steps[3].waypoint.id = this.data.waypointId;

            return this.videoAskService.LIONEL_HUMAN_POI;
          }

          return humanPoi;
        }),
        tap(humanPoi => {
          this.currentStepId = humanPoi.steps.findIndex(
            step => step.waypoint?.id === this.data.waypointId
          );

          if (this.currentStepId !== -1) {
            this.currentStep = humanPoi.steps[this.currentStepId];

            if (this.currentStep.videoAskId) {
              this.videoAskUrl = this.sanitizer.bypassSecurityTrustResourceUrl(
                `${this.videoAskService.getVideoAskUrl(this.currentStep.videoAskId)}`
              );
            }

            if (
              SCHOOLS_ENABLED_CONFIG(this.context.application, this.context.product) &&
              !this.data.stage &&
              isDiploma(this.currentStep.type)
            ) {
              this.schools$ = this.waypointService.schools(
                this.currentStep.waypoint,
                this.context.application
              );

              if (
                !NO_PROFILE_SCHOOL_CONFIG(this.context.application) &&
                this.currentStep?.location?.id
              )
                this.currentSchool$ = this.waypointService.school(
                  this.currentStep.waypoint,
                  this.context.application,
                  this.currentStep.location.id
                );
            }
          }
        })
      );
  }

  public onDown(): void {
    this.dialogContent.nativeElement.scrollIntoView({ behavior: 'smooth' });
  }

  public close(): void {
    this.dialogRef.close();
  }

  public onClickSocialNetworkLink(socialNetworkLink: string): void {
    this.analyticsService.pushTag(new LinkedInLinkClickAnalyticsEvent(socialNetworkLink));
  }

  public onTestimony(testimony: string, humanPoiName: string, humanPoiGender: Gender): void {
    this.dialog.open(
      TestimonyDialogComponent,
      this.dialogService.defaultDialogConfig({
        title: this.translateService.instant('team.testimony', {
          name: humanPoiName
        }),
        content: testimony,
        icon: situationIcon(Status.student, humanPoiGender)
      })
    );
  }
}
