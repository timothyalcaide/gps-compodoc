import { Component, OnInit } from '@angular/core';
import { INTERNSHIP_ADVENTURE_OPTIONS_CONFIG } from '@configs/internship-adventure-options.config';
import { IS_RECRUTER_CONFIG } from '@configs/recruter.config';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { ApplicationContext } from '@core/services/context/application/application.context.service';
import { DefinitionService } from '@core/services/definition/definition.service';
import { AdventureType } from '@models/adventure-type/adventure-type';
import { Filters } from '@models/filters/filters';
import { WaypointType } from '@models/waypoint/waypoint-type';
import { map, Observable, of } from 'rxjs';

@Component({
  selector: 'app-intentions-choice',
  templateUrl: './intentions-choice.component.html',
  styleUrls: ['./intentions-choice.component.scss']
})
export class IntentionsChoiceComponent implements OnInit {
  public readonly AppRouteUrls = AppRouteUrls;
  public readonly AdventureType = AdventureType;
  public readonly withInternshipAdventure: boolean;
  public hasDiplomaChoice$: Observable<{ result: boolean }>;

  constructor(private context: ApplicationContext, private definitionService: DefinitionService) {
    this.withInternshipAdventure = INTERNSHIP_ADVENTURE_OPTIONS_CONFIG(this.context.product);
  }

  public ngOnInit(): void {
    if (IS_RECRUTER_CONFIG(this.context.application)) {
      this.hasDiplomaChoice$ = this.definitionService
        .hasDefinitionsOfType(new Filters(), WaypointType.diploma)
        .pipe(map(result => ({ result })));
    } else this.hasDiplomaChoice$ = of({ result: true });
  }
}
