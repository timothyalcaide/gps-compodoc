import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { ApplicationContext, AuthService } from '@core/services';
import { ExpeditionInstance } from '@features/expeditions/models';
import { ExpeditionService } from '@features/expeditions/services';
import { Product } from '@models/context/product';
import { Instance } from '@models/instance/instance';
import { InstanceService } from '@services/instance.service';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';

@Component({
  templateUrl: './intentions.component.html',
  styleUrls: ['./intentions.component.scss']
})
export class IntentionsPageComponent implements OnInit {
  public roadmaps$: Observable<Instance[]>;
  public expeditionInstances$: Observable<ExpeditionInstance[]>;

  public get noRoadmapLabelKey(): string {
    if (!this.auth.isAuthenticated && this.auth.email)
      return 'roadmap.no-adventures-hint-authenticate';

    return 'roadmap.no-adventures-hint';
  }

  constructor(
    private auth: AuthService,
    private context: ApplicationContext,
    private expedition: ExpeditionService,
    private instance: InstanceService,
    private router: Router
  ) {}

  public ngOnInit(): void {
    this.roadmaps$ = this.instance
      .instances()
      .pipe(map(instances => instances.filter(instance => instance.saved)));

    if (this.context.product === Product.Connect) {
      this.expedition.getAllExpeditionInstances();
      this.expeditionInstances$ = this.expedition.expeditionInstancesInProgress$;
    }
  }

  public onClickRoadmapCard(roadmapId: string): void {
    this.router.navigate([`/${AppRouteUrls.ADVENTURE_ROADMAP(roadmapId)}`]);
  }
}
