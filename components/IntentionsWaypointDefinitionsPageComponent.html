import { Component, OnDestroy, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { isDiploma } from '@clients/adventure/bean/waypoint';
import { ON_BOARDING_FILTER_CONFIG } from '@configs/on-boarding-filter.config';
import { TRENDS_REGION_CONFIG } from '@configs/trends-region.config';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import {
  AdventureService,
  AnalyticsService,
  ApplicationContext,
  AuthService,
  LoaderService,
  PageTitleService,
  SituationContext
} from '@core/services';
import { DefinitionService } from '@core/services/definition/definition.service';
import { AdventureType } from '@models/adventure-type/adventure-type';
import { OnBoardingFinalizeAnalyticsEvent } from '@models/analytics-event/on-boarding-finalize.analytics-event';
import { SelectAdventureAnalyticsEvent } from '@models/analytics-event/select-adventure.analytics-event';
import { Definition } from '@models/definition/definition';
import { ErrorCodes } from '@models/error-codes/error-codes';
import { Filters } from '@models/filters/filters';
import { Instance } from '@models/instance/instance';
import { Situation } from '@models/situation/situation';
import { WaypointType } from '@models/waypoint/waypoint-type';
import { TranslateService } from '@ngx-translate/core';
import { Observable, Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

@Component({
  templateUrl: './intentions-waypoint-definitions.component.html',
  styleUrls: ['./intentions-waypoint-definitions.component.scss']
})
export class IntentionsWaypointDefinitionsPageComponent implements OnInit, OnDestroy {
  public readonly AppRouteUrls = AppRouteUrls;
  public situation: Situation;
  public filters = new Filters();
  public hasFilters: boolean;
  public waypointType: WaypointType;
  public definitionsByTag$: Observable<{ tagLabel: string; definitions: Definition[] }[]>;

  private onDestroySubject$ = new Subject<void>();

  constructor(
    private adventure: AdventureService,
    private analytics: AnalyticsService,
    private auth: AuthService,
    private context: ApplicationContext,
    private definition: DefinitionService,
    private loading: LoaderService,
    private route: ActivatedRoute,
    private router: Router,
    private situationContext: SituationContext,
    private pageTitle: PageTitleService,
    private translate: TranslateService
  ) {
    this.situation = this.situationContext.situation;
  }

  public ngOnInit(): void {
    const { application, product } = this.context;

    this.hasFilters = ON_BOARDING_FILTER_CONFIG(application, product);
    this.filters.trendsRegion = TRENDS_REGION_CONFIG(application);
    this.waypointType = this.route.snapshot.params.type;
    this.updatePageTitle(this.waypointType);

    if (this.hasFilters) {
      this.filters.situationFilters.situation = this.situationContext.situation;
      this.filters.situationFilters.useSituation = true;
    }

    this.initDefinitions(this.filters);
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public onSelectDefinition(definition: Definition): void {
    this.analytics.pushTag(new SelectAdventureAnalyticsEvent(definition.name));
    this.analytics.pushTag(
      new OnBoardingFinalizeAnalyticsEvent(
        'MAP',
        AdventureType.Definition.toUpperCase(),
        this.situationContext.situation.accurateStatus
      )
    );

    this.loading
      .globalObservableLoadingSession<Instance>(
        this.adventure.createAdventure(this.auth.email, { definition })
      )
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe({
        next: instance => {
          this.router.navigateByUrl(`/${AppRouteUrls.ADVENTURE_MAP(instance.id)}`);
        },
        // If adventure creation fails, redirect to 404
        error: () => this.router.navigateByUrl(`/${AppRouteUrls.ERROR(ErrorCodes.ERROR_404)}`)
      });
  }

  /**
   * Initialise definitions
   */
  public initDefinitions(definitionFilters: Filters): void {
    // Get an observable to get definitions for each tags
    this.definitionsByTag$ = this.definition.definitionsByTag(definitionFilters, this.waypointType);
  }

  public onRemoveFilters(): void {
    this.filters.empty();
    this.initDefinitions(this.filters);
  }

  private updatePageTitle(type: WaypointType): void {
    let title = 'suggestions.title.default';

    if (type === WaypointType.job) {
      title = 'suggestions.title.job';
    }

    if (isDiploma(type)) {
      title = 'suggestions.title.diploma';
    }

    this.pageTitle.overrideTitle(this.translate.instant(title));
  }
}

<app-simple-template
  [fullWidth]="true"
  [backRouterLink]="'/' + AppRouteUrls.INTENTIONS_WAYPOINT(waypointType)"
>
  <div class="definitions-main-container">
    <div class="mat-headline mb-48" [innerHTML]="'suggestions.header' | translate"></div>

    <div>
      <app-filters
        *ngIf="hasFilters"
        class="filters"
        [filters]="filters"
        [withExamsFilter]="true"
        [withSituationFilter]="true"
        (filter)="initDefinitions($event)"
      ></app-filters>
    </div>

    <ng-container *ngIf="definitionsByTag$ | async as listOfDefinitionsByTag; else loader">
      <ng-container *ngFor="let definitionsByTag of listOfDefinitionsByTag">
        <div *ngIf="definitionsByTag.definitions.length" class="mb-48">
          <div class="mat-title mb-16">
            {{ definitionsByTag.tagLabel | translate }}
          </div>
          <app-scroll-wrapper [objects]="definitionsByTag.definitions" [templateRef]="waypointCard">
          </app-scroll-wrapper>

          <ng-template #waypointCard let-definition>
            <app-waypoint-card
              [title]="definition.name"
              [waypoint]="definition.origin || definition.destination"
            ></app-waypoint-card>
          </ng-template>
        </div>
      </ng-container>

      <app-no-records
        *ngIf="!listOfDefinitionsByTag.length"
        [filters]="filters"
        (removeFilters)="onRemoveFilters()"
      ></app-no-records>
    </ng-container>
  </div>
</app-simple-template>
<ng-template #loader>
  <mat-spinner class="spinner"></mat-spinner>
</ng-template>
