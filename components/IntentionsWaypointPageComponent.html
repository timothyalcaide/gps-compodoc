import { Component, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { IS_RECRUTER_CONFIG } from '@configs/recruter.config';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { AnalyticsService, BreakpointService, PageTitleService } from '@core/services';
import { ApplicationContext } from '@core/services/context/application/application.context.service';
import { DefinitionService } from '@core/services/definition/definition.service';
import { AdventureType } from '@models/adventure-type/adventure-type';
import { AnalyticsEvent } from '@models/analytics-event/analytics-event';
import { AnalyticsEventType } from '@models/analytics-event/analytics-event-type';
import { Product } from '@models/context/product';
import { Filters } from '@models/filters/filters';
import { WaypointType } from '@models/waypoint/waypoint-type';
import { TranslateService } from '@ngx-translate/core';
import { map, Observable, of } from 'rxjs';

@Component({
  templateUrl: './intentions-waypoint.component.html',
  styleUrls: ['./intentions-waypoint.component.scss']
})
export class IntentionsWaypointPageComponent implements OnInit {
  public readonly AppRouteUrls = AppRouteUrls;
  public readonly isMobile$: Observable<boolean>;
  public readonly AdventureType = AdventureType;
  public readonly WaypointType = WaypointType;
  public readonly isRecruter: boolean;
  public readonly isConnect: boolean;
  public adventureType: AdventureType;
  public hasDefinitions$: Observable<{ result: boolean }>;

  constructor(
    private analytics: AnalyticsService,
    private breakpoint: BreakpointService,
    private context: ApplicationContext,
    private definitionService: DefinitionService,
    private pageTitle: PageTitleService,
    private route: ActivatedRoute,
    private translate: TranslateService
  ) {
    this.isMobile$ = this.breakpoint.isMobile$;
    this.isRecruter = IS_RECRUTER_CONFIG(this.context.application);
    this.isConnect = this.context.product === Product.Connect;
  }

  public ngOnInit(): void {
    this.adventureType = this.route.snapshot.params.type;

    this.pageTitle.overrideTitle(this.translate.instant(`intentions.${this.adventureType}.title`));

    if (this.adventureType !== AdventureType.Internship) {
      const waypointType =
        this.adventureType === AdventureType.Job ? WaypointType.job : WaypointType.diploma;

      this.hasDefinitions$ = this.definitionService
        .hasDefinitionsOfType(new Filters(), waypointType)
        .pipe(map(result => ({ result })));
    } else this.hasDefinitions$ = of({ result: true });
  }

  public onOpenDefinitions(): void {
    this.analytics.pushTag(new AnalyticsEvent(AnalyticsEventType.DiscoverAdventures));
  }
}
