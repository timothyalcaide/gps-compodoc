import { Component, OnDestroy, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { Waypoint } from '@clients/adventure/bean/waypoint';
import { TRENDS_REGION_CONFIG } from '@configs/trends-region.config';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import {
  AnalyticsService,
  ApplicationContext,
  PageTitleService,
  WaypointService
} from '@core/services';
import { JobOrDiplomaSearchAnalyticsEvent } from '@models/analytics-event/job-diploma-search.analytics-event';
import { Product } from '@models/context/product';
import { Filters } from '@models/filters/filters';
import { WaypointType } from '@models/waypoint/waypoint-type';
import { TranslateService } from '@ngx-translate/core';
import { BehaviorSubject, map, Observable, Subject, takeUntil, tap } from 'rxjs';

@Component({
  templateUrl: './intentions-waypoint-search.component.html'
})
export class IntentionsWaypointSearchPageComponent implements OnInit, OnDestroy {
  public backLink: string;
  public waypoints: Waypoint[];
  public waypointType: WaypointType;
  public readonly isLoading$: Observable<boolean>; // TODO:TIM export ds service d'UI State management

  private readonly isLoading = new BehaviorSubject<boolean>(false); // TODO:TIM export ds service d'UI State management
  private readonly filters: Filters = new Filters();
  private readonly destroy$: Subject<void> = new Subject<void>();

  constructor(
    private analytics: AnalyticsService,
    private context: ApplicationContext,
    private pageTitle: PageTitleService,
    private route: ActivatedRoute,
    private translate: TranslateService,
    private waypoint: WaypointService
  ) {
    this.isLoading$ = this.isLoading.asObservable();
  }

  public ngOnInit(): void {
    this.waypointType = this.route.snapshot.params.type;
    this.backLink = `/${AppRouteUrls.INTENTIONS_WAYPOINT(this.waypointType)}`;

    this.pageTitle.overrideTitle(
      this.translate.instant(`intentions.${this.waypointType}.search.title`)
    );

    this.filters.trendsRegion = TRENDS_REGION_CONFIG(this.context.application);

    this.onSearch('');
  }

  public ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  public onSearch(text: string): void {
    this.isLoading.next(true);
    this.searchWaypoint(text)
      .pipe(takeUntil(this.destroy$))
      .subscribe({
        next: value => {
          this.waypoints = value;
        },
        error: () => this.isLoading.next(false),
        complete: () => this.isLoading.next(false)
      });
  }

  /**
   * Search waypoints based on a provided search text
   *
   * @param searchText The search text
   * @returns The matching waypoints
   */
  private searchWaypoint(searchText: string): Observable<Waypoint[]> {
    // Hack to avoid backend refacto, search on connect is the same of gps grand public
    const nodeId =
      this.context.product === Product.Connect
        ? '300e4d54-172c-4c34-9324-952c57620c16' // nodeId of gps grand public
        : this.context.application.nodeId;

    return this.waypoint.resumeNodes(nodeId, this.waypointType, searchText, this.filters).pipe(
      tap(waypoints => {
        // Track search waypoint event
        if (searchText.trim()) {
          this.analytics.pushTag(
            new JobOrDiplomaSearchAnalyticsEvent(
              this.waypointType,
              searchText.toLowerCase(),
              waypoints.length
            )
          );
        }
      }),
      // Set the waypoint type which is not sent back
      map(waypoints =>
        waypoints.map(waypoint => ({
          ...waypoint,
          type: this.waypointType
        }))
      )
    );
  }
}

<app-simple-template [backRouterLink]="backLink">
  <app-search-form

    [placeholder]="'intentions.' + waypointType + '.search.search' | translate"
    (search)="onSearch($event)"

  ></app-search-form>

  <ng-container *ngIf="isLoading$ | async; else loaded">
    <mat-spinner class="spinner"></mat-spinner>
  </ng-container>

  <ng-template #loaded>
    <app-waypoint-card-list
      *ngIf="waypoints?.length; else empty"
      [waypoints]="waypoints"
    ></app-waypoint-card-list>
    <ng-template #empty>
      <app-no-records (removeFilters)="form.clear()"></app-no-records>
    </ng-template>
  </ng-template>
</app-simple-template>
