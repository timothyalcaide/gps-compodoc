import { Component, Input, OnInit } from '@angular/core';
import { isJob } from '@clients/adventure/bean/waypoint';
import { Step } from '@clients/poi/poi';
import { ItineraryStep } from '@models/itinerary-step/itinerary-step';
import { Situation } from '@models/situation/situation';
import { StepLocation } from '@models/step-location/step-location';
import { TranslateService } from '@ngx-translate/core';
import dayjs from 'dayjs';

@Component({
  selector: 'app-itinerary-step-card',
  templateUrl: './itinerary-step-card.component.html',
  styleUrls: ['./itinerary-step-card.component.scss']
})
export class ItineraryStepCardComponent implements OnInit {
  /**
   * The itinerary step
   */
  @Input() public itineraryStep!: ItineraryStep;

  /**
   * The number of line after which the label is shorten with an ellipsis
   */
  @Input() public ellipsis: 0 | 1 | 2 | 3 = 1;

  /**
   * Whether waypoint summary should be displayed
   */
  @Input() public withWaypointSummary = false;

  /**
   * Whether the itinerary card label is a title
   */
  @Input() public titleLabel = false;

  @Input() public situation!: Situation;

  public readonly isJob = isJob;
  public isFirstStepWithSituation: boolean;

  constructor(private translateService: TranslateService) {}

  public ngOnInit(): void {
    this.isFirstStepWithSituation = this.itineraryStep.isFirst && !!this.situation;
  }

  /**
   * Resolve the icon of a substep
   */
  public substepIcon(substep: Step): string {
    if (
      this.itineraryStep.location?.countryCode &&
      substep.location?.countryCode &&
      this.itineraryStep.location.countryCode !== substep.location.countryCode
    )
      return '✈';

    if (substep.isInternship || substep.isSandwichTraining || isJob(substep.type)) return '💼';

    return '📅';
  }

  /**
   * Resolve a user readable duration of a substep
   */
  public substepDuration(substep: Step): string {
    if (!substep.end || !substep.start) return this.translateService.instant('team.period');

    const monthDiff = dayjs(substep.end).diff(dayjs(substep.start), 'months');

    if (isJob(substep.type)) {
      if (monthDiff < 1) return this.translateService.instant('team.month-less');

      return `${monthDiff} ${this.translateService.instant('team.months')}`;
    }

    if (monthDiff < 6) return this.translateService.instant('team.semester-less');

    return `${Math.floor(monthDiff / 6)} ${this.translateService.instant('team.semesters')}`;
  }

  /**
   * Get the translation key from the type of a substep
   */
  public substepType(substep: Step): string {
    if (substep.isInternship) return 'team.internship';

    if (substep.isSandwichTraining) return 'team.sandwich-training';

    if (isJob(substep.type)) return 'team.job';

    return 'team.courses';
  }

  /**
   * Resolve a user readable location of a substep
   */
  public substepLocation(location: StepLocation): string {
    if (location.city && location.country)
      return `${this.translateService.instant('team.to')} ${location.city}, ${location.country}`;

    return `${this.translateService.instant('team.to')} ${location.city || location.country || ''}`;
  }
}

<mat-card class="small-padding-card itinerary-step-card">
  <div class="itinerary-step-card-label full-width">
    <ng-container *ngIf="itineraryStep.isFirst; else notFirst">
      <mat-icon
        *ngIf="isFirstStepWithSituation; else ellipse"
        [svgIcon]="situation | situationIcon"
        class="mr-8"
      ></mat-icon>

      <ng-template #ellipse>
        <div class="mr-8">
          <div class="ellipse"></div>
        </div>
      </ng-template>
    </ng-container>

    <ng-template #notFirst>
      <mat-icon [svgIcon]="itineraryStep.pinIcon" class="mr-8"></mat-icon>
    </ng-template>

    <div
      [ngClass]="[
        ellipsis === 1 ? 'ellipsis' : '',
        ellipsis !== 0 && ellipsis !== 1 ? 'ellipsis-' + ellipsis + '-lines' : '',
        titleLabel ? 'mat-title mb-0' : ''
      ]"
    >
      {{ itineraryStep.label | translate }}
    </div>
  </div>

  <div
    *ngIf="itineraryStep.location || itineraryStep.substeps.length || withWaypointSummary"
    class="details-container"
  >
    <ng-container *ngIf="itineraryStep.location">
      <div class="mat-caption mt-8">{{ itineraryStep.location.name }}</div>
      <div *ngIf="itineraryStep.location.city" class="mat-caption medium-grey-2 mt-8">
        📍 {{ itineraryStep.location.city }}
      </div>
    </ng-container>

    <details *ngIf="itineraryStep.substeps.length" class="mat-caption medium-grey-2 mt-24">
      <summary>
        {{
          (itineraryStep.substeps.length > 1 ? "team.parallel-xps" : "team.parallel-xp")
            | translate: { nb: itineraryStep.substeps.length }
        }}
      </summary>
      <div
        *ngFor="let substep of itineraryStep.substeps; last as isLast"
        class="mt-16 container-baseline"
        [ngClass]="{ 'mb-8': !isLast }"
      >
        <div class="mr-8">{{ substepIcon(substep) }}</div>
        <span class="mat-caption">
          {{ substepDuration(substep) }} {{ substepType(substep) | translate
          }}<ng-container *ngIf="substep.location?.city || substep.location?.country">
            {{ substepLocation(substep.location) }}</ng-container
          >
        </span>
      </div>
    </details>

    <app-waypoint-summary
      *ngIf="
        withWaypointSummary &&
        (itineraryStep.userCount ||
          itineraryStep.cost ||
          (!isJob(itineraryStep.type) && itineraryStep.level))
      "
      class="mat-caption medium-grey-2 mt-16"
      [userCount]="itineraryStep.userCount"
      [duration]="!isJob(itineraryStep.type) && itineraryStep.level"
      [cost]="itineraryStep.cost"
    ></app-waypoint-summary>
  </div>
</mat-card>
