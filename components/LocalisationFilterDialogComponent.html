import { Component, Inject, OnDestroy, OnInit, TemplateRef, ViewChild } from '@angular/core';
import { FormBuilder, FormGroup } from '@angular/forms';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { AnalyticsService, BreakpointService, DialogService, HroadsService } from '@core/services';
import { LocationFilterApplyAnalyticsEvent } from '@models/analytics-event/location-filter-apply.analytics-event';
import { RegionFilterApplyAnalyticsEvent } from '@models/analytics-event/region-filter-apply.analytics-event';
import { City } from '@models/city/city';
import { LocalisationFilterType } from '@models/filters/localisation-filters/localisation-filter-type';
import { LocalisationFilters } from '@models/filters/localisation-filters/localisation-filters';
import { SearchDialogComponent } from '@organisms/search-dialog/search-dialog.component';
import { BehaviorSubject, merge, Observable, of, Subject } from 'rxjs';
import {
  debounceTime,
  distinctUntilChanged,
  filter,
  map,
  switchMap,
  takeUntil
} from 'rxjs/operators';
import { Regions } from './regions';

@Component({
  selector: 'app-localisation-filter-dialog',
  templateUrl: './localisation-filter-dialog.component.html',
  styleUrls: ['./localisation-filter-dialog.component.scss']
})
export class LocalisationFilterDialogComponent implements OnInit, OnDestroy {
  @ViewChild('mobileSearchCityResult') private mobileSearchCityResultTemplate: TemplateRef<City>;

  public readonly localisationFilterType = LocalisationFilterType;
  public localisationFilterForm: FormGroup;
  public citySearchResults$: Observable<City[]>;
  public isGeolocationLoading$ = new BehaviorSubject<boolean>(false);

  public readonly Regions = Regions;
  private readonly CITY_RESULT_LIMIT = 5;
  private onDestroySubject$: Subject<void> = new Subject<void>();

  public get hasSearchText(): boolean {
    return !!this.localisationFilterForm.controls.citySearchText.value;
  }

  public get isCitySelected(): boolean {
    return this.localisationFilterForm.controls[LocalisationFilterType.City].value;
  }

  public get selectedCity(): City {
    return this.localisationFilterForm.controls.selectedCity.value as City;
  }

  constructor(
    @Inject(MAT_DIALOG_DATA) private initialState: LocalisationFilters,
    private analyticsService: AnalyticsService,
    private breakpointService: BreakpointService,
    private dialog: MatDialogRef<LocalisationFilterDialogComponent>,
    private dialogContainer: MatDialog,
    private dialogService: DialogService,
    private formBuilder: FormBuilder,
    private hroadsService: HroadsService
  ) {}

  public ngOnInit(): void {
    this.initForm();
    this.listenFilterSelections();
    this.listenNearByFilterSelections();
    this.listenNearByFilterDeselections();
    this.listenCityFilterDeselections();
    this.listenCitySearch();
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public onSeeResults(): void {
    const locationFilters = new LocalisationFilters(
      this.localisationFilterForm.controls.userGeolocation.value,
      this.localisationFilterForm.controls.selectedCity.value,
      this.localisationFilterForm.controls[LocalisationFilterType.Regional].value
    );

    // Track location filter apply event
    if (locationFilters.city || locationFilters.userGeolocation) {
      const locationFilterApplyAnalyticsEvent = new LocationFilterApplyAnalyticsEvent();

      if (locationFilters.city)
        locationFilterApplyAnalyticsEvent.filterCity = locationFilters.city.name;
      else if (locationFilters.userGeolocation)
        locationFilterApplyAnalyticsEvent.filterAroundMe = true;

      this.analyticsService.pushTag(locationFilterApplyAnalyticsEvent);
    }

    locationFilters.regional.forEach(selectedRegion => {
      this.analyticsService.pushTag(new RegionFilterApplyAnalyticsEvent(selectedRegion));
    });

    this.dialog.close(locationFilters);
  }

  public onSearchCity(): void {
    if (!this.breakpointService.snapshot.isDesktop) {
      this.dialogContainer
        .open(
          SearchDialogComponent,
          this.dialogService.defaultDialogConfig({
            searchFn: (searchText: string) => this.searchCities(searchText),
            title: 'dialog.mobile-search.city-title',
            inputPlaceholder: 'dialog.localisation-filter.form.city',
            resultTemplate: this.mobileSearchCityResultTemplate
          })
        )
        .afterClosed()
        .pipe(
          filter((city: City | '') => !!city),
          takeUntil(this.onDestroySubject$)
        )
        .subscribe((city: City) => {
          this.onSelectCity(city);
        });
    }
  }

  public onEmptySearch(): void {
    this.localisationFilterForm.controls.citySearchText.setValue('');
  }

  /**
   * Select the city localisation filter type associated with the city
   *
   * @param city The selected city
   */
  public onSelectCity(city: City): void {
    this.localisationFilterForm.controls[LocalisationFilterType.City].setValue(true);
    this.localisationFilterForm.controls.selectedCity.setValue(city);

    this.onEmptySearch();
  }

  public onErase(): void {
    this.localisationFilterForm.controls[LocalisationFilterType.Nearby].setValue(false);
    this.localisationFilterForm.controls[LocalisationFilterType.City].setValue(false);
    this.localisationFilterForm.controls[LocalisationFilterType.Regional].setValue([]);
    this.localisationFilterForm.controls.selectedCity.setValue(null);
    this.localisationFilterForm.controls.userGeolocation.setValue(null);

    this.onEmptySearch();
  }

  /**
   * Initialise localisation filter form
   */
  private initForm(): void {
    this.localisationFilterForm = this.formBuilder.group({
      citySearchText: '',
      [LocalisationFilterType.Nearby]: !!this.initialState.userGeolocation,
      [LocalisationFilterType.City]: !!this.initialState.city,
      [LocalisationFilterType.Regional]: [this.initialState.regional],
      selectedCity: this.initialState.city,
      userGeolocation: this.initialState.userGeolocation
    });
  }

  /**
   * Listen all filter selections and deselect other filters when one is selected
   */
  private listenFilterSelections(): void {
    merge(
      // Listen to near by filter selection
      this.localisationFilterForm.controls[LocalisationFilterType.Nearby].valueChanges.pipe(
        filter((selected: boolean) => selected),
        map(() => LocalisationFilterType.Nearby)
      ),
      // Listen to city filter selection
      this.localisationFilterForm.controls[LocalisationFilterType.City].valueChanges.pipe(
        filter((selected: boolean) => selected),
        map(() => LocalisationFilterType.City)
      )
    )
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe(localisationFilterType => this.deselectOthers(localisationFilterType));
  }

  /**
   * Listen to near by localisation filter selections to invite user to enable geolocation
   */
  private listenNearByFilterSelections(): void {
    this.localisationFilterForm.controls[LocalisationFilterType.Nearby].valueChanges
      .pipe(
        filter((selected: boolean) => selected),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe(() => {
        // Ignores if the user location is already know
        if (this.localisationFilterForm.controls.userGeolocation.value) return;

        this.isGeolocationLoading$.next(true);

        /**
         * Deselect the near by localisation filter until the user geolocation is got so if the user
         * does not enable geolocation, the filter is not selected
         */
        this.localisationFilterForm.controls[LocalisationFilterType.Nearby].setValue(false);

        navigator.geolocation.getCurrentPosition(position => {
          // When the user has enabled geolocation, select the near by filter
          this.localisationFilterForm.controls.userGeolocation.setValue(position.coords);
          this.localisationFilterForm.controls[LocalisationFilterType.Nearby].setValue(true);
          this.isGeolocationLoading$.next(false);
        });
      });
  }

  /**
   * Listen to near by localisation filter deselections to empty user location
   */
  private listenNearByFilterDeselections(): void {
    this.localisationFilterForm.controls[LocalisationFilterType.Nearby].valueChanges
      .pipe(
        filter((selected: boolean) => !selected),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe(() => {
        this.localisationFilterForm.controls.userGeolocation.setValue(null);
      });
  }

  /**
   * Listen to city localisation filter deselections to empty selected city
   */
  private listenCityFilterDeselections(): void {
    this.localisationFilterForm.controls[LocalisationFilterType.City].valueChanges
      .pipe(
        filter((selected: boolean) => !selected),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe(() => {
        this.localisationFilterForm.controls.selectedCity.setValue(null);
      });
  }

  /**
   * Listen city search changes to launch the city search along API
   */
  private listenCitySearch(): void {
    this.citySearchResults$ = this.localisationFilterForm.controls.citySearchText.valueChanges.pipe(
      debounceTime(400),
      distinctUntilChanged(),
      filter((searchText: string) => !!searchText),
      switchMap(searchText => this.searchCities(searchText))
    );
  }

  /**
   * Deselect other localisation filters
   *
   * @param localisationFilter The new selected localisation filter
   */
  private deselectOthers(localisationFilter: LocalisationFilterType): void {
    Object.keys(LocalisationFilterType).forEach(key => {
      if (
        LocalisationFilterType[key] !== localisationFilter &&
        LocalisationFilterType[key] !== LocalisationFilterType.Regional &&
        this.localisationFilterForm.controls[LocalisationFilterType[key]].value
      )
        this.localisationFilterForm.controls[LocalisationFilterType[key]].setValue(false);
    });
  }

  /**
   * Search cities based on a provided search text
   *
   * @param searchText The search text
   */
  private searchCities(searchText: string): Observable<City[]> {
    // Prevent error when searching from mobile search dialog
    if (!searchText) return of([]);

    return this.hroadsService.searchCity(searchText, this.CITY_RESULT_LIMIT);
  }
}
