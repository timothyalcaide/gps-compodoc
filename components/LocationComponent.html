import {
  ChangeDetectionStrategy,
  Component,
  HostBinding,
  Input,
  OnDestroy,
  OnInit
} from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { Waypoint } from '@clients/adventure/bean/waypoint';
import { CompanyClient, DialogService } from '@core/services';
import { Company } from '@models/company/company';
import { Instance } from '@models/instance/instance';
import { Location } from '@models/location/location';
import { School } from '@models/school/school';
import { CompanyDetailsDialogComponent } from '@organisms/company-details-dialog/company-details-dialog.component';
import { SchoolDetailsDialogComponent } from '@organisms/school-details/school-details-dialog.component';
import { ApplicationService } from '@services/application.service';
import { combineLatest, Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

@Component({
  selector: 'app-location',
  templateUrl: './location.component.html',
  styleUrls: ['./location.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class LocationComponent implements OnInit, OnDestroy {
  @Input() public location: Location;
  @Input() public waypoint: Waypoint;
  @Input() public asCard = false;
  @Input() public instance: Instance;
  @HostBinding('attr.id') public id: string;

  public readonly School: School;
  public readonly Company: Company;
  public isSchool: boolean;
  public isCompany: boolean;

  private destroy$: Subject<void> = new Subject<void>();

  constructor(
    private applicationService: ApplicationService,
    private dialog: MatDialog,
    private dialogService: DialogService,
    private companyClient: CompanyClient
  ) {}

  public ngOnInit(): void {
    this.isSchool = this.location instanceof School;
    this.isCompany = this.location instanceof Company;

    if (this.isSchool) {
      this.id = `school-${this.location.id}`;
    }
  }

  public ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  public onDiscover(): void {
    const application$ = (this.location as School).hasAcquisition
      ? this.applicationService.getApplication(this.location.application)
      : this.applicationService.getSchoolApplication(this.location.id);

    application$.pipe(takeUntil(this.destroy$)).subscribe(application =>
      this.dialog.open(
        SchoolDetailsDialogComponent,
        this.dialogService.defaultDialogConfig({
          application,
          school: this.location,
          waypoint: this.waypoint,
          instance: this.instance
        })
      )
    );
  }

  public onCompanyDetails(id: string): void {
    combineLatest([this.companyClient.getCompany(id), this.companyClient.getHumansFromCompany(id)])
      .pipe(takeUntil(this.destroy$))
      .subscribe(([company, humans]) => {
        this.dialog.open(
          CompanyDetailsDialogComponent,
          this.dialogService.defaultDialogConfig({
            company,
            humans
          })
        );
      });
  }
}
