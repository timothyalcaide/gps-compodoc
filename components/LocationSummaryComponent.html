import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { OFFER_COST_CONFIG } from '@configs/offer-cost.config';
import { ApplicationContext } from '@core/services';
import { Location } from '@models/location/location';
import { School } from '@models/school/school';
import { WaypointType } from '@models/waypoint/waypoint-type';
import { TranslateService } from '@ngx-translate/core';

@Component({
  selector: 'app-location-summary',
  templateUrl: './location-summary.component.html',
  styleUrls: ['./location-summary.component.scss']
})
export class LocationSummaryComponent implements OnInit {
  @Input() public location: Location;
  @Input() public condensed = false;
  @Input() public title = 'mat-title';

  /**
   * The type of waypoint where the location is present
   * Useful only for locations of type school
   * It is not required for condensed summary
   */
  @Input() public waypointType: WaypointType;

  @Output() public readonly contactStudent = new EventEmitter<void>();

  public readonly School: School;
  public hasCost: boolean;

  public get distance(): number {
    return Math.round(this.location.distance / 1000);
  }

  public get moreCampus(): string {
    const citiesSlice = this.location.cities.slice(3);

    return citiesSlice.length === 1
      ? this.translateService.instant('school.condensed-campus')
      : this.translateService.instant('school.condensed-campuses', {
          count: citiesSlice.length
        });
  }

  constructor(private context: ApplicationContext, private translateService: TranslateService) {}

  public ngOnInit(): void {
    if (this.location instanceof School) {
      this.hasCost =
        OFFER_COST_CONFIG(this.context.application) &&
        this.location.cost !== null &&
        this.location.cost !== undefined;
    }
  }
}
