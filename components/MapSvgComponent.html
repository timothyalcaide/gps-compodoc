import {
  Component,
  ElementRef,
  EventEmitter,
  Input,
  OnDestroy,
  OnInit,
  Output,
  ViewChild
} from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { Waypoint } from '@clients/adventure/bean/waypoint';
import { Poi } from '@clients/poi/poi';
import { GAMIFICATION_ENABLED_CONFIG } from '@configs/gamification-enabled.config';
import { AdventureService, ApplicationContext, DialogService, RewardService } from '@core/services';
import { AnimateMapDirective } from '@features/map/directives/animate-map/animate-map.directive';
import { District, Kind, MapBowElement, Sign, Village } from '@features/map/models';
import { ApplicationName } from '@models/context/application-name';
import { RegionInfoDialogComponent } from '@organisms/region-info-dialog/region-info-dialog.component';
import { BehaviorSubject, Observable, of, Subject } from 'rxjs';
import { filter, map, switchMapTo, takeUntil } from 'rxjs/operators';

@Component({
  selector: 'app-map-svg',
  templateUrl: './map-svg.component.html',
  styleUrls: ['./map-svg.component.scss']
})
export class MapSvgComponent implements OnInit, OnDestroy {
  @Input() public village: Village;
  @Input() public villageIndex: number;
  @Input() public initialDistrict: District;
  @Input() public focusedPoi$: BehaviorSubject<Poi>;
  @Input() public siblings: Waypoint[];
  @Output() public readonly villageIn = new EventEmitter<Village>();
  @Output() public readonly villageOut = new EventEmitter<void>();
  @Output() public readonly openVillageDetails = new EventEmitter<void>();
  @Output() public readonly roadmap = new EventEmitter<void>();
  @Output() public readonly pickVillage = new EventEmitter<Waypoint>();

  @ViewChild(AnimateMapDirective, { static: true }) public animateMap: AnimateMapDirective;
  @ViewChild('svg', { static: true }) public svg: ElementRef<SVGGraphicsElement>;

  public animationComplete$ = new BehaviorSubject<boolean>(false);
  public showEasterEgg$: Observable<boolean>;
  public IS_REGION_SUD: boolean;

  private onDestroySubject$: Subject<void> = new Subject<void>();

  /**
   * Get the element to focus with the user pin
   * It is the element following the last selected element
   */
  public get focusedElement(): MapBowElement {
    let selectedElement: MapBowElement = this.initialDistrict;

    while (selectedElement?.selected) {
      switch (selectedElement.kind) {
        case Kind.Crossroad:
          selectedElement = selectedElement.children.find(child => child.selected);
          break;
        case Kind.District:
        case Kind.Road:
          [selectedElement] = selectedElement.children;
          break;
        default:
          throw new Error('Unexpected kind for a MapBowElement.');
      }
    }

    return selectedElement;
  }

  constructor(
    private adventureService: AdventureService,
    private context: ApplicationContext,
    private dialog: MatDialog,
    private dialogService: DialogService,
    private rewardService: RewardService
  ) {}

  public ngOnInit(): void {
    this.showEasterEgg$ = this.animationComplete$.pipe(
      map(
        animationComplete =>
          GAMIFICATION_ENABLED_CONFIG(this.context.product, this.context.applicationName) &&
          this.rewardService.showEasterEgg() &&
          animationComplete
      )
    );
    this.IS_REGION_SUD = this.context.applicationName === ApplicationName.RegionSud;
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public onPinchedOut(): void {
    if (!this.village || this.village.stage < 1) return;

    this.onVillageOut();
  }

  /**
   * Get into a village
   */
  public onVillageIn(village: Village): void {
    this.villageIn.emit(village);

    this.animateMap.zoomIn(village, this.animationComplete$);
  }

  /**
   * Get out of the current village
   */
  public onVillageOut(): void {
    const villageDistrict = this.village.district;

    this.villageOut.emit();

    this.animateMap.zoomOut(villageDistrict, this.animationComplete$);
  }

  /**
   * Move to the next step
   *
   * @param followingElement The element to which the map is moving
   */
  public moveToNextStep(followingElement: MapBowElement): void {
    this.animateMap.zoomOut(followingElement, this.animationComplete$);
  }

  public onAddVillageToInstance(village: Village): void {
    this.adventureService.pushToInstance(village.waypoint, village.stage);
  }

  public onSelectSign(sign: Sign): void {
    this.animateMap.glide(sign, this.animationComplete$);
  }

  /**
   * When a POI is clicked, select it or unselect it if already selected
   */
  public onTogglePoi(poi: Poi): void {
    /**
     * // FIXME : Unselection on Poi click is not working because a tap on the SVG box unselects
     * Poi, right before the toggle call
     */
    if (poi.id === this.focusedPoi$.value?.id) this.focusedPoi$.next(null);
    else {
      this.focusedPoi$.next(poi);

      if (GAMIFICATION_ENABLED_CONFIG(this.context.product, this.context.applicationName)) {
        this.rewardService
          .scoringPoi(poi)
          .pipe(filter(score => !!score.points))
          .pipe(
            switchMapTo(this.rewardService.getInstanceScore(this.rewardService.instanceScore.id)),
            takeUntil(this.onDestroySubject$)
          )
          .subscribe();
      }
    }
  }

  public onClickEasterEgg(): void {
    this.showEasterEgg$ = of(false);
  }

  public onNext(): void {
    this.pickVillage.emit(this.siblings[++this.villageIndex]);
  }

  public onPrev(): void {
    this.pickVillage.emit(this.siblings[--this.villageIndex]);
  }

  public onInfo(): void {
    this.dialog.open(
      RegionInfoDialogComponent,
      this.dialogService.defaultDialogConfig(this.adventureService.instance.situation)
    );
  }
}
