import {
  Component,
  ElementRef,
  EventEmitter,
  Input,
  OnDestroy,
  OnInit,
  Output,
  ViewChild
} from '@angular/core';
import { MatDrawer } from '@angular/material/sidenav';
import { isJob, Waypoint } from '@clients/adventure/bean/waypoint';
import { Poi } from '@clients/poi/poi';
import { IN_OFFER_CONFIG } from '@configs/in-offer.config';
import { ON_BOARDING_ENABLED_CONFIG } from '@configs/on-boarding-enabled.config';
import {
  AdventureService,
  AnalyticsService,
  ApplicationContext,
  BreakpointService,
  LoaderService
} from '@core/services';
import { Village } from '@features/map/models';
import { MapService } from '@features/map/services/map.service';
import { MoreDetailsVillageFromPoiClickAnalyticsEvent } from '@models/analytics-event/more-details-village-from-poi-click.analytics-event';
import { Instance } from '@models/instance/instance';
import { ItineraryStep } from '@models/itinerary-step/itinerary-step';
import { TabId } from '@models/tab-ids/tab-ids';
import { VillageCardComponent } from '@molecules/village-card/village-card.component';
import { ItineraryService } from '@services/itinerary.service';
import { BehaviorSubject, map, Observable, Subject } from 'rxjs';
import { MapDrawerMode } from './map-drawer-mode';

@Component({
  selector: 'app-map-template',
  templateUrl: './map-template.component.html',
  styleUrls: ['./map-template.component.scss']
})
export class MapTemplateComponent implements OnInit, OnDestroy {
  @Input() public village: Village;
  @Input() public villageIndex: number;
  @Input() public siblings: Waypoint[];
  @Input() public focusedPoi$: BehaviorSubject<Poi>;
  @Output() public readonly exit = new EventEmitter<void>();
  @Output() public readonly roadmap = new EventEmitter<void>();
  @Output() public readonly moveVillage = new EventEmitter<void>();
  @Output() public readonly pickVillage = new EventEmitter<Waypoint>();
  @Output() public readonly chooseVillage = new EventEmitter<void>();
  @Output() public readonly closeVillageDetails = new EventEmitter<void>();
  @ViewChild('mapDrawer', { static: true }) public mapDrawer: MatDrawer;
  @ViewChild('currentVillageCard') public currentVillageCardComponent: VillageCardComponent;
  @ViewChild('desktopCardList') public desktopCardList: ElementRef<HTMLDivElement>;

  public readonly MapDrawerMode = MapDrawerMode;
  public readonly TabId = TabId;
  public readonly isDesktop$: Observable<boolean>;
  public readonly instance: Instance;
  public mapDrawerMode: MapDrawerMode = MapDrawerMode.VillageDetailsMode;
  public orderPois$: Observable<Poi[]>;
  public withOnBoarding: boolean;
  public itinerarySteps: ItineraryStep[];

  private onDestroySubject$: Subject<void> = new Subject<void>();

  public get isLoading$(): Observable<boolean> {
    return this.loaderService.isLoading$;
  }

  public get onlySiblings(): Waypoint[] {
    return this.mapService.onlySiblings(this.villageIndex, this.siblings);
  }

  public get inOfferSiblings(): boolean {
    return IN_OFFER_CONFIG(this.context.application, this.village.stage, this.context.product);
  }

  public get chooseButtonLabelKey(): string {
    return isJob(this.village.waypoint.type) ? 'villages.choose-job-btn' : 'villages.choose-btn';
  }

  constructor(
    private adventureService: AdventureService,
    private analyticsService: AnalyticsService,
    private breakpointService: BreakpointService,
    private context: ApplicationContext,
    private itinerary: ItineraryService,
    private loaderService: LoaderService,
    private mapService: MapService
  ) {
    this.isDesktop$ = this.breakpointService.isDesktop$;
    this.instance = this.adventureService.instance;
  }

  public ngOnInit(): void {
    this.withOnBoarding = ON_BOARDING_ENABLED_CONFIG(this.context.applicationName);

    this.orderPois$ = this.focusedPoi$.pipe(
      map(focusedPoi => {
        if (!focusedPoi) return this.allPois();

        return this.mapService.orderPois(focusedPoi, this.allPois());
      })
    );

    this.itinerarySteps = this.itinerary.getItinerarySteps(
      this.instance?.jobOrDiplomaWps,
      !this.village || this.village.selected,
      true,
      this.instance.situation
    );
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public onOpenMenu(): void {
    this.mapDrawer.open();
    this.mapDrawerMode = MapDrawerMode.MenuMode;
  }

  public onOpenVillageDetails(): void {
    this.mapDrawer.open();
    this.mapDrawerMode = MapDrawerMode.VillageDetailsMode;
  }

  public onCloseVillageDetails(): void {
    this.mapDrawer.close();

    this.closeVillageDetails.emit();
  }

  /**
   * Choose the village from village details
   */
  public onVillageChooseInVillageDetails(): void {
    this.onCloseVillageDetails();

    // Unselect the focused POI so the Netflix button is shown to the user
    if (this.focusedPoi$.value) this.focusedPoi$.next(null);

    if (this.village.waypoint.final) this.currentVillageCardComponent.onRoadmap();
    else this.currentVillageCardComponent.onChoose();
  }

  public onVillageChoose(): void {
    this.village.selected = true;
  }

  public scrollToTopOfCardList(): void {
    this.desktopCardList.nativeElement.scroll({
      left: 0,
      top: 0,
      behavior: 'smooth'
    });
  }

  public isNotSelected$(poi: Poi): Observable<boolean> {
    return this.focusedPoi$.pipe(map(focusedPoi => focusedPoi && poi.id !== focusedPoi.id));
  }

  public trackByWaypoint(_: number, waypoint: Waypoint): string {
    return waypoint.id;
  }

  public trackByPoi(_: number, poi: Poi): string {
    return poi.id;
  }

  public onClickPoiCard(poi: Poi): void {
    this.analyticsService.pushTag(
      new MoreDetailsVillageFromPoiClickAnalyticsEvent(this.village.waypoint.label, poi.type)
    );

    this.focusedPoi$.next(poi);
  }

  /**
   * Build an array of all the POIs of the village
   *
   * @returns The list of Human and Parcours POIs
   */
  private allPois(): Poi[] {
    const allPois: Poi[] = [];

    if (this.village.teamPoi) allPois.push(this.village.teamPoi);

    if (this.village.positionablePois) allPois.push(...this.village.positionablePois);

    return allPois;
  }
}
