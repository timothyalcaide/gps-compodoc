import { Component, Inject, OnInit } from '@angular/core';
import { MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Domain } from '@clients/adventure/bean/waypoint';
import { AdventureService } from '@core/services';
import { MoreDomain } from '@features/map/models';
import { AdventureType } from '@models/adventure-type/adventure-type';
import { CursusFilters } from '@models/filters/cursus-filters/cursus-filters';
import { Filters } from '@models/filters/filters';
import { LocalisationFilters } from '@models/filters/localisation-filters/localisation-filters';
import { SandwichFilters } from '@models/filters/sandwich-filters/sandwich-filters';
import { SituationFilters } from '@models/filters/situation-filters/situation-filters';
import { TranslateService } from '@ngx-translate/core';
import { InstanceService } from '@services/instance.service';
import { BehaviorSubject, combineLatest, Observable } from 'rxjs';
import { map } from 'rxjs/operators';

@Component({
  selector: 'app-more-domain-dialog',
  templateUrl: './more-domain-dialog.component.html'
})
export class MoreDomainDialogComponent implements OnInit {
  public filteredDomains$: Observable<Domain[]>;

  private searchText$ = new BehaviorSubject<string>('');

  constructor(
    @Inject(MAT_DIALOG_DATA) private moreDomain: MoreDomain,
    private adventureService: AdventureService,
    private instanceService: InstanceService,
    private translateService: TranslateService
  ) {}

  public ngOnInit(): void {
    this.filteredDomains$ = combineLatest({
      domains: this.loadDomains$(),
      searchText: this.searchText$.asObservable()
    }).pipe(
      map(({ domains, searchText }) =>
        searchText
          ? domains.filter(
              domain =>
                !domain.label || domain.label.toLowerCase().includes(searchText.toLowerCase())
            )
          : domains
      )
    );
  }

  public domainLabel(domain: Domain): string {
    return domain.label || this.translateService.instant('sign.various');
  }

  public trackByDomain(index: number, domain: Domain): string {
    return domain.id || '0';
  }

  public onSearch(text: string): void {
    this.searchText$.next(text);
  }

  private loadDomains$(): Observable<Domain[]> {
    const filters: Filters = new Filters(
      new CursusFilters(),
      new LocalisationFilters(),
      new SituationFilters(),
      new SandwichFilters(),
      3,
      this.moreDomain.domainCount,
      4
    );

    if (this.adventureService.instance.adventureType === AdventureType.Internship)
      filters.isInternship = true;

    return this.instanceService
      .expand(
        [],
        this.moreDomain.stage,
        this.adventureService.instance.situation.gender,
        filters,
        this.moreDomain.originDomains[0].type
      )
      .pipe(map(expd => expd[`${this.moreDomain.originDomains[0].type}s`].domains));
  }
}
