import { Component, OnInit } from '@angular/core';
import { ApplicationContext, LoaderService, MenuEntriesService } from '@core/services';
import { MenuEntry } from '@models/menu-entry/menu-entry';
import { Observable } from 'rxjs';

@Component({
  selector: 'app-navbar-expand',
  templateUrl: './navbar-expand.component.html',
  styleUrls: ['../navbars.component.scss']
})
export class NavbarExpandComponent implements OnInit {
  public applicationLogo: string;
  public isLoading$: Observable<boolean>;
  public menuEntries$: Observable<MenuEntry[]>;

  constructor(
    private context: ApplicationContext,
    private loaderService: LoaderService,
    private menuEntriesService: MenuEntriesService
  ) {}

  public ngOnInit(): void {
    this.applicationLogo = this.context.application?.logo;
    this.isLoading$ = this.loaderService.isLoading$;
    this.menuEntries$ = this.menuEntriesService.menuEntries$;
  }
}
