import { Component, EventEmitter, HostBinding, Input, OnInit, Output } from '@angular/core';
import { Poi } from '@clients/poi/poi';
import { PoiType } from '@models/poi/poi-type';
import { TabId } from '@models/tab-ids/tab-ids';

@Component({
  selector: 'app-poi-card',
  templateUrl: './poi-card.component.html',
  styleUrls: ['./poi-card.component.scss']
})
export class PoiCardComponent implements OnInit {
  @Input() public poi: Poi;

  /**
   * Whether the card has thin padding (16px instead of 24px)
   */
  @Input() public thinPadding = false;

  /**
   * Whether the card has a link to lead to a specific tab or element
   */
  @Input() public showLink = true;

  /**
   * Whether the poi is not focused
   */
  @Input() public notFocusedPoi = false;

  /**
   * Whether the icon is a big gendered human icon
   */
  @Input() public humanGenderedIcon = false;

  /**
   * Whether the label contains a link to a social network (only for Human POI)
   */
  @Input() public withSocialNetworkLink = false;

  /**
   * The typography class of the card label
   */
  @Input() public labelTypoClass = 'mat-title';

  /**
   * The number of lines for the description. If 0, the description has no limit
   */
  @Input() public descriptionLineNumber: 0 | 2 | 3 = 2;

  /**
   * Event triggered when the card is clicked
   * When used, the card default link (showLink) is disabled and a ng-content with the label action-button-label should be given
   */
  @Output() public readonly poiCardClick = new EventEmitter<void>();

  /**
   * Whether the card has full height (used in list of cards)
   */
  @HostBinding('class.full-height') @Input() public fullHeight = true;

  public readonly TabId = TabId;
  public readonly PoiType = PoiType;

  public queryParams: { details?: 'opened'; tabId?: TabId } = {};
  public fragment = null;

  public ngOnInit(): void {
    this.showLink = this.showLink && !this.poiCardClick.observed;

    if (this.showLink) {
      this.initFragment();
      this.initQueryParams();
    }
  }

  private initQueryParams(): void {
    switch (this.poi.type) {
      case PoiType.Chat:
        this.queryParams = { details: 'opened', tabId: TabId.Team };
        break;
      case PoiType.Event:
        this.queryParams = { details: 'opened', tabId: TabId.Events };
        break;
      case PoiType.Parcoursup:
        this.queryParams = { details: 'opened', tabId: TabId.Schools };
        break;
      case PoiType.Mooc:
        this.queryParams = { details: 'opened', tabId: TabId.Moocs };
        break;
      default:
        this.queryParams = { details: 'opened', tabId: TabId.Cursus };
        break;
    }
  }

  private initFragment(): void {
    switch (this.poi.type) {
      case PoiType.Chat:
        this.fragment = `human-${this.poi.id}`;
        break;
      case PoiType.Event:
        this.fragment = `event-${this.poi.id}`;
        break;
      case PoiType.Parcoursup:
        this.fragment = 'schools-tab-top';
        break;
      case PoiType.Mooc:
        this.fragment = `mooc-${this.poi.id}`;
        break;
      default:
        this.fragment = `cursus-${this.poi.id}`;
        break;
    }
  }
}
