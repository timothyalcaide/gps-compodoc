import { ADMIN_ROUTE } from '@admin/admin.routing';
import { Component, Input, OnInit } from '@angular/core';
import { EXTERNAL_LINKS_CONFIG } from '@configs/external-links.config';
import { GAMIFICATION_ENABLED_CONFIG } from '@configs/gamification-enabled.config';
import { AdminRouteUrls } from '@core/routing/admin/route-urls';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { ApplicationContext, AuthService, RewardService } from '@core/services';
import { environment } from 'environments/environment';
import { Observable } from 'rxjs';

@Component({
  selector: 'app-profile-list',
  templateUrl: './profile-list.component.html',
  styleUrls: ['./profile-list.component.scss']
})
export class ProfileListComponent implements OnInit {
  @Input() public menu = false;

  public readonly AppRouteUrls = AppRouteUrls;
  public readonly AdminRouteUrls = AdminRouteUrls;
  public rewardsWonNb$: Observable<number>;
  public hasRewards: boolean;
  public hasAccess: boolean;
  public externalLinks: boolean;
  public isAuthenticated = this.auth.isAuthenticated;

  public get editPasswordUrl(): string {
    return `${environment.keycloak.issuer}/account/password`;
  }

  constructor(
    public auth: AuthService,
    private context: ApplicationContext,
    private rewardService: RewardService
  ) {}

  public ngOnInit(): void {
    if (this.isAuthenticated) {
      this.rewardsWonNb$ = this.rewardService.rewardsWonNb();
      this.hasRewards = GAMIFICATION_ENABLED_CONFIG(
        this.context.product,
        this.context.applicationName
      );
      this.hasAccess = this.auth.user?.hasAccess(
        this.context.applicationName,
        ADMIN_ROUTE.data.roles,
        ADMIN_ROUTE.data.groups
      );
    }

    this.externalLinks = EXTERNAL_LINKS_CONFIG(this.context.product);
  }

  public logout(): void {
    this.auth.logout();
  }
}
