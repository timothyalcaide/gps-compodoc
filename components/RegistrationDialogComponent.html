import { Component, Inject, OnDestroy, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Router } from '@angular/router';
import { AUTHENTICATION_ENABLED_CONFIG } from '@configs/authentication.config';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { ApplicationContext, AuthService, LoaderService } from '@core/services';
import { NameValidator } from '@directives/custom-validators/name-validator.directive';
import { ApplicationName } from '@models/context/application-name';
import { Product } from '@models/context/product';
import { Instance } from '@models/instance/instance';
import { UaaService } from '@services/uaa.service';
import { BehaviorSubject, Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

@Component({
  selector: 'app-registration-dialog',
  templateUrl: './registration-dialog.component.html',
  styleUrls: ['./registration-dialog.component.scss']
})
export class RegistrationDialogComponent implements OnInit, OnDestroy {
  public readonly IS_SHOW_PRODUCT: boolean;
  public readonly AUTHENTICATION_ENABLED: boolean;
  public readonly IS_MY_ROAD: boolean;

  public isLoading$ = new BehaviorSubject<boolean>(false);
  public registrationWithoutPasswordForm: FormGroup;

  private onDestroySubject$: Subject<void> = new Subject<void>();

  public get emojis(): string {
    return this.data.emojis || 'ðŸ‘Œ';
  }

  public get returnUrl(): string {
    // After login, brings the user to roadmap if asked or back to this page
    if (this.data.beforeRoadmap) return `/${AppRouteUrls.ADVENTURE_ROADMAP(this.data.instance.id)}`;

    return this.router.routerState.snapshot.url;
  }

  public get isAuthenticated(): boolean {
    return this.authService.isAuthenticated;
  }

  constructor(
    @Inject(MAT_DIALOG_DATA)
    public data: {
      /**
       * The translation key of the dialog title
       */
      titleKey: string;
      /**
       * The translation params for the dialog title
       */
      titleParams?: { [key: string]: unknown };
      /**
       * The translation key of the dialog subtitle
       */
      subtitleKey?: string;
      /**
       * The translation params for the dialog subtitle
       */
      subtitleParams?: { [key: string]: unknown };
      /**
       * The dialog emojis
       */
      emojis?: string;
      /**
       * The adventure instance. Only used for registration dialog before roadmap
       */
      instance?: Instance;
      /**
       * Whether the registration dialog is opened before the roadmap or not
       */
      beforeRoadmap?: boolean;
    },
    private authService: AuthService,
    private context: ApplicationContext,
    private dialog: MatDialogRef<RegistrationDialogComponent>,
    private formBuilder: FormBuilder,
    private loaderService: LoaderService,
    private router: Router,
    private uaaService: UaaService
  ) {
    this.IS_SHOW_PRODUCT = this.context.product === Product.Show;
    this.IS_MY_ROAD = this.context.applicationName === ApplicationName.MyRoad;
    this.AUTHENTICATION_ENABLED = AUTHENTICATION_ENABLED_CONFIG(
      this.context.product,
      this.context.applicationName
    );
  }

  public ngOnInit(): void {
    const user = this.authService.userSessionStorage;

    this.registrationWithoutPasswordForm = this.formBuilder.group({
      email: [user?.email || '', [Validators.required, Validators.email]],
      firstName: [user?.firstName || '', [Validators.required, NameValidator()]],
      lastName: [user?.lastName || '', [Validators.required, NameValidator()]],
      lEtudiantOptIn: ''
    });
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public onSubmit(): void {
    // Register the user without password
    this.loaderService
      .loadingObservable(
        this.uaaService.checkIfUserExistsAndGetOrCreate(
          this.registrationWithoutPasswordForm.value.email,
          this.registrationWithoutPasswordForm.value.firstName,
          this.registrationWithoutPasswordForm.value.lastName,
          this.IS_MY_ROAD ? !!this.registrationWithoutPasswordForm.value.lEtudiantOptIn : null
        ),
        this.isLoading$
      )
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe(user => {
        if (user) {
          /**
           * When the user is successfully registered, finalize the contact request even if he is
           * not authenticated
           */
          this.authService.userSessionStorage = user;

          this.dialog.close(user);
        }
      });
  }
}
