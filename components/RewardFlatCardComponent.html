import { Component, Input, OnDestroy, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { UserScore } from '@clients/adventure/bean/user-score';
import { Waypoint } from '@clients/adventure/bean/waypoint';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { AuthService, DialogService, RewardService } from '@core/services';
import { Reward } from '@models/reward/reward';
import { User } from '@models/user/user';
import { RegistrationDialogComponent } from '@organisms/registration-dialog/registration-dialog.component';
import { SuccessDialogComponent } from '@organisms/success-dialog/success-dialog.component';
import { Observable, Subject } from 'rxjs';
import { map, takeUntil, tap } from 'rxjs/operators';

@Component({
  selector: 'app-reward-flat-card',
  templateUrl: './reward-flat-card.component.html',
  styleUrls: ['./reward-flat-card.component.scss']
})
export class RewardFlatCardComponent implements OnInit, OnDestroy {
  @Input() public waypoint: Waypoint;

  public readonly AppRouteUrls = AppRouteUrls;

  public reward$: Observable<Reward>;
  public hasClaim$: Observable<boolean>;

  private rewardId: string;

  private onDestroySubject$: Subject<void> = new Subject<void>();

  public get userScore(): UserScore {
    return this.rewardService.userScore;
  }

  constructor(
    private authService: AuthService,
    private dialog: MatDialog,
    private dialogService: DialogService,
    private rewardService: RewardService
  ) {}

  public ngOnInit(): void {
    // Load the student interview reward
    this.reward$ = this.rewardService.studentInterviewReward().pipe(
      // Load the user claimed associated to this reward
      tap((reward: Reward) => {
        this.hasClaim$ = this.rewardService.hasClaimReward(reward.id);
        this.rewardId = reward.id;
      })
    );
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  /**
   * Start the user claim saving process
   * If the user is authenticated it is directly achieved
   * Otherwise, it ask the user for registration (or login)
   */
  public onRegister(): void {
    if (!this.authService.hasEmail) {
      this.dialog
        .open(
          RegistrationDialogComponent,
          this.dialogService.defaultDialogConfig({
            titleKey: 'dialog.registration.reward.title',
            subtitleKey: 'dialog.registration.reward.subtitle'
          })
        )
        .afterClosed()
        .pipe(takeUntil(this.onDestroySubject$))
        .subscribe((user: User) => {
          /**
           * When the user is successfully registered, save its claim even if he is not
           * authenticated. Dialog closing does not happen if the user logins
           */
          if (user) {
            this.authService.userSessionStorage = user;
            this.claimReward(this.rewardId);
          }
        });
    } else this.claimReward(this.rewardId);
  }

  /**
   * Save a user claim to the reward service
   */
  private claimReward(rewardId: string): void {
    this.hasClaim$ = this.rewardService.claimReward(rewardId).pipe(map(userClaim => !!userClaim));
    this.successDialog();
  }

  private successDialog(): void {
    this.dialog.open(
      SuccessDialogComponent,
      this.dialogService.simpleDialogConfig({ subtitleKey: 'success.message' })
    );
  }
}

    <mat-icon
      [svgIcon]="userScore.points < data.reward?.points ? 'chest-close' : 'chest-open'"
      class="icon-xl"
    ></mat-icon>
    <mat-card-title translate [translateParams]="{ name: waypoint.label }">
      rewards.contact-card.title
    </mat-card-title>

    <ng-container *ngIf="userScore.points < data.reward?.points; else wonReward">
      <div
        class="mb-16"
        translate
        [translateParams]="{ points: data.reward.points - userScore.points }"
      >
        rewards.contact-card.message-miss
      </div>

      <div class="left medium-grey-2 mb-8">{{ userScore.points }}/{{ data.reward?.points }}</div>
      <mat-progress-bar
        color="light-yellow"
        value="{{ (userScore.points / data.reward?.points) * 100 }}"
      ></mat-progress-bar>

      <mat-card-actions class="centered-card-actions">
        <a [routerLink]="'/' + AppRouteUrls.REWARDS_REWARD(data.reward.id)" translate>
          see-detail
        </a>
      </mat-card-actions>
    </ng-container>

    <ng-template #wonReward>
      <ng-container *ngIf="!data.hasClaim; else claimedReward">
        <div *ngIf="!data.hasClaim" translate>rewards.contact-card.message-win</div>

        <mat-card-actions class="centered-card-actions">
          <button mat-flat-button type="button" color="dark" (click)="onRegister()">
            {{ "rewards.contact-card.contact-btn" | translate }}
          </button>
        </mat-card-actions>
      </ng-container>
