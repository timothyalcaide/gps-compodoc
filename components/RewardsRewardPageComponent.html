import { Component, OnDestroy, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { ActivatedRoute } from '@angular/router';
import { UserScore } from '@clients/adventure/bean/user-score';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import {
  AuthService,
  DialogService,
  PageTitleService,
  RewardService,
  SituationContext
} from '@core/services';
import { ORIGIN_QUERY_STRING } from '@models/adventure/adventure-query-string';
import { LeadReward } from '@models/lead/lead-reward';
import { Reward } from '@models/reward/reward';
import { RewardType } from '@models/reward/reward-type';
import { User } from '@models/user/user';
import { WaypointType } from '@models/waypoint/waypoint-type';
import { RegistrationDialogComponent } from '@organisms/registration-dialog/registration-dialog.component';
import { RewardHowToDialogComponent } from '@organisms/reward-how-to-dialog/reward-how-to-dialog.component';
import { RewardPartnerDialogComponent } from '@organisms/reward-partner-dialog/reward-partner-dialog.component';
import { BehaviorSubject, Observable, Subject } from 'rxjs';
import { filter, mergeMap, takeUntil, tap } from 'rxjs/operators';

@Component({
  selector: 'app-page-rewards-reward',
  templateUrl: './rewards-reward.component.html',
  styleUrls: ['./rewards-reward.component.scss']
})
export class RewardsRewardPageComponent implements OnInit, OnDestroy {
  public readonly AppRouteUrls = AppRouteUrls;
  public readonly RewardType = RewardType;
  public readonly WaypointType = WaypointType;
  public reward$: Observable<Reward>;
  public leadReward$ = new BehaviorSubject<LeadReward>(null);
  public adventureOriginId: string;

  private rewardId: string;
  private onDestroySubject$: Subject<void> = new Subject<void>();

  public get userScore(): UserScore {
    return this.rewardService.userScore;
  }

  constructor(
    private authService: AuthService,
    private dialog: MatDialog,
    private dialogService: DialogService,
    private pageTitle: PageTitleService, // NOTE : Do not delete, used for title resolution
    private rewardService: RewardService,
    private situationContext: SituationContext,
    private route: ActivatedRoute
  ) {}

  public ngOnInit(): void {
    this.rewardId = this.route.snapshot.params.id;

    this.reward$ = this.rewardService.reward(this.rewardId);

    this.rewardService
      .getClaimReward(this.rewardId)
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe((leadReward: LeadReward) => {
        this.leadReward$.next(leadReward);
      });

    const { situation } = this.route.snapshot.data;

    if (situation) this.situationContext.saveToSituationContext(situation);

    this.adventureOriginId = this.route.snapshot.queryParams[ORIGIN_QUERY_STRING];
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public score(points: number, userScore: UserScore): number {
    if (!userScore) return 0;

    if (userScore.points > points) return points;

    return userScore.points;
  }

  /**
   * Start the lead reward saving process
   * If the user is authenticated it is directly achieved
   * Otherwise, it ask the user for registration (or login)
   *
   * @param reward The reward
   */
  public onConfirm(reward: Reward): void {
    if (!this.authService.hasEmail) {
      this.dialog
        .open(
          RegistrationDialogComponent,
          this.dialogService.defaultDialogConfig({
            titleKey: 'dialog.registration.reward.title',
            subtitleKey: 'dialog.registration.reward.subtitle'
          })
        )
        .afterClosed()
        .pipe(
          filter(user => !!user),
          mergeMap((user: User) => {
            /**
             * When the user is successfully registered, save its lead even if he is not
             * authenticated. Dialog closing does not happen if the user logins
             */
            this.authService.userSessionStorage = user;

            return this.rewardService.claimReward(this.rewardId);
          }),
          tap(leadReward => {
            this.onOpenCouponCodeDialog(reward, leadReward);
          }),
          takeUntil(this.onDestroySubject$)
        )
        .subscribe(this.leadReward$);
    } else
      this.rewardService
        .claimReward(this.rewardId)
        .pipe(
          tap(leadReward => {
            this.onOpenCouponCodeDialog(reward, leadReward);
          }),
          takeUntil(this.onDestroySubject$)
        )
        .subscribe(this.leadReward$);
  }

  public openHowToWinPoints(): void {
    this.dialog.open(RewardHowToDialogComponent, this.dialogService.defaultDialogConfig());
  }

  public isWon(rewardPoints: number): boolean {
    return this.userScore.points >= rewardPoints;
  }

  /**
   * Open the coupon code dialog if the reward is a coupon code reward
   *
   * @param reward The reward
   * @param leadReward The lead reward
   */
  public onOpenCouponCodeDialog(reward: Reward, leadReward: LeadReward): void {
    if (
      leadReward &&
      ((reward.type === RewardType.COUPON_CODE && leadReward.couponCode) || reward.partnerUrl)
    ) {
      this.dialog.open(
        RewardPartnerDialogComponent,
        this.dialogService.simpleDialogConfig({
          reward,
          leadReward
        })
      );
    }
  }

  public getRewardClaimBtnLabelKey(rewardType: RewardType): string {
    switch (rewardType) {
      case RewardType.ADMISSION_INTERVIEW:
      case RewardType.STUDENT_INTERVIEW:
        return 'rewards.reward-details.be-contacted';
      case RewardType.IMMERSION_DAY:
        return 'rewards.reward-details.participate-event';
      case RewardType.COUPON_CODE:
        return 'rewards.reward-details.claim-coupon-code';
      default:
        return 'rewards.reward-details.claim';
    }
  }

  public getRewardShowBtnLabelKey(rewardType: RewardType): string {
    switch (rewardType) {
      case RewardType.COUPON_CODE:
        return 'rewards.reward-details.show-coupon-code';
      default:
        return 'rewards.reward-details.show-reward';
    }
  }
}

    {{ score(reward.points, userScore) }}/{{ reward.points }} pts
  </div>
  <mat-progress-bar

    value="{{ (userScore.points / reward.points) * 100 }}"
  >
  </mat-progress-bar>

  <div class="mb-8 center">

      {{ "rewards.reward-details.win" | translate }}
    </ng-container>

    <ng-template #notWonText>
      {{ "rewards.reward-details.still-label-1" | translate }}
      <span class="mat-body-strong">
        {{
          "rewards.reward-details.still-label-2"
            | translate: { points: reward.points - userScore.points }
        }}
      </span>
      {{ "rewards.reward-details.still-label-3" | translate }}
    </ng-template>
  </div>

  <div class="mat-title center">{{ reward.title }}</div>

  <div *ngIf="reward.description" class="mb-24">{{ reward.description }}</div>

  <ng-container *ngIf="reward.partnerLogo">
    <div class="mb-8" translate>reward.partner</div>
    <img class="partner-image mb-24" [src]="reward.partnerLogo" alt="Partner logo" />
  </ng-container>

  <a
    *ngIf="!isWon(reward.points); else wonReward"
    class="mb-24"
    type="button"
    (click)="openHowToWinPoints()"
    translate
  >
    rewards.how-to
  </a>

  <ng-template #wonReward>
    <ng-container *ngIf="leadReward$ | async as leadReward; else notClaimedReward">
      <div
        *ngIf="reward.type !== RewardType.COUPON_CODE && !reward.partnerUrl; else showReward"
        class="mat-body-strong center medium-grey-2"
        translate
      >
         <span class="pre-wrap" translate>rewards.reward-details.been-contacted</span>
      </div>

      <ng-template #showReward>
        <button
          mat-flat-button
          color="primary"
          type="button"
          (click)="onOpenCouponCodeDialog(reward, leadReward)"
        >
          {{ getRewardShowBtnLabelKey(reward.type) | translate }}
        </button>
      </ng-template>
    </ng-container>

    <ng-template #notClaimedReward>
      <button mat-flat-button color="primary" type="button" (click)="onConfirm(reward)">
        {{ getRewardClaimBtnLabelKey(reward.type) | translate }}
      </button>
    </ng-template>
  </ng-template>

  <div class="mt-24">
    <a
      *ngIf="adventureOriginId"
      mat-flat-button
      color="primary"
      [routerLink]="
        '/' + AppRouteUrls.INTENTIONS_WAYPOINT_DETAILS(WaypointType.diploma, adventureOriginId)
      "
    >
      {{ "intentions.start" | translate }}
    </a>
  </div>
</app-simple-template>
