import { ViewportRuler } from '@angular/cdk/scrolling';
import {
  AfterViewInit,
  Component,
  ElementRef,
  EventEmitter,
  Input,
  OnDestroy,
  OnInit,
  Output,
  TemplateRef,
  ViewChild
} from '@angular/core';
import { BreakpointService } from '@core/services';
import { TabId } from '@models/tab-ids/tab-ids';
import { BehaviorSubject, fromEvent, Observable, Subject } from 'rxjs';
import { debounceTime, map, takeUntil } from 'rxjs/operators';

@Component({
  selector: 'app-scroll-wrapper',
  templateUrl: './scroll-wrapper.component.html',
  styleUrls: ['./scroll-wrapper.component.scss']
})
export class ScrollWrapperComponent<T> implements OnInit, AfterViewInit, OnDestroy {
  @ViewChild('wrapper', { static: true }) public wrapper: ElementRef;

  /**
   * Find the ng-template reference provided to the ScrollWrapper component in order to load it into
   * the dynamic ng-template in the scroll wrapper (see ScrollWrapper template - HTML -)
   */
  @Input() public templateRef: TemplateRef<T>;

  @Input() public objects: T[];
  @Input() public seeAllQueryParams: { tabId?: TabId } = {};
  @Input() public seeAllFragment = '';

  /**
   * If the scroll wrapper is contained in a small space, like the village details.
   * Disable arrows and change media width.
   */
  @Input() public smallContainer = false;

  /**
   * An optional URL to provide when it is necessary to override the './' route from Angular Routing
   * (ex : in a dialog)
   */
  @Input() public currentUrl = './';

  /**
   * If the scroll wrapper contains images or others media.
   * Used to adapt the style of the component like centering or removing margins
   */
  @Input() public hasMedias = false;

  /**
   * Event raised when the See all link of the See all card are clicked
   */
  @Output() public readonly seeAllClick = new EventEmitter<void>();

  public readonly isDesktop$: Observable<boolean>;
  public hasSingleMedia$: BehaviorSubject<boolean>;
  public isPreviousArrowHidden$: Observable<boolean>;
  public isNextArrowHidden$: Observable<boolean>;
  private onDestroySubject$ = new Subject<void>();

  constructor(private breakpoint: BreakpointService, private viewportRuler: ViewportRuler) {
    this.isDesktop$ = this.breakpoint.isDesktop$;
  }

  public ngOnInit(): void {
    if (this.hasMedias && this.objects.length === 1) {
      if (this.smallContainer) this.hasSingleMedia$ = new BehaviorSubject<boolean>(true);
      else {
        this.hasSingleMedia$ = this.breakpoint.isDesktop$.pipe(
          map(isDesktop => !isDesktop)
        ) as BehaviorSubject<boolean>;
      }
    } else {
      this.hasSingleMedia$ = new BehaviorSubject<boolean>(false);
    }

    this.isPreviousArrowHidden$ = fromEvent(this.wrapper.nativeElement, 'scroll').pipe(
      debounceTime(100),
      map((e: MouseEvent) => (e.target as Element).scrollLeft === 0)
    );

    this.isNextArrowHidden$ = fromEvent(this.wrapper.nativeElement, 'scroll').pipe(
      debounceTime(100),
      map((e: MouseEvent) => {
        const { scrollWidth, clientWidth, scrollLeft } = e.target as Element;

        return scrollLeft + clientWidth === scrollWidth || clientWidth === scrollWidth;
      })
    );
  }

  public ngAfterViewInit(): void {
    // Used to initialize the scroll arrows
    (this.wrapper.nativeElement as Element).dispatchEvent(new CustomEvent('scroll'));

    this.viewportRuler
      .change(400)
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe(() => {
        (this.wrapper.nativeElement as Element).dispatchEvent(new CustomEvent('scroll'));
      });
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }
}
