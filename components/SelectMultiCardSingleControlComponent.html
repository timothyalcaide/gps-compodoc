import { Component, forwardRef, Input, OnInit } from '@angular/core';
import { ControlValueAccessor, NG_VALUE_ACCESSOR } from '@angular/forms';
import { combineLatest, Observable } from 'rxjs';
import { map } from 'rxjs/operators';
import { SelectMultiCardControlComponent } from '../select-multi-card-control.component';

@Component({
  selector: 'app-select-multi-card-single-control',
  templateUrl: './select-multi-card-single-control.component.html',
  styleUrls: [
    '../select-multi-card-control.component.scss',
    './select-multi-card-single-control.component.scss'
  ],
  providers: [
    // SelectMultiCardSingleControlComponent is a custom form control
    {
      provide: NG_VALUE_ACCESSOR,
      useExisting: forwardRef(() => SelectMultiCardSingleControlComponent),
      multi: true
    }
  ]
})
export class SelectMultiCardSingleControlComponent
  extends SelectMultiCardControlComponent<string>
  implements ControlValueAccessor, OnInit
{
  @Input() public cardsByLineInDesktop = 2;

  public cardsByLine$: Observable<number>;

  protected value = '';

  private readonly CARDS_BY_LINE_IN_MOBILE = 1;
  private readonly CARDS_BY_LINE_IN_TABLET = 2;

  public get oneOptionSelected(): boolean {
    return !!this.value;
  }

  public ngOnInit(): void {
    this.cardsByLine$ = combineLatest({
      isMobile: this.breakpointService.isMobile$,
      isTablet: this.breakpointService.isTablet$
    }).pipe(
      map(({ isMobile, isTablet }) => {
        if (isMobile) return this.CARDS_BY_LINE_IN_MOBILE;

        if (isTablet) return this.CARDS_BY_LINE_IN_TABLET;

        return this.cardsByLineInDesktop;
      })
    );
  }

  /**
   * Event occurring when a card is clicked to toggle its selection state
   */
  public onToggle(optionValue: string): void {
    super.onToggle(optionValue);

    // Toggle the value
    if (this.value === optionValue) this.value = '';
    else this.value = optionValue;

    this.onChange(this.value);
  }

  public writeValue(value: string): void {
    this.value = value;
  }

  public isSelected(optionValue: string): boolean {
    return this.value === optionValue;
  }
}
