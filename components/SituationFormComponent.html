import { Component, EventEmitter, OnInit, Output } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import {
  bacSpecialties,
  BacSpecialty,
  GenderEmoji,
  HighSchoolClass,
  HighSchoolType,
  highSchoolTypes,
  Status
} from '@clients/adventure/bean/situation';
import { handleBacStatus, statusKeyLabel } from '@clients/adventure/bean/situation.helper';
import { AnalyticsEvent } from '@core/models/analytics-event/analytics-event';
import { AnalyticsEventType } from '@core/models/analytics-event/analytics-event-type';
import { AnalyticsService, SituationContext } from '@core/services';
import { SituationValidator } from '@directives/custom-validators/situation-validator.directive';
import { OnBoardingFinalizeBacSpecialtyAnalyticsEvent } from '@models/analytics-event/on-boarding-finalize-bac-specialty.analytics-event';
import { Gender } from '@models/gender/gender';
import { SelectMultiCardOption } from '@models/select-multi-card-option/select-multi-card-option';
import { Situation } from '@models/situation/situation';
import { TranslateService } from '@ngx-translate/core';
import { SituationService } from '@services/situation/situation.service';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';

@Component({
  selector: 'app-situation-form',
  templateUrl: './situation-form.component.html',
  styleUrls: ['./situation-form.component.scss']
})
export class SituationFormComponent implements OnInit {
  @Output() public readonly saveSituation = new EventEmitter<Situation>();

  public situation: Situation = new Situation();
  public situationForm: FormGroup;
  public statuses$: Observable<SelectMultiCardOption[]>;
  public diplomaLevels$: Observable<SelectMultiCardOption[]>;

  public get genders(): SelectMultiCardOption[] {
    return [
      ...Object.values(Gender).map(gender => ({
        value: gender,
        label: this.translateService.instant(`situations.gender.sexe.${gender}`),
        emoji: GenderEmoji[gender]
      }))
    ];
  }

  public get highschoolClasses(): SelectMultiCardOption[] {
    return Object.values(HighSchoolClass).map(highschoolClass => ({
      value: highschoolClass,
      label: this.translateService.instant(`situations.highschool.${highschoolClass}`)
    }));
  }

  public get highschoolTypes(): SelectMultiCardOption[] {
    return highSchoolTypes(this.situation.highSchoolClass).map(highschoolType => ({
      value: highschoolType,
      label: this.translateService.instant(`situations.highSchoolType.${highschoolType}`)
    }));
  }

  public get bacSpecialties(): SelectMultiCardOption[] {
    return bacSpecialties(this.situation.highSchoolType).map(bacSpecialty => ({
      value: bacSpecialty,
      label: this.translateService.instant(`situations.specialities.${bacSpecialty}`)
    }));
  }

  private get statusOptions(): Observable<SelectMultiCardOption[]> {
    return this.situationService.statusesGlobal().pipe(
      map(statuses =>
        Object.values(statuses).map(status => ({
          value: status,
          label: this.translateService.instant(statusKeyLabel(status, this.situation.gender)),
          icon: `${handleBacStatus(status)}-${this.situation.gender.toString()}`
        }))
      )
    );
  }

  constructor(
    private analyticsService: AnalyticsService,
    private formBuilder: FormBuilder,
    private situationContext: SituationContext,
    private situationService: SituationService,
    private translateService: TranslateService
  ) {}

  public ngOnInit(): void {
    this.situation = new Situation(
      this.situationContext.situation?.id || '',
      this.situationContext.situation?.gender || null,
      this.situationContext.situation?.status || null,
      this.situationContext.situation?.highSchoolClass || null,
      this.situationContext.situation?.highSchoolType || null,
      this.situationContext.situation?.bacSpecialties || []
    );

    if (this.situation.gender) this.statuses$ = this.statusOptions;

    this.diplomaLevels$ = this.situationService.statusesBac().pipe(
      map(diplomaLevels =>
        Object.values(diplomaLevels).map(diplomaLevel => ({
          value: diplomaLevel,
          label: this.translateService.instant(`situations.status.${diplomaLevel}`)
        }))
      )
    );

    this.initForm();
    this.onGenderChanges();
    this.onStatusChanges();
    this.onDiplomaLevelChanges();
    this.onHighschoolClassChanges();
    this.onHighschoolTypeChanges();
    this.onBacSpecialtiesChanges();
  }

  public onSubmit(): void {
    this.saveSituation.emit(this.situation);

    this.situation.bacSpecialties.forEach(bacSpecialty => {
      this.analyticsService.pushTag(
        new OnBoardingFinalizeBacSpecialtyAnalyticsEvent(
          this.translateService.instant(`situations.specialities.${bacSpecialty}`)
        )
      );
    });
  }

  private initForm(): void {
    this.situationForm = this.formBuilder.group(
      {
        gender: [this.situationContext.situation?.gender || null, Validators.required],
        status: [
          this.situationContext.situation?.gender
            ? this.situationContext.situation?.globalStatus
            : null,
          Validators.required
        ],
        diplomaLevel: [
          this.situationContext.situation?.hasBac ? this.situationContext.situation?.status : null
        ],
        highschoolClass: [
          this.situationContext.situation?.isHighSchool
            ? this.situationContext.situation?.highSchoolClass
            : null
        ],
        highschoolType: [
          this.situationContext.situation?.hasHighSchoolClass
            ? this.situationContext.situation?.highSchoolType
            : null
        ],
        bacSpecialties: [
          this.situationContext.situation?.numberOfRequiredBacSpecialties
            ? [...(this.situationContext.situation?.bacSpecialties || [])]
            : []
        ]
      },
      { validators: SituationValidator() }
    );
  }

  /**
   * Listen gender changes
   */
  private onGenderChanges(): void {
    this.situationForm.controls.gender.valueChanges.subscribe((newGender: Gender) => {
      this.situation.gender = newGender;
      this.situationForm.controls.status.setValue('');

      // Refresh statuses
      this.statuses$ = this.statusOptions;
    });
  }

  /**
   * Listen status changes
   */
  private onStatusChanges(): void {
    this.situationForm.controls.status.valueChanges.subscribe((newStatus: Status) => {
      this.situation.status = newStatus;

      this.situationForm.controls.highschoolClass.setValue('');
      this.situationForm.controls.diplomaLevel.setValue('');
    });
  }

  /**
   * Listen diploma level changes
   */
  private onDiplomaLevelChanges(): void {
    this.situationForm.controls.diplomaLevel.valueChanges.subscribe((newDiplomaLevel: Status) => {
      if (newDiplomaLevel) this.situation.status = newDiplomaLevel;
      // Occurs when the user unselect a diploma level but does not unselect the student status
      else if (this.situation.isStudent) this.situation.status = Status.student;

      if (newDiplomaLevel) this.pushOnBoardingComplete();
    });
  }

  /**
   * Listen highschool class changes
   */
  private onHighschoolClassChanges(): void {
    this.situationForm.controls.highschoolClass.valueChanges.subscribe(
      (newHighSchoolClass: HighSchoolClass) => {
        this.situation.highSchoolClass = newHighSchoolClass;
        this.situationForm.controls.highschoolType.setValue('');
      }
    );
  }

  /**
   * Listen highschool class changes
   */
  private onHighschoolTypeChanges(): void {
    this.situationForm.controls.highschoolType.valueChanges.subscribe(
      (newHighSchoolType: HighSchoolType) => {
        this.situation.highSchoolType = newHighSchoolType;
        this.situationForm.controls.bacSpecialties.setValue([]);

        if (newHighSchoolType && !this.situation.numberOfRequiredBacSpecialties)
          this.pushOnBoardingComplete();
      }
    );
  }

  /**
   * Listen bac specialties changes
   */
  private onBacSpecialtiesChanges(): void {
    this.situationForm.controls.bacSpecialties.valueChanges.subscribe(
      (newBacSpecialties: BacSpecialty[]) => {
        this.situation.setBacSpecialties(newBacSpecialties);

        if (newBacSpecialties.length && this.situation.hasValidBacSpecialties)
          this.pushOnBoardingComplete();
      }
    );
  }

  /**
   * Push the OnBoardingComplete tag to the Analytics service
   */
  private pushOnBoardingComplete(): void {
    this.analyticsService.pushTag(new AnalyticsEvent(AnalyticsEventType.OnBoardingComplete));
  }
}
