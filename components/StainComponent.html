import { AfterViewInit, Component, Input, OnInit } from '@angular/core';
import { Observable } from 'rxjs';
import { Utils } from '../../utils';
import { MapElementComponent } from '../map-element/map-element.component';

@Component({
  // eslint-disable-next-line @angular-eslint/component-selector
  selector: '[svg-stain]',
  templateUrl: './stain.component.html',
  styleUrls: ['./stain.component.scss']
})
export class StainComponent extends MapElementComponent implements OnInit, AfterViewInit {
  @Input() public inDistrict = false;
  @Input() public parentSvgPath: SVGPathElement = null;
  @Input() public disabled = false;

  public stainId = '';
  public isMobile$: Observable<boolean>;

  private readonly STAIN_MOBILE_POSITION_RATIO: number = 0.8;
  private readonly STAIN_TABLET_POSITION_RATIO: number = 0.75;
  private readonly STAIN_DESKTOP_POSITION_RATIO: number = 0.7;
  private readonly STAIN_IN_DISTRICT_POSITION_RATIO: number = 0.5;

  public ngOnInit(): void {
    this.stainId = `#stain-${Utils.randomBetween(1, 5)}`;
    this.isMobile$ = this.breakpointService.isMobile$;
  }

  public ngAfterViewInit(): void {
    // As position is depending on dimensions, we initialize it
    super.initDimensions();
    super.initPosition();
  }

  protected getPosition(): DOMPoint {
    if (this.inDistrict) {
      const districtRelativePosition: DOMPoint = this.parentSvgPath.getPointAtLength(
        this.parentSvgPath.getTotalLength() * this.STAIN_IN_DISTRICT_POSITION_RATIO
      );

      return new DOMPoint(
        districtRelativePosition.x - this.width / 2,
        districtRelativePosition.y - this.height / 2
      );
    }

    let stainPositionRatio: number;

    if (this.breakpointService.snapshot.isMobile)
      stainPositionRatio = this.STAIN_MOBILE_POSITION_RATIO;
    else if (this.breakpointService.snapshot.isTablet)
      stainPositionRatio = this.STAIN_TABLET_POSITION_RATIO;
    else stainPositionRatio = this.STAIN_DESKTOP_POSITION_RATIO;

    const parentRelativePoint: DOMPoint = this.parentSvgPath.getPointAtLength(
      this.parentSvgPath.getTotalLength() * stainPositionRatio
    );

    return new DOMPoint(parentRelativePoint.x - this.width / 2, parentRelativePoint.y);
  }
}
