import { Component, Input, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { ActivatedRoute, Router } from '@angular/router';
import { Application } from '@clients/adventure/bean/application';
import { isDiploma, isJob, Waypoint } from '@clients/adventure/bean/waypoint';
import { EXTERNAL_LINKS_CONFIG } from '@configs/external-links.config';
import { GAMIFICATION_ENABLED_CONFIG } from '@configs/gamification-enabled.config';
import { SHOW_APPLICATION_NAME_CONFIG } from '@configs/show-application-name.config';
import { RouteIds } from '@core/routing/route-ids';
import { ApplicationContext, DialogService } from '@core/services';
import { ApplicationName } from '@models/context/application-name';
import { Instance } from '@models/instance/instance';
import { HumanPoi } from '@models/poi/human-poi/human-poi';
import { HumanProfileDialogComponent } from '@organisms/human-profile-dialog/human-profile-dialog.component';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';

@Component({
  selector: 'app-team-full',
  templateUrl: './team.full.component.html',
  styleUrls: ['./team.full.component.scss']
})
export class TeamFullComponent implements OnInit {
  @Input() public alumniPois: HumanPoi[];
  @Input() public studentPois: HumanPoi[];
  @Input() public waypoint: Waypoint;
  @Input() public inOffer: boolean;
  @Input() public instance: Instance;
  @Input() public smallContainer = false;

  /**
   * The village stage
   */
  @Input() public stage: number;

  public readonly isDiploma = isDiploma;
  public readonly isJob = isJob;
  public application: Application;
  public SHOW_APPLICATION_NAME: boolean;
  public IS_PAYS_DE_LA_LOIRE: boolean;
  public GAMIFICATION_ENABLED: boolean;
  public isRoadmapActive$: Observable<boolean>;
  public externalLinks: boolean;

  constructor(
    private context: ApplicationContext,
    private dialog: MatDialog,
    private dialogService: DialogService,
    private route: ActivatedRoute,
    private router: Router
  ) {}

  public ngOnInit(): void {
    this.application = this.context.application;
    this.SHOW_APPLICATION_NAME = SHOW_APPLICATION_NAME_CONFIG(this.context.application);
    this.IS_PAYS_DE_LA_LOIRE = this.context.applicationName === ApplicationName.PaysDeLaLoire;
    this.GAMIFICATION_ENABLED = GAMIFICATION_ENABLED_CONFIG(
      this.context.product,
      this.context.applicationName
    );
    this.externalLinks = EXTERNAL_LINKS_CONFIG(this.context.product);

    this.isRoadmapActive$ = this.route.data.pipe(
      map(data => data.routeId === RouteIds.ADVENTURE_ROADMAP)
    );
  }

  public onOpenProfile(humanPoi: HumanPoi): void {
    this.dialog.open(
      HumanProfileDialogComponent,
      this.dialogService.defaultDialogConfig({
        humanPoiId: humanPoi.id,
        waypointId: this.waypoint.id,
        currentUrl: this.router.url,
        instance: this.instance,
        stage: this.stage
      })
    );
  }
}
