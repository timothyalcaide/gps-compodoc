import {
  AfterViewInit,
  Component,
  ElementRef,
  EventEmitter,
  Input,
  Output,
  ViewChild
} from '@angular/core';
import { Waypoint } from '@clients/adventure/bean/waypoint';
import { HOME_BANNER_IMAGE_CONFIG } from '@configs/home-banner-image.config';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { ApplicationContext, BreakpointService } from '@core/services';
import { AbstractHomeComponent } from '@pages/home/abstract-home.component';

@Component({
  selector: 'app-testing-my-road-home',
  templateUrl: './testing-my-road-home.component.html',
  styleUrls: [
    '../default-home/default-home.component.scss',
    '../my-road-home/my-road-home.component.scss',
    './testing-my-road-home.component.scss'
  ]
})
export class TestingMyRoadHomeComponent extends AbstractHomeComponent implements AfterViewInit {
  @Input() public origin: Waypoint;
  @Input() public destination: Waypoint;
  @Output() public readonly startClick = new EventEmitter<void>();

  @ViewChild('mapVideo') public mapVideoEl: ElementRef<HTMLVideoElement>;

  public readonly AppRouteUrls = AppRouteUrls;
  public readonly image: string;

  constructor(
    protected breakpointService: BreakpointService,
    protected context: ApplicationContext
  ) {
    super(breakpointService, context);

    this.image = HOME_BANNER_IMAGE_CONFIG(this.context.application);
  }

  public ngAfterViewInit(): void {
    if (this.mapVideoEl.nativeElement.paused) {
      this.mapVideoEl.nativeElement.play().catch(() => {
        // NOTE : Autoplay can be prevented, in example on Chrome
        this.mapVideoEl.nativeElement.muted = true;
        this.mapVideoEl.nativeElement.play();
      });
    }
  }
}
