import { Platform } from '@angular/cdk/platform';
import { Component, ElementRef, Input, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { Router } from '@angular/router';
import { Waypoint } from '@clients/adventure/bean/waypoint';
import { AdventureService, BreakpointService, DialogService } from '@core/services';
import { HumanPoi } from '@models/poi/human-poi/human-poi';
import { HumanProfileDialogComponent } from '@organisms/human-profile-dialog/human-profile-dialog.component';
import { MapElementComponent } from '../../map-element/map-element.component';

@Component({
  // eslint-disable-next-line @angular-eslint/component-selector
  selector: '[svg-video-ask-poi]',
  templateUrl: './video-ask-poi.component.html',
  styleUrls: ['../pois.scss', './video-ask-poi.component.scss']
})
export class VideoAskPoiComponent extends MapElementComponent implements OnInit {
  private static readonly VIDEO_ASK_POI_MOBILE_POSITION_RATIO: number = 0.7;
  private static readonly VIDEO_ASK_POI_DESKTOP_POSITION_RATIO: number = 0.65;

  @Input() public videoAskPoi: HumanPoi;
  @Input() public waypoint: Waypoint;
  @Input() public villageSvgPath: SVGPathElement = null;
  @Input() public positionIndex: number;

  /**
   * The village stage
   */
  @Input() public stage: number;

  public pinId: string;

  constructor(
    protected elRef: ElementRef<SVGGElement>,
    protected breakpointService: BreakpointService,
    private adventureService: AdventureService,
    private dialog: MatDialog,
    private dialogService: DialogService,
    private platform: Platform,
    private router: Router
  ) {
    super(elRef, breakpointService);
  }

  public ngOnInit(): void {
    this.pinId = `#poi-pin-l-big${this.platform.SAFARI ? '-no-shadow' : ''}`;

    super.initPosition();
  }

  public onClickVideoAskPoi(): void {
    this.dialog.open(
      HumanProfileDialogComponent,
      this.dialogService.defaultDialogConfig({
        humanPoiId: this.videoAskPoi.id,
        waypointId: this.waypoint.id,
        currentUrl: this.router.url,
        instance: this.adventureService.instance,
        stage: this.stage
      })
    );
  }

  protected getPosition(): DOMPoint {
    const poiPositionRatio: number = this.breakpointService.snapshot.isDesktop
      ? VideoAskPoiComponent.VIDEO_ASK_POI_DESKTOP_POSITION_RATIO
      : VideoAskPoiComponent.VIDEO_ASK_POI_MOBILE_POSITION_RATIO;

    const villageRelativePoint = this.villageSvgPath.getPointAtLength(
      this.villageSvgPath.getTotalLength() * poiPositionRatio
    );

    return new DOMPoint(
      villageRelativePoint.x + (this.positionIndex ? -136 : -150),
      villageRelativePoint.y + (this.positionIndex ? -44 : -30)
    );
  }
}
