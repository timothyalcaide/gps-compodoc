import {
  Component,
  EventEmitter,
  Input,
  OnDestroy,
  OnInit,
  Output,
  ViewChild
} from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { Application } from '@clients/adventure/bean/application';
import { isDiploma, Waypoint } from '@clients/adventure/bean/waypoint';
import {
  AdventureService,
  AnalyticsService,
  ApplicationContext,
  BreakpointService,
  DialogService
} from '@core/services';
import { NetflixLoaderDirective } from '@directives/netflix-loader/netflix-loader.directive';
import { Village } from '@features/map/models';
import { AnalyticsEventType } from '@models/analytics-event/analytics-event-type';
import { VillageNameAnalyticsEvent } from '@models/analytics-event/village-name.analytics-event';
import { Campus } from '@models/campus/campus';
import { PoiType } from '@models/poi/poi-type';
import { DiplomaSuccessDialogComponent } from '@organisms/diploma-success-dialog/diploma-success-dialog.component';
import { Observable, Subject } from 'rxjs';
import { filter, takeUntil } from 'rxjs/operators';

@Component({
  selector: 'app-village-card',
  templateUrl: './village-card.component.html',
  styleUrls: ['./village-card.component.scss']
})
export class VillageCardComponent implements OnInit, OnDestroy {
  @Input() public waypoint: Waypoint;
  @Input() public waypointInOffer = false;
  @Output() public readonly roadmap = new EventEmitter<void>();
  @Output() public readonly selectVillage = new EventEmitter<void>();
  @Output() public readonly chooseVillage = new EventEmitter<void>();
  @Output() public readonly moveVillage = new EventEmitter<void>();
  @ViewChild('moveButton', { read: NetflixLoaderDirective })
  public moveButtonNetflixLoader: NetflixLoaderDirective;

  public isDiploma: boolean;
  public cost: number;
  public campuses: Campus[] = [];
  public readonly isDesktop$: Observable<boolean>;

  private _village: Village;
  private onDestroySubject$: Subject<void> = new Subject<void>();

  public get application(): Application {
    return this.context.application;
  }

  constructor(
    private adventureService: AdventureService,
    private analyticsService: AnalyticsService,
    private context: ApplicationContext,
    private breakpointService: BreakpointService,
    private dialog: MatDialog,
    private dialogService: DialogService
  ) {
    this.isDesktop$ = this.breakpointService.isDesktop$;
  }

  public ngOnInit(): void {
    this.isDiploma = isDiploma(this.waypoint.type);

    if (
      this.waypoint.id === this.adventureService.instance.definition.originId ||
      this.waypoint.id === this.adventureService.instance.definition.destinationId
    )
      this.cost = this.adventureService.instance.definition.cost;

    if (this.inOffer) this.campuses = this.context.application.campus;
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  @Input() public set village(village: Village) {
    this._village = village;

    if (this._village) this.waypoint = this.village.waypoint;
  }

  public get village(): Village {
    return this._village;
  }

  public get inOffer(): boolean {
    return this.village?.inOffer(this.application, this.context.product) || this.waypointInOffer;
  }

  // EVENT

  public onShowMoreInfo(): void {
    this.analyticsService.pushTag(
      new VillageNameAnalyticsEvent(AnalyticsEventType.MoreDetailsVillageClick, this.waypoint.label)
    );

    if (this.waypoint) this.selectVillage.emit();
  }

  public onChoose(): void {
    this.analyticsService.pushTag(
      new VillageNameAnalyticsEvent(AnalyticsEventType.VillageChoose, this.waypoint.label)
    );

    // Success dialog
    if (this.village.stage === 0) {
      this.showSuccess()
        .pipe(
          filter(res => !!res),
          takeUntil(this.onDestroySubject$)
        )
        .subscribe(() => {
          this.chooseVillage.emit();
          this.moveButtonNetflixLoader.start();
        });
    } else {
      this.chooseVillage.emit();
      this.moveButtonNetflixLoader.start();
    }
  }

  public onRoadmap(): void {
    if (this.moveButtonNetflixLoader.isStarted) this.moveButtonNetflixLoader.stop();
    else this.chooseVillage.emit();

    if (!this.village.waypoint.final) {
      this.analyticsService.pushTag(
        new VillageNameAnalyticsEvent(
          AnalyticsEventType.UncompletedAdventureRoadmapGenerate,
          this.waypoint.label
        )
      );
    }

    this.roadmap.emit();
  }

  // PRIVATE

  private showSuccess(): Observable<boolean | undefined> {
    let description: string;

    if (
      this.village.positionablePois?.filter(poi => poi.type === PoiType.SandwichTrainingJob)
        ?.length > 0
    )
      description = 'adventure.success-dialog.sandwich-training-description';
    else if (
      this.village.positionablePois?.filter(poi => poi.type === PoiType.InternshipJob)?.length > 0
    )
      description = 'adventure.success-dialog.internship-description';
    else description = 'adventure.success-dialog.default-description';

    return this.dialog
      .open(
        DiplomaSuccessDialogComponent,
        this.dialogService.simpleDialogConfig({
          title: 'adventure.success-dialog.title',
          label: this.village.waypoint.label,
          description
        })
      )
      .afterClosed();
  }
}
