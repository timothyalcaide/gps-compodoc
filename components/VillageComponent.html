import {
  Component,
  ElementRef,
  EventEmitter,
  Input,
  OnChanges,
  OnDestroy,
  OnInit,
  Output
} from '@angular/core';
import { isDiploma, Waypoint } from '@clients/adventure/bean/waypoint';
import { Poi } from '@clients/poi/poi';
import {
  AdventureService,
  AnalyticsService,
  ApplicationContext,
  BreakpointService,
  WaypointService
} from '@core/services';
import { MapElementComponent } from '@features/map/components';
import { Village } from '@features/map/models';
import { Paths, Utils } from '@features/map/utils';
import { AdventureType } from '@models/adventure-type/adventure-type';
import { AnalyticsEventType } from '@models/analytics-event/analytics-event-type';
import { VillageNameAnalyticsEvent } from '@models/analytics-event/village-name.analytics-event';
import { Filters } from '@models/filters/filters';
import { HumanPoi } from '@models/poi/human-poi/human-poi';
import { PositionablePoi } from '@models/poi/positionable-poi/positionable-poi';
import { POSITIONABLE_POI_MOBILE_POSITION_SETS } from '@models/poi/positionable-poi/positionable-poi-position-set';
import { TeamPoi } from '@models/poi/team-poi/team-poi';
import { InstanceService } from '@services/instance.service';
import { PoiService } from '@services/poi/poi.service';
import { VideoAskService } from '@services/video-ask/video-ask.service';
import { BehaviorSubject, Observable, of, Subject } from 'rxjs';
import { filter, map, share, takeUntil, tap } from 'rxjs/operators';

@Component({
  // eslint-disable-next-line @angular-eslint/component-selector
  selector: '[svg-village]',
  templateUrl: './village.component.html'
})
export class VillageComponent extends MapElementComponent implements OnInit, OnChanges, OnDestroy {
  @Input() public village: Village;
  @Input() public animationComplete$: BehaviorSubject<boolean>;
  @Input() public focusedPoi$: BehaviorSubject<Poi>;
  @Output() public readonly clickPoi = new EventEmitter<Poi>();
  @Output() public readonly back = new EventEmitter<void>();
  @Output() public readonly details: EventEmitter<void> = new EventEmitter<void>();
  @Output() public readonly gettingIn: EventEmitter<Village> = new EventEmitter<Village>();

  public villageRoadPathD = '';
  public villageSvgPath: SVGPathElement = null;
  public backwards$: Observable<Waypoint[]>;
  public teamPoi$: Observable<TeamPoi>;
  public videoAskPoi$ = new BehaviorSubject<HumanPoi>(null);
  public positionablePois$: Observable<PositionablePoi[]>;
  public poisPositionIndex: number;
  public horizons$: Observable<Waypoint[]>;

  private onDestroySubject$: Subject<void> = new Subject<void>();

  constructor(
    protected elRef: ElementRef<SVGGElement>,
    protected breakpointService: BreakpointService,
    private adventureService: AdventureService,
    private analyticsService: AnalyticsService,
    private instanceService: InstanceService,
    private poiService: PoiService,
    private videoAskService: VideoAskService,
    private waypointService: WaypointService,
    private context: ApplicationContext
  ) {
    super(elRef, breakpointService);
  }

  public ngOnInit(): void {
    this.villageRoadPathD = Paths.computePathStraight(
      this.village.start.x,
      this.village.start.y,
      this.village.end.x,
      this.village.end.y
    );

    // Initialize a fictive SVG path so child components can position them according to it
    this.villageSvgPath = document.createElementNS('http://www.w3.org/2000/svg', 'path');
    this.villageSvgPath.setAttribute('d', this.villageRoadPathD);
  }

  public ngOnChanges(): void {
    this.initVillageElements();
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public onBack(): void {
    this.analyticsService.pushTag(
      new VillageNameAnalyticsEvent(AnalyticsEventType.MapBackClick, this.village.waypoint.label)
    );

    this.village.selected = false;

    this.back.emit();
  }

  protected getPosition(): DOMPoint {
    /**
     * This component doesn't initialize its position as all its children are placed
     */
    throw new Error('VillageComponent position not implemented.');
  }

  /**
   * Initialise all village elements
   */
  private initVillageElements(): void {
    this.poisPositionIndex = Utils.randomBetween(
      0,
      POSITIONABLE_POI_MOBILE_POSITION_SETS.length - 1
    );

    if (this.adventureService.instance.adventureType !== AdventureType.Internship) {
      this.initBackwards();
      this.initHorizons();
    }

    this.initTeamPoi();
    this.initPositionablePois();
  }

  /**
   * Initialise backwards observable
   */
  private initBackwards(): void {
    this.backwards$ = this.village.backwards
      ? of(this.village.backwards)
      : this.waypointService.backwards(this.village.waypoint, this.adventureService.instance).pipe(
          filter((backwards: Waypoint[]) => !!backwards && !!backwards.length),
          map((backwards: Waypoint[]) =>
            backwards.length > 2 ? backwards.slice(0, 2) : backwards
          ),
          tap((backwards: Waypoint[]) => {
            this.village.backwards = backwards;
          }),
          takeUntil(this.onDestroySubject$)
        );
  }

  /**
   * Initialise the Team POI observable
   */
  private initTeamPoi(): void {
    this.videoAskPoi$ = new BehaviorSubject<HumanPoi>(this.village.videoAskPoi);

    this.teamPoi$ = this.village.teamPoi
      ? of(this.village.teamPoi)
      : (this.teamPoi$ = this.getHumanPois().pipe(
          map(humanPois => {
            if (humanPois.length) {
              // FIXME : Remove when Video ask is ready
              this.videoAskService
                .getMockedHumanPoisWithVideoAsks(humanPois, this.village.waypoint.id, true)
                .forEach(humanPoi => {
                  if (this.videoAskPoi$.value) return;

                  if (humanPoi.videoAskId) {
                    this.videoAskPoi$.next(humanPoi);
                    this.village.videoAskPoi = humanPoi;
                  }
                });

              return this.poiService.teamPoi(humanPois[0].id, this.village.waypoint);
            }

            return null;
          }),
          tap(teamPoi => {
            this.village.teamPoi = teamPoi;
          })
        ));
  }

  /**
   * Initialise the Positionable POIs observable
   */
  private initPositionablePois(): void {
    this.positionablePois$ = this.village.positionablePois
      ? of(this.village.positionablePois)
      : this.poiService
          .positionablePois(this.village.waypoint, new Filters().context(this.context.application))
          .pipe(
            tap((positionablePois: PositionablePoi[]) => {
              this.village.positionablePois = positionablePois;
            })
          );
  }

  /**
   * Initialise the horizons observable
   */
  private initHorizons(): void {
    this.horizons$ = this.village.horizons
      ? of(this.village.horizons)
      : this.instanceService
          .horizons(
            this.village.expand,
            this.village.stage,
            this.adventureService.instance.situation.gender
          )
          .pipe(
            filter((horizons: Waypoint[]) => !!horizons && !!horizons.length),
            map((horizons: Waypoint[]) => horizons.slice(0, 3)),
            tap((horizons: Waypoint[]) => {
              this.village.horizons = horizons;
            }),
            share()
          );
  }

  /**
   * Get the list of the human POIs of the village
   *
   * @returns An observable emitting the human POIs of the village
   */
  private getHumanPois(): Observable<HumanPoi[]> {
    return this.village.humanPois
      ? of(this.village.humanPois)
      : this.instanceService
          .humanPois(
            this.village.expand,
            this.village.stage,
            isDiploma(this.village.waypoint.type),
            this.adventureService.instance
          )
          .pipe(
            tap(humanPois => {
              this.village.humanPois = humanPois;
            })
          );
  }
}
