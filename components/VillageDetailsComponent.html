import {
  Component,
  EventEmitter,
  Input,
  OnChanges,
  OnDestroy,
  OnInit,
  Output
} from '@angular/core';
import { MatTabChangeEvent } from '@angular/material/tabs';
import { Application } from '@clients/adventure/bean/application';
import { isDiploma, isJob, Waypoint } from '@clients/adventure/bean/waypoint';
import { EXTERNAL_LINKS_CONFIG } from '@configs/external-links.config';
import { GAMIFICATION_ENABLED_CONFIG } from '@configs/gamification-enabled.config';
import { IN_OFFER_CONFIG } from '@configs/in-offer.config';
import { SCHOOLS_ENABLED_CONFIG } from '@configs/schools-enabled.config';
import {
  AnalyticsService,
  ApplicationContext,
  HroadsService,
  LoaderService,
  WaypointService
} from '@core/services';
import { AnalyticsEventType } from '@models/analytics-event/analytics-event-type';
import { TabChangeAnalyticsEvent } from '@models/analytics-event/tab-change.analytics-event';
import { BacSpecialtiesGroup } from '@models/bac-specialties-group/bac-specialties-group';
import { Campus } from '@models/campus/campus';
import { Company } from '@models/company/company';
import { Filters } from '@models/filters/filters';
import { Instance } from '@models/instance/instance';
import { Kpi } from '@models/kpi.model';
import { EventPoi } from '@models/poi/event/event-poi';
import { HumanPoi } from '@models/poi/human-poi/human-poi';
import { Mooc } from '@models/poi/mooc/mooc';
import { ParcoursPoi } from '@models/poi/parcours-poi/parcours-poi';
import { School } from '@models/school/school';
import { Situation } from '@models/situation/situation';
import { TabId } from '@models/tab-ids/tab-ids';
import { InstanceService } from '@services/instance.service';
import { PoiService } from '@services/poi/poi.service';
import { SituationService } from '@services/situation/situation.service';
import { VideoAskService } from '@services/video-ask/video-ask.service';
import { BehaviorSubject, Observable, Subject } from 'rxjs';
import { delay, map, share, takeUntil, tap } from 'rxjs/operators';

@Component({
  selector: 'app-village-details',
  templateUrl: './village-details.component.html',
  styleUrls: ['./village-details.component.scss']
})
export class VillageDetailsComponent implements OnInit, OnChanges, OnDestroy {
  @Input() public domain: Waypoint;
  @Input() public stage: number;
  @Input() public waypoint: Waypoint;
  @Input() public instance: Instance;
  @Input() public situation: Situation;
  @Output() public readonly closeVillageDetails = new EventEmitter<void>();

  public readonly TabId = TabId;
  public readonly GAMIFICATION_ENABLED: boolean;
  public readonly EXTERNAL_LINKS: boolean;
  public readonly application: Application;
  public isLoading$ = new BehaviorSubject<boolean>(false);
  public cost: number;
  public campuses: Campus[] = [];
  public hasVideos: boolean;

  // Data
  public parcoursPois$: Observable<ParcoursPoi[]>;
  public alumniPois$: Observable<HumanPoi[]>;
  public studentPois$: Observable<HumanPoi[]>;
  public eventPois$: Observable<EventPoi[]>;
  public moocs$: Observable<Mooc[]>;
  public kpis$: Observable<Kpi[]>;
  public schools$: Observable<School[]>;
  public companies$: Observable<Company[]>;
  public videosUrl$: Observable<string[]>;
  public recommendedBacSpecialties$: Observable<BacSpecialtiesGroup[]>;

  private readonly WITH_SCHOOL_TAB: boolean;
  private onDestroySubject$ = new Subject<void>();
  private loadingSessions$ = new BehaviorSubject<Record<string, never>[]>([]);

  public get isDiploma(): boolean {
    return isDiploma(this.waypoint.type);
  }

  public get isJob(): boolean {
    return isJob(this.waypoint.type);
  }

  public get inOffer(): boolean {
    return IN_OFFER_CONFIG(this.context.application, this.stage, this.context.product);
  }

  constructor(
    public context: ApplicationContext,
    private analyticsService: AnalyticsService,
    private hroadsService: HroadsService,
    private instanceService: InstanceService,
    private loaderService: LoaderService,
    private poiService: PoiService,
    private situationService: SituationService,
    private videoAskService: VideoAskService,
    private waypointService: WaypointService
  ) {
    this.application = this.context.application;

    this.GAMIFICATION_ENABLED = GAMIFICATION_ENABLED_CONFIG(
      this.context.product,
      this.context.applicationName
    );
    this.WITH_SCHOOL_TAB = SCHOOLS_ENABLED_CONFIG(this.context.application, this.context.product);
    this.EXTERNAL_LINKS = EXTERNAL_LINKS_CONFIG(this.context.product);
  }

  public ngOnInit(): void {
    this.loadingSessions$
      .pipe(
        map(loadingSessions => !!loadingSessions.length),
        // NOTE : Trick to prevent ExpressionChangedAfterItHasBeenCheckedError on progress bar
        delay(0),
        takeUntil(this.onDestroySubject$)
      )
      .subscribe(this.isLoading$);

    if (this.instance) {
      const { cost, originId, destinationId } = this.instance.definition;

      if (this.waypoint.id === originId || this.waypoint.id === destinationId) this.cost = cost;
    }

    if (this.inOffer) this.campuses = this.context.application.campus;
  }

  public ngOnChanges(): void {
    this.loadVillageData();
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  /**
   * Track tab change
   *
   * @param event The tab change event
   */
  public onSelectedTabChange(event: MatTabChangeEvent): void {
    this.analyticsService.pushTag(
      new TabChangeAnalyticsEvent(
        AnalyticsEventType.MoreDetailsVillageTabChange,
        `Village - ${event.tab.textLabel}`
      )
    );
  }

  /**
   * Load all the village data
   */
  private loadVillageData(): void {
    const filters: Filters = new Filters().context(this.context.application);

    if (this.isDiploma) {
      // FIXME: Hide until this project is started -> https://www.notion.so/millionroads/Bug-Dans-l-quipe-Anciens-l-ves-dans-les-tudiants-2189d68535ec409eaf7937a5c636a4c3
      //   this.studentPois$ = this.loaderService.observableLoadingSession<HumanPoi[]>(
      //     this.instanceService.studentPois(
      //       this.domain ? [this.domain, this.waypoint] : [this.waypoint],
      //       this.stage
      //     ),
      //     this.loadingSessions$
      //   );

      this.parcoursPois$ = this.loaderService.observableLoadingSession<ParcoursPoi[]>(
        this.poiService.cursusPois(this.waypoint).pipe(share()),
        this.loadingSessions$
      );

      this.eventPois$ = this.loaderService.observableLoadingSession<EventPoi[]>(
        this.poiService.events(this.waypoint, filters).pipe(share()),
        this.loadingSessions$
      );

      if (!this.stage) {
        this.recommendedBacSpecialties$ = this.situationService.recommendedBacSpecialties(
          this.waypoint.id
        );
      }
    } else if (this.isJob) {
      this.moocs$ = this.loaderService.observableLoadingSession<Mooc[]>(
        this.poiService.moocs(this.waypoint).pipe(share()),
        this.loadingSessions$
      );

      this.videosUrl$ = this.loaderService.observableLoadingSession<string[]>(
        this.poiService.videos(this.waypoint).pipe(
          tap(videos => {
            this.hasVideos = !!videos.length;
          }),
          share()
        ),
        this.loadingSessions$
      );

      this.companies$ = this.loaderService.observableLoadingSession<Company[]>(
        this.waypointService.companies(this.waypoint, this.application).pipe(share()),
        this.loadingSessions$
      );
    }

    if (this.WITH_SCHOOL_TAB && !this.stage) {
      this.schools$ = this.loaderService.observableLoadingSession<School[]>(
        this.waypointService.schools(this.waypoint, this.context.application).pipe(share()),
        this.loadingSessions$
      );
    }

    if (this.instance) {
      this.alumniPois$ = this.loaderService.observableLoadingSession<HumanPoi[]>(
        this.instanceService
          .alumniPois(
            this.domain ? [this.domain, this.waypoint] : [this.waypoint],
            this.stage,
            this.instance
          )
          .pipe(
            map(alumniPois =>
              // FIXME : Remove when VideoAsk is ready
              this.videoAskService.getMockedHumanPoisWithVideoAsks(
                alumniPois,
                this.waypoint.id,
                true
              )
            )
          ),
        this.loadingSessions$
      );
    }
    // TODO: Uncomment when the new requests for humans are done and a fix to get single humans without instance is found
    // else {
    //   this.alumniPois$ = this.loaderService.observableLoadingSession<HumanPoi[]>(
    //     this.waypointService.humanPois(this.waypoint.id, Nomenclature.alumni, filters).pipe(
    //       map(alumniPois =>
    //         // FIXME : Remove when VideoAsk is ready
    //         this.videoAskService.getMockedHumanPoisWithVideoAsks(alumniPois, this.waypoint.id, true)
    //       )
    //     ),
    //     this.loadingSessions$
    //   );
    // }

    this.kpis$ = this.loaderService.observableLoadingSession<Kpi[]>(
      this.hroadsService.kpis(this.waypoint, this.inOffer).pipe(
        map(p => p.filter((value: Kpi) => value.id !== 'diploma-school' || this.stage)),
        share()
      ),
      this.loadingSessions$
    );
  }
}
