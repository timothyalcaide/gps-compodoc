import {
  AfterViewInit,
  Component,
  EventEmitter,
  Input,
  OnChanges,
  OnInit,
  Output
} from '@angular/core';
import { Waypoint } from '@clients/adventure/bean/waypoint';
import { MapElementComponent } from '@features/map/components';
import { Village } from '@features/map/models';
import { Popularity } from '@models/popularity/popularity';

@Component({
  // eslint-disable-next-line @angular-eslint/component-selector
  selector: '[svg-village-label]',
  templateUrl: './village-label.component.html',
  styleUrls: ['./village-label.component.scss']
})
export class VillageLabelComponent
  extends MapElementComponent
  implements OnInit, AfterViewInit, OnChanges
{
  @Input() public village: Village = null;
  /**
   * Selected input is required even if redundant with village input to update label position on
   * deselection
   */
  @Input() public selected = false;
  @Input() public horizon: Waypoint = null;
  @Input() public villageSvgPath: SVGPathElement = null;
  @Input() public districtSvgPath: SVGPathElement = null;
  @Input() public index = 0;
  @Input() public disabled = false;
  @Output() public readonly gettingIn = new EventEmitter<Village>();

  public villagePopularity: Popularity;

  private readonly VILLAGE_LABEL_MOBILE_POSITION_RATIO: number = 0.65;
  private readonly VILLAGE_LABEL_TABLET_POSITION_RATIO: number = 0.67;
  private readonly VILLAGE_LABEL_DESKTOP_POSITION_RATIO: number = 0.61;
  private readonly VILLAGE_LABEL_MULTIPLE_IN_DISTRICT_SHIFT = 20;
  private readonly VILLAGE_LABEL_IN_HORIZON_POSITIONS: DOMPoint[] = [
    new DOMPoint(35, 100),
    new DOMPoint(30, 65),
    new DOMPoint(-145, 75)
  ];

  public get label(): string {
    if (this.horizon) return this.horizon.label;

    return this.village.waypoint.label;
  }

  public get isDestination(): boolean {
    if (this.village) return this.village.waypoint.final;

    if (this.horizon) return this.horizon.final;

    return false;
  }

  public get foreignObjectWidth(): number {
    if (this.districtSvgPath) return 300;

    if (this.horizon) return 120;

    return 200;
  }

  public get foreignObjectHeight(): number {
    if (this.districtSvgPath) return 155;

    if (this.horizon) return 28;

    return 72;
  }

  public get destSymbolWidth(): number {
    if (this.districtSvgPath) return 23;

    if (this.horizon) return 12;

    throw new Error(
      'VillageLabelComponent.destSymbolWidth should not be called if not in district or horizon.'
    );
  }

  public get isTop(): boolean {
    return this.districtSvgPath && !this.selected && this.index === 0;
  }

  public get isLeft(): boolean {
    return (
      (this.districtSvgPath && !this.selected && this.index === 1) ||
      (this.horizon && this.index === 2)
    );
  }

  public get isRight(): boolean {
    return (
      (this.districtSvgPath && !this.selected && this.index === 2) ||
      (this.horizon && (this.index === 0 || this.index === 1))
    );
  }

  /**
   * Indicates if the village label is placed with other village labels (in not selected district or
   * horizon)
   */
  public get inMultipleSituation(): boolean {
    return (this.districtSvgPath && !this.selected) || !!this.horizon;
  }

  public ngOnInit(): void {
    this.villagePopularity = new Popularity(
      this.village?.waypoint.userCount || this.horizon?.userCount
    );
  }

  public ngAfterViewInit(): void {
    // As position is depending on dimensions, we initialize it
    super.initDimensions();
    super.initPosition();
  }

  public ngOnChanges(): void {
    // As position is depending on dimensions, we initialize it
    super.initDimensions();
    super.initPosition();
  }

  public onVillageLabelClick(): void {
    if (this.districtSvgPath) this.gettingIn.emit(this.village);
  }

  protected getPosition(): DOMPoint {
    if (this.districtSvgPath && this.selected) return this.getInDistrictAndSelectedPosition();

    if (this.districtSvgPath && !this.selected) return this.getInDistrictAndNotSelectedPosition();

    if (this.horizon) return this.getInHorizonPosition();

    return this.getInVillagePosition();
  }

  /**
   * Get the position if the village label is in a village
   */
  private getInVillagePosition(): DOMPoint {
    let villageLabelPositionRatio: number;

    if (this.breakpointService.snapshot.isMobile)
      villageLabelPositionRatio = this.VILLAGE_LABEL_MOBILE_POSITION_RATIO;
    else if (this.breakpointService.snapshot.isTablet)
      villageLabelPositionRatio = this.VILLAGE_LABEL_TABLET_POSITION_RATIO;
    else villageLabelPositionRatio = this.VILLAGE_LABEL_DESKTOP_POSITION_RATIO;

    const villageRelativePosition: DOMPoint = this.villageSvgPath.getPointAtLength(
      this.villageSvgPath.getTotalLength() * villageLabelPositionRatio
    );

    return new DOMPoint(villageRelativePosition.x - this.width / 2, villageRelativePosition.y);
  }

  /**
   * Get the position if the village label is in a district and selected
   */
  private getInDistrictAndSelectedPosition(): DOMPoint {
    return new DOMPoint(-this.width / 2);
  }

  /**
   * Get the position if the village label is in a district and not selected
   */
  private getInDistrictAndNotSelectedPosition(): DOMPoint {
    switch (this.index) {
      case 0:
        return new DOMPoint(-this.width / 2);
      case 1:
        return new DOMPoint(
          -this.width - this.VILLAGE_LABEL_MULTIPLE_IN_DISTRICT_SHIFT,
          this.height
        );
      case 2:
        return new DOMPoint(this.VILLAGE_LABEL_MULTIPLE_IN_DISTRICT_SHIFT, this.height);
      default:
        throw Error(`Invalid village label index (${this.label} - ${this.index}).`);
    }
  }

  /**
   * Get the position if the village label is in horizon
   */
  private getInHorizonPosition(): DOMPoint {
    return this.VILLAGE_LABEL_IN_HORIZON_POSITIONS[this.index];
  }
}
