import { Directive, ElementRef } from '@angular/core';
import { BreakpointService } from '@core/services';
import { MapBowElement, Position, Sign, Village } from '@features/map/models';
import gsap from 'gsap';
import { BehaviorSubject } from 'rxjs';

@Directive({
  selector: '[appAnimateMap]'
})
export class AnimateMapDirective {
  private readonly MOBILE_MAP_FACTOR = 2;
  private readonly TABLET_MAP_FACTOR = 1.9;
  private readonly DESKTOP_MAP_FACTOR = 2.3;
  private readonly ANIMATION_DURATION = 1;
  private readonly ANIMATION_TYPE = 'power1.out';

  private get deviceMapFactor(): number {
    if (this.breakpointService.snapshot.isMobile) return this.MOBILE_MAP_FACTOR;

    if (this.breakpointService.snapshot.isTablet) return this.TABLET_MAP_FACTOR;

    return this.DESKTOP_MAP_FACTOR;
  }

  constructor(
    private svg: ElementRef<SVGGraphicsElement>,
    private breakpointService: BreakpointService
  ) {}

  /**
   * Zoom in the given village
   *
   * @param village The village
   */
  public zoomIn(village: Village, animationComplete$: BehaviorSubject<boolean>): void {
    this.animation(
      new DOMRect(
        village.start.x - this.svg.nativeElement.clientWidth / 2,
        village.start.y - this.svg.nativeElement.clientHeight,
        this.svg.nativeElement.clientWidth,
        this.svg.nativeElement.clientHeight
      ),
      animationComplete$
    );
  }

  /**
   * Zoom out on the given map bow element
   *
   * @param mapBowElement The map bow element
   * @param animationComplete$ The animation complete behavior subject used for callback
   */
  public zoomOut(mapBowElement: MapBowElement, animationComplete$: BehaviorSubject<boolean>): void {
    let newRect: DOMRect;

    if (this.breakpointService.snapshot.isMobile) {
      newRect = new DOMRect(
        mapBowElement.start.x - this.svg.nativeElement.clientWidth,
        mapBowElement.start.y - this.svg.nativeElement.clientHeight * 1.9,
        this.svg.nativeElement.clientWidth * this.deviceMapFactor,
        this.svg.nativeElement.clientHeight * this.deviceMapFactor
      );
    } else {
      newRect = new DOMRect(
        mapBowElement.start.x - this.svg.nativeElement.clientWidth * 1.184,
        mapBowElement.start.y - this.svg.nativeElement.clientHeight * 1.79,
        this.svg.nativeElement.clientWidth * this.deviceMapFactor,
        this.svg.nativeElement.clientHeight * this.deviceMapFactor
      );
    }

    this.animation(newRect, animationComplete$);
  }

  /**
   * Glide to one of the child element associated to the road of the given sign
   *
   * @param sign The sign
   * @param animationComplete$ The animation complete behavior subject used for callback
   */
  public glide(sign: Sign, animationComplete$: BehaviorSubject<boolean>): void {
    const childCenter = new Position(
      (sign.road.children[0].start.x + sign.road.children[0].end.x) / 2,
      (sign.road.children[0].start.y + sign.road.children[0].end.y) / 2
    );

    this.animation(
      new DOMRect(
        childCenter.x - (this.svg.nativeElement.clientWidth * this.deviceMapFactor) / 2,
        childCenter.y - (this.svg.nativeElement.clientHeight * this.deviceMapFactor) / 2,
        this.svg.nativeElement.clientWidth * this.deviceMapFactor,
        this.svg.nativeElement.clientHeight * this.deviceMapFactor
      ),
      animationComplete$
    );
  }

  private animation(newRect: DOMRect, animationComplete$?: BehaviorSubject<boolean>): void {
    const hasAlreadyViewBow = !!this.svg.nativeElement.getAttribute('viewBox');

    if (animationComplete$) animationComplete$.next(false);

    gsap.to(this.svg.nativeElement, {
      duration: hasAlreadyViewBow ? this.ANIMATION_DURATION : 0,
      attr: { viewBox: `${newRect.x} ${newRect.y} ${newRect.width} ${newRect.height}` },
      ease: this.ANIMATION_TYPE,
      onComplete: () => {
        if (animationComplete$) animationComplete$.next(true);
      }
    });
  }
}
