import { Directive, HostListener, Input, OnDestroy } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { Router } from '@angular/router';
import { isDiploma } from '@clients/adventure/bean/waypoint';
import { REGISTER_BEFORE_ROADMAP_CONFIG } from '@configs/register-before-roadmap.config';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import {
  AdventureService,
  ApplicationContext,
  AuthService,
  DialogService,
  LoaderService
} from '@core/services';
import { Instance } from '@models/instance/instance';
import { User } from '@models/user/user';
import { TranslateService } from '@ngx-translate/core';
import { RegistrationDialogComponent } from '@organisms/registration-dialog/registration-dialog.component';
import { ApplicationService } from '@services/application.service';
import { InstanceService } from '@services/instance.service';
import { forkJoin, Subject } from 'rxjs';
import { filter, takeUntil } from 'rxjs/operators';

@Directive({
  selector: '[appGenerateRoadmap]'
})
export class GenerateRoadmapDirective implements OnDestroy {
  @Input() public instance: Instance;

  private onDestroySubject$ = new Subject<void>();

  private get registrationDialogTitleLabel(): void {
    return isDiploma(this.instance.lastWp.type)
      ? this.translateService.instant(
          `dialog.registration.generate-roadmap.diploma.${this.instance.situation.gender}`
        )
      : this.instance.lastWp.label;
  }

  constructor(
    private adventure: AdventureService,
    private applicationService: ApplicationService,
    private authService: AuthService,
    private context: ApplicationContext,
    private dialog: MatDialog,
    private dialogService: DialogService,
    private instanceService: InstanceService,
    private loaderService: LoaderService,
    private router: Router,
    private translateService: TranslateService
  ) {}

  @HostListener('click')
  public onGenerate(): void {
    if (REGISTER_BEFORE_ROADMAP_CONFIG(this.context.application)) {
      if (!this.authService.hasEmail) {
        /**
         * We don't need to wait for dialog closing as the user will be directly brought from
         * registration or authentication to the roadmap
         */
        this.dialog
          .open(
            RegistrationDialogComponent,
            this.dialogService.defaultDialogConfig({
              titleKey: 'dialog.registration.generate-roadmap.title',
              titleParams: { label: this.registrationDialogTitleLabel },
              subtitleKey: `dialog.registration.generate-roadmap.description`,
              emojis: `üéâ${isDiploma(this.instance.lastWp.type) ? 'üéì' : 'üèÅ'}üéä`,
              instance: this.instance,
              beforeRoadmap: true
            })
          )
          .afterClosed()
          .pipe(
            filter((user: User) => !!user),
            takeUntil(this.onDestroySubject$)
          )
          .subscribe((user: User) => {
            this.authService.userSessionStorage = user;
            this.generateRoadmap(user.email);
          });
      } else this.generateRoadmap(this.authService.email);
    } else {
      this.router.navigateByUrl(`/${AppRouteUrls.ADVENTURE_ROADMAP(this.instance.id)}`);
    }
  }

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  /**
   * Attach instances to the user's email
   * Send the roadmap by mail to the user and redirect the user to the roadmap page
   *
   * @param email The email of the user to whom the instances has to be attached and to whom the roadmap is sent
   */
  private generateRoadmap(email: string): void {
    // Even if the user is not really authenticated, attach the unsaved instances to its email
    this.loaderService
      .globalObservableLoadingSession(
        forkJoin([
          this.adventure.attachInstance(this.instance.id, email),
          this.instanceService.sendRoadmap(
            this.context.applicationName,
            email,
            this.instance.id,
            this.applicationService.getApplicationUrl(
              this.context.product,
              this.context.applicationName
            )
          )
        ])
      )
      .pipe(takeUntil(this.onDestroySubject$))
      .subscribe(() => {
        this.router.navigateByUrl(`/${AppRouteUrls.ADVENTURE_ROADMAP(this.instance.id)}`);
      });
  }
}
