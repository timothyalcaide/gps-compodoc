import { Directive, ElementRef, HostListener, Input, Renderer2 } from '@angular/core';
import { IPosition } from '@features/map/models';
import { Box } from '@features/map/models/box/box.model';

@Directive({
  selector: '[appMoveMap]'
})
export class MoveMapDirective {
  @Input() public inVillage: boolean;

  /**
   * Define limits so the map can not be placed out of the screen
   */
  private readonly MAP_MOVE_MARGIN = 300;

  /**
   * Define the movement done from the pan start to last pan event
   */
  private lastDelta: IPosition = { x: 0, y: 0 };
  private viewBox: Box;
  private svgRect: Box;

  constructor(private hostSvg: ElementRef<SVGGraphicsElement>, private renderer: Renderer2) {}

  /**
   * Pan start event occurs when the user clicks or touches the map
   */
  @HostListener('panstart') public onPanStart(): void {
    const viewBoxValues: string[] = this.hostSvg.nativeElement.getAttribute('viewBox').split(' ');

    this.viewBox = {
      x: +viewBoxValues[0],
      y: +viewBoxValues[1],
      width: +viewBoxValues[2],
      height: +viewBoxValues[3]
    };

    this.svgRect = this.hostSvg.nativeElement.getBBox();

    this.lastDelta = { x: 0, y: 0 };
  }

  /**
   * Pan event occurs on and on until the click or touch is released
   */
  @HostListener('pan', ['$event']) public onPan(event: HammerInput): void {
    // Ignore map moving if the user is in a village
    if (this.inVillage) return;

    /**
     * event.deltaX and event.deltaY define the movement done from the pan start to the current pan
     * event
     * deltaX and deltaY define the movement done between the current pan event and the last pan
     * event
     */
    const deltaX: number = event.deltaX - this.lastDelta.x;
    const deltaY: number = event.deltaY - this.lastDelta.y;

    // Update last delta with the current pan event delta
    this.lastDelta.x = event.deltaX;
    this.lastDelta.y = event.deltaY;

    // Compute new SVG positions, using opposite deltas
    const newViewBoxPos: IPosition = {
      x: this.viewBox.x - deltaX * 2.5,
      y: this.viewBox.y - deltaY * 2.5
    };

    if (this.panLimitX(newViewBoxPos)) newViewBoxPos.x = this.viewBox.x;

    if (this.panLimitY(newViewBoxPos)) newViewBoxPos.y = this.viewBox.y;

    this.updateViewBox(newViewBoxPos);
  }

  /**
   * Returns if the new view box position is at the limit in the x axis
   *
   * @param newViewBoxPos The new view box position
   */
  private panLimitX(newViewBoxPos: IPosition): boolean {
    return (
      newViewBoxPos.x > this.svgRect.x + this.svgRect.width - this.MAP_MOVE_MARGIN ||
      newViewBoxPos.x + this.viewBox.width < this.svgRect.x + this.MAP_MOVE_MARGIN
    );
  }

  /**
   * Returns if the new view box position is at the limit in the y axis
   *
   * @param newViewBoxPos The new view box position
   */
  private panLimitY(newViewBoxPos: IPosition): boolean {
    return (
      newViewBoxPos.y > this.svgRect.y + this.svgRect.height - this.MAP_MOVE_MARGIN ||
      newViewBoxPos.y + this.viewBox.height < this.svgRect.y + this.MAP_MOVE_MARGIN
    );
  }

  /**
   * Update the map view box positions
   *
   * @param newViewBoxPos The new view bow positions
   */
  private updateViewBox(newViewBoxPos: IPosition): void {
    this.renderer.setAttribute(
      this.hostSvg.nativeElement,
      'viewBox',
      `${newViewBoxPos.x} ${newViewBoxPos.y} ${this.viewBox.width} ${this.viewBox.height}`
    );

    this.viewBox.x = newViewBoxPos.x;
    this.viewBox.y = newViewBoxPos.y;
  }
}
