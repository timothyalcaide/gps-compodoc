import {
  Directive,
  EventEmitter,
  HostBinding,
  HostListener,
  Input,
  OnDestroy,
  Output
} from '@angular/core';
import { interval, Subject } from 'rxjs';
import { map, take, takeUntil } from 'rxjs/operators';
import { NetflixLoaderService } from './netflix-loader.service';

@Directive({
  selector: '[appNetflixLoader]'
})
export class NetflixLoaderDirective implements OnDestroy {
  /**
   * The duration of the loader in millisecond
   * It must be a multiple of 20.
   */
  @Input() public duration = 2500;
  @Input() public color = '';
  @Output() public readonly netflixLoaderCompleted = new EventEmitter<void>();
  @Output() public readonly netflixStarted = new EventEmitter<void>();

  @HostBinding('style.background') public get background(): string {
    const initialColor =
      this.color === this.PRIMARY_COLOR_NAME ? this.PRIMARY_COLOR_CODE : this.DEFAULT_COLOR_CODE;

    const loadedColor =
      this.color === this.PRIMARY_COLOR_NAME
        ? this.PRIMARY_LOADED_COLOR_CODE
        : this.DEFAULT_LOADED_COLOR_CODE;

    return `linear-gradient(90deg, #${loadedColor} ${this.progress}%, #${initialColor} ${this.progress}%)`;
  }

  private progress = 0;
  private onProgressSubject$: Subject<void> = new Subject<void>();

  private readonly NETFLIX_LOADER_INTERVAL = 20;
  private PRIMARY_COLOR_NAME = 'primary';
  private PRIMARY_COLOR_CODE = '0a99b7';
  private PRIMARY_LOADED_COLOR_CODE = '006b87';
  private DEFAULT_COLOR_CODE = 'd8eff3';
  private DEFAULT_LOADED_COLOR_CODE = 'b1dee8';

  public get isStarted(): boolean {
    return !!this.progress;
  }

  constructor(private netflixService: NetflixLoaderService) {}

  /**
   * Start the netflix loader
   */
  @HostListener('click')
  public start(): void {
    this.netflixStarted.emit();
    this.netflixService.load();

    if (this.progress) {
      this.progress = 100;
      this.onProgressSubject$.next();
      this.onProgressSubject$.complete();
      this.netflixLoaderCompleted.emit();
      this.netflixService.complete();
    } else {
      this.onProgressSubject$ = new Subject<void>();

      interval(this.NETFLIX_LOADER_INTERVAL)
        .pipe(
          map(i => ((i * this.NETFLIX_LOADER_INTERVAL) / this.duration) * 100),
          take(this.duration / this.NETFLIX_LOADER_INTERVAL),
          takeUntil(this.onProgressSubject$)
        )
        .subscribe({
          next: percent => {
            this.progress = percent;
          },
          complete: () => {
            if (this.progress && this.progress !== 100) {
              this.netflixLoaderCompleted.emit();
              this.netflixService.complete();
            }
          }
        });
    }
  }

  public ngOnDestroy(): void {
    this.stop();
  }

  /**
   * Stop the netflix loader
   */
  public stop(): void {
    this.progress = 0;
    this.onProgressSubject$.next();
    this.onProgressSubject$.complete();
  }
}
