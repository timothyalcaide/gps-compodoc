import { Directive, ElementRef, Input, OnChanges } from '@angular/core';
import { fromString, Matrix, scale, toSVG, transform, translate } from 'transformation-matrix';

/**
 * Scale a SVG element while keeping it centered
 */
@Directive({
  selector: '[appScaleCenter]'
})
export class ScaleCenterDirective implements OnChanges {
  @Input() public scale = 0;

  private readonly INITIAL_MATRIX = 'matrix(1,0,0,1,0,0)';

  private current: Matrix = fromString(this.INITIAL_MATRIX);

  constructor(private hostSvg: ElementRef<SVGGraphicsElement>) {}

  public ngOnChanges(): void {
    this.setCenteredScale();
  }

  private setCenteredScale(): void {
    const { width, height } = this.hostSvg.nativeElement.getBBox();

    this.hostSvg.nativeElement.setAttribute(
      'transform',
      toSVG(
        transform(
          this.current,
          translate(width / 2, height / 2),
          scale(this.scale, this.scale),
          translate(-(width / 2), -(height / 2))
        )
      )
    );
  }
}
