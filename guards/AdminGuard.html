import { DOCUMENT } from '@angular/common';
import { Inject, Injectable } from '@angular/core';
import {
  ActivatedRouteSnapshot,
  CanActivateChild,
  Router,
  RouterStateSnapshot
} from '@angular/router';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { ApplicationContext, AuthService } from '@core/services';
import { Product } from '@models/context/product';
import { ErrorCodes } from '@models/error-codes/error-codes';
import { AdminApplicationContextService } from '@services/admin-application-context.service';
import { environment } from 'environments/environment';
import { Observable, of } from 'rxjs';
import { map, mapTo, switchMap } from 'rxjs/operators';

@Injectable()
export class AdminGuard implements CanActivateChild {
  constructor(
    @Inject(DOCUMENT) private document: Document,
    private adminApplicationContextService: AdminApplicationContextService,
    private context: ApplicationContext,
    private authService: AuthService,
    private router: Router
  ) {}

  public canActivateChild(
    childRoute: ActivatedRouteSnapshot,
    state: RouterStateSnapshot
  ): Observable<boolean> {
    return this.authService.canActivate$.pipe(
      switchMap((canActivate: boolean) => {
        if (canActivate) {
          if (this.authService.user.hasFullRights) return of(true);

          if (
            this.authService.user.manageAnAcquisitionApp &&
            !this.authService.user.manageAcquisitionApp(this.context.applicationName)
          )
            return of(this.redirect());

          if (
            this.authService.user.hasAccess(
              this.context.applicationName,
              childRoute.data.roles || [],
              childRoute.data.groups || []
            )
          )
            return of(true);

          return this.router.navigateByUrl(`/${AppRouteUrls.ERROR(ErrorCodes.ERROR_401)}`);
        }

        return this.authService
          .login(state.url, environment.keycloak.adminClientId, this.context.applicationName)
          .pipe(mapTo(false));
      }),
      // FIXME : A hack while we wait for MODULE_INITIALIZER : https://github.com/angular/angular/issues/17606
      switchMap((canActivate: boolean) =>
        canActivate
          ? this.adminApplicationContextService.initApplicationContext().pipe(map(() => true))
          : of(false)
      )
    );
  }

  /**
   * Redirect to the appropriate back office if we are on My Road
   */
  private redirect(): boolean {
    this.document.defaultView.location.host = `${this.authService.user.managedApplicationName}.${
      environment.hostSuffix[Product.Explore]
    }`;

    return false;
  }
}
