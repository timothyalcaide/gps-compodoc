import { Injectable } from '@angular/core';
import { ActivatedRouteSnapshot, CanActivate, Router, UrlTree } from '@angular/router';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { AdventureService, AuthService, SituationContext, WaypointService } from '@core/services';
import { DefinitionService } from '@core/services/definition/definition.service';
import { Adventure } from '@models/adventure/adventure';
import {
  DEFINITION_QUERY_STRING,
  DESTINATION_QUERY_STRING,
  ORIGIN_QUERY_STRING,
  PARTNER_ID_QUERY_STRING,
  PARTNER_QUERY_STRING,
  PARTNER_TYPE_QUERY_STRING
} from '@models/adventure/adventure-query-string';
import { ErrorCodes } from '@models/error-codes/error-codes';
import { PartnerName } from '@models/partner/partner-name';
import { OriginAndDestination } from '@models/waypoint/origin-with-destination';
import { WaypointType } from '@models/waypoint/waypoint-type';
import { catchError, forkJoin, map, Observable, of, switchMap } from 'rxjs';

@Injectable()
export class AdventureInUrlGuard implements CanActivate {
  constructor(
    private adventureService: AdventureService,
    private authService: AuthService,
    private definitionService: DefinitionService,
    private router: Router,
    private situationContext: SituationContext,
    private waypointService: WaypointService
  ) {}

  public canActivate(route: ActivatedRouteSnapshot): Observable<UrlTree> | boolean | UrlTree {
    const definitionId: string = route.queryParams[DEFINITION_QUERY_STRING];

    const originId: string = route.queryParams[ORIGIN_QUERY_STRING];

    const destinationId: string = route.queryParams[DESTINATION_QUERY_STRING];
    const partnerName: PartnerName = route.queryParams[PARTNER_QUERY_STRING];
    const partnerId: string = route.queryParams[PARTNER_ID_QUERY_STRING];

    if (partnerName && partnerId) {
      return this.partnerAdventure(
        partnerName,
        partnerId,
        route.queryParams[PARTNER_TYPE_QUERY_STRING]
      );
    }

    if (destinationId && originId)
      return this.destinationAndOriginAdventure(originId, destinationId);

    if (destinationId) {
      return this.router.parseUrl(
        AppRouteUrls.INTENTIONS_WAYPOINT_DETAILS(WaypointType.job, destinationId)
      );
    }

    if (originId) {
      return this.router.parseUrl(
        AppRouteUrls.INTENTIONS_WAYPOINT_DETAILS(WaypointType.diploma, originId)
      );
    }

    if (definitionId) return this.definitionAdventure(definitionId);

    return true;
  }

  private createAdventure(adventure: Adventure): Observable<UrlTree> {
    return this.adventureService.createAdventure(this.authService.email, adventure).pipe(
      map(instance => this.router.parseUrl(AppRouteUrls.ADVENTURE_MAP(instance.id))),
      catchError(() => this.redirectTo404())
    );
  }

  private partnerAdventure(
    partnerName: PartnerName,
    partnerId: string,
    partnerType: WaypointType
  ): Observable<UrlTree> {
    return this.waypointService.fromPartner(partnerName, partnerId, partnerType).pipe(
      switchMap((originAndDestination: OriginAndDestination) =>
        this.createAdventure({
          origin: originAndDestination.origin,
          destination: originAndDestination.destination
        })
      ),
      catchError(() => this.redirectTo404())
    );
  }

  private destinationAndOriginAdventure(
    originId: string,
    destinationId: string
  ): Observable<UrlTree> {
    return forkJoin({
      origin: this.waypointService.waypoint(originId, this.situationContext.situation.gender),
      destination: this.waypointService.waypoint(
        destinationId,
        this.situationContext.situation.gender
      )
    }).pipe(
      switchMap(adventure => this.createAdventure(adventure)),
      catchError(() => this.redirectTo404())
    );
  }

  private definitionAdventure(definitionId: string): Observable<UrlTree> {
    return this.definitionService.definition(definitionId).pipe(
      switchMap(adventure => this.createAdventure(adventure)),
      catchError(() => this.redirectTo404())
    );
  }

  private redirectTo404(): Observable<UrlTree> {
    return of(this.router.parseUrl(`/${AppRouteUrls.ERROR(ErrorCodes.ERROR_404)}`));
  }
}
