import { Injectable } from '@angular/core';
import { ActivatedRouteSnapshot, CanActivate, Router, UrlTree } from '@angular/router';
import { INTERNSHIP_ADVENTURE_OPTIONS_CONFIG } from '@configs/internship-adventure-options.config';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import {
  AdventureService,
  AnalyticsService,
  ApplicationContext,
  AuthService,
  HroadsClient,
  SituationContext,
  WaypointService
} from '@core/services';
import { AdventureType } from '@models/adventure-type/adventure-type';
import { OnBoardingFinalizeAnalyticsEvent } from '@models/analytics-event/on-boarding-finalize.analytics-event';
import { ErrorCodes } from '@models/error-codes/error-codes';
import { map, Observable, of, switchMap, tap } from 'rxjs';

@Injectable()
export class IntentionsInternshipGuard implements CanActivate {
  constructor(
    private adventureService: AdventureService,
    private analyticsService: AnalyticsService,
    private authService: AuthService,
    private context: ApplicationContext,
    private hroadsClient: HroadsClient,
    private router: Router,
    private waypointService: WaypointService,
    private situationContext: SituationContext
  ) {}

  public canActivate(route: ActivatedRouteSnapshot): Observable<UrlTree> | boolean | UrlTree {
    if (route.params.type !== AdventureType.Internship) return true;

    if (!INTERNSHIP_ADVENTURE_OPTIONS_CONFIG(this.context.product))
      return this.router.parseUrl(`/${AppRouteUrls.ERROR(ErrorCodes.ERROR_404)}`);

    if (!this.authService.user.situation) {
      return this.hroadsClient.getUser(this.authService.user).pipe(
        switchMap(() => {
          if (
            !this.authService.user.situation?.applicationStep ||
            !this.authService.user.situation?.currentStep
          )
            return of(this.router.parseUrl(`/${AppRouteUrls.ERROR(ErrorCodes.ERROR_404)}`));

          return this.createInstance();
        })
      );
    }

    return this.createInstance();
  }

  private createInstance(): Observable<UrlTree> {
    return this.waypointService
      .waypoint(
        this.situationContext.situation.applicationStep?.waypoint.id ||
          this.situationContext.situation.currentStep?.waypoint.id,
        this.situationContext.situation.gender
      )
      .pipe(
        switchMap(waypoint =>
          this.adventureService.createAdventure(this.authService.email, { origin: waypoint })
        ),
        tap(instance => {
          this.adventureService.instance = instance;
          this.adventureService.instance.adventureType = AdventureType.Internship;
          this.analyticsService.pushTag(
            new OnBoardingFinalizeAnalyticsEvent(
              'MAP',
              AdventureType.Internship.toUpperCase(),
              this.situationContext.situation.accurateStatus
            )
          );
        }),
        map(instance => this.router.parseUrl(`/${AppRouteUrls.ADVENTURE_MAP(instance.id)}`))
      );
  }
}
