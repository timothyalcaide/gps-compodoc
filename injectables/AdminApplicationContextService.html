import { Injectable } from '@angular/core';
import { Application } from '@clients/adventure/bean/application';
import { ApplicationClient, ApplicationContext, AuthService } from '@core/services';
import { User } from '@models/user/user';
import { BehaviorSubject, Observable } from 'rxjs';
import { tap } from 'rxjs/operators';

@Injectable()
export class AdminApplicationContextService {
  public managedApplication$ = new BehaviorSubject<Application>(null);

  public get managedApplication(): Application {
    return this.managedApplication$.value;
  }

  public set managedApplication(managedApplication: Application) {
    this.managedApplication$.next(managedApplication);
  }

  constructor(
    private authService: AuthService,
    private applicationClient: ApplicationClient,
    private context: ApplicationContext
  ) {}

  public initApplicationContext(): Observable<Application> {
    return this.getUserApplication(this.authService.user).pipe(
      tap(managedApplication => {
        this.managedApplication$.next(managedApplication);
      })
    );
  }

  /**
   * Get the application from an authenticated admin user
   *
   * @param user The authenticated admin user
   * @returns The corresponding application
   */
  private getUserApplication(user: User): Observable<Application> {
    if (user.isSchool) return this.applicationClient.getSchoolApplication(user.managedSchoolId);

    // It needs context.applicationName for super administrators
    return this.applicationClient.getApplication(
      user.hasFullRights ? this.context.applicationName : user.managedApplicationName
    );
  }
}
