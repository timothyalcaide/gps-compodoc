import { Injectable } from '@angular/core';
import { Application } from '@clients/adventure/bean/application';
import { ApplicationClient } from '@core/services';
import { User } from '@models/user/user';
import { AdminApplicationContextService } from '@services/admin-application-context.service';
import { Observable } from 'rxjs';

@Injectable()
export class AdminApplicationService {
  constructor(
    private applicationClient: ApplicationClient,
    private adminApplicationContextService: AdminApplicationContextService
  ) {}

  /**
   * Save the application, taking into account whether the authenticated admin user is a school
   * admin or an application admin
   *
   * @param user The authenticated admin user
   * @param application The application to save
   * @returns The saved application
   */
  public saveUserApplication(user: User, application: Application): Observable<Application> {
    if (user.isSchool) return this.saveSchoolApplication(application);

    return this.saveApplication(application);
  }

  // -- UPLOAD --

  /**
   * Upload an image within the application images array
   *
   * @param user The authenticated admin user
   * @param image The image file
   */
  public uploadImages(user: User, image: Blob): Observable<string> {
    const formData = new FormData();

    formData.append('file', image);

    if (user.isSchool)
      return this.applicationClient.uploadSchoolImages(
        this.adminApplicationContextService.managedApplication.nodeId,
        formData
      );

    return this.applicationClient.uploadImages(
      this.adminApplicationContextService.managedApplication.application,
      formData
    );
  }

  /**
   * Upload square logo for ambassador POIs
   *
   * @param logo The logo file
   */
  public uploadSquareLogo(logo: Blob): Observable<string> {
    const formData = new FormData();

    formData.append('file', logo);

    return this.applicationClient.uploadAmbassadorLogo(
      this.adminApplicationContextService.managedApplication.application,
      formData
    );
  }

  /**
   * Upload the Application logo
   *
   * @param user The authenticated admin user
   * @param logo The logo file
   */
  public uploadSchoolLogo(user: User, logo: Blob): Observable<string> {
    const formData = new FormData();

    formData.append('file', logo);

    if (user.isSchool)
      return this.applicationClient.uploadSchoolLogo(
        this.adminApplicationContextService.managedApplication.nodeId,
        formData
      );

    return this.applicationClient.uploadApplicationLogo(
      this.adminApplicationContextService.managedApplication.application,
      formData
    );
  }

  /**
   * Delete an image within the application images array
   *
   * @param user The authenticated admin user
   * @param image The path of image file
   */
  public deleteImage(user: User, image: string): Observable<void> {
    if (user.isSchool)
      return this.applicationClient.deleteSchoolImage(
        this.adminApplicationContextService.managedApplication.nodeId,
        image
      );

    return this.applicationClient.deleteImage(
      this.adminApplicationContextService.managedApplication.application,
      image
    );
  }

  /**
   * Save application info for a Acquisition client
   *
   * @param application The application of the client
   * @returns The saved application
   */
  private saveApplication(application: Application): Observable<Application> {
    return this.applicationClient.saveApplication(application);
  }

  /**
   * Save application info corresponding to a school
   *
   * @param application The application of the school
   * @returns The saved application
   */
  private saveSchoolApplication(application: Application): Observable<Application> {
    return this.applicationClient.saveSchoolApplication(application);
  }
}
