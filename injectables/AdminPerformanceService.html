import { ListMetric } from '@admin-models/list-metric/list-metric';
import { NumericMetric } from '@admin-models/numeric-metric/numeric-metric';
import { Injectable } from '@angular/core';
import { LeadClient } from '@clients/adventure/lead.client';
import { InstanceClient } from '@core/services';
import { Dayjs } from 'dayjs';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';

@Injectable()
export class AdminPerformanceService {
  constructor(private instanceClient: InstanceClient, private client: LeadClient) {}

  /**
   * Get the number of leads for an application from a date
   *
   * @param applicationName The application name
   * @param startDate The date when the count has to start
   * @returns The metric of the number of leads from the given date
   */
  public countLeadsForApplication(
    applicationName: string,
    startDate: Dayjs
  ): Observable<NumericMetric> {
    return this.client
      .countLeadsForApplication(applicationName, startDate)
      .pipe(map(leadsNumber => this.buildCountLeadsMetric(leadsNumber)));
  }

  /**
   * Get the number of leads for a school from a date
   *
   * @param school The school uuid
   * @param startDate The date when the count has to start
   * @returns The metric of the number of leads from the given date
   */
  public countLeadsForSchool(school: string, startDate: Dayjs): Observable<NumericMetric> {
    return this.client
      .countLeadsForSchool(school, startDate)
      .pipe(map(leadsNumber => this.buildCountLeadsMetric(leadsNumber)));
  }

  /**
   * Get the number of started instances for an application from a date
   *
   * @param applicationName The application name
   * @param startDate The date when the count has to start
   * @returns The metric of the number of started instances from the given date
   */
  public countStartedInstances(
    applicationName: string,
    startDate: Dayjs
  ): Observable<NumericMetric> {
    return this.instanceClient.countStartedInstances(applicationName, startDate).pipe(
      map(
        startedInstancesNumber =>
          ({
            labelKey: 'admin.performance.numeric-metrics.started-instances-number',
            score: startedInstancesNumber
          } as NumericMetric)
      )
    );
  }

  /**
   * Get the number of completed instances for an application from a date
   *
   * @param applicationName The application name
   * @param startDate The date when the count has to start
   * @returns The number of completed instances from the given date
   */
  public countCompletedInstances(
    applicationName: string,
    startDate: Dayjs
  ): Observable<NumericMetric> {
    return this.instanceClient.countCompletedInstances(applicationName, startDate).pipe(
      map(
        completedInstancesNumber =>
          ({
            labelKey: 'admin.performance.numeric-metrics.completed-instances-number',
            score: completedInstancesNumber
          } as NumericMetric)
      )
    );
  }

  /**
   * Get the number of started instances for an application from a date
   *
   * @param applicationName The application name
   * @param startDate The date when the count has to start
   * @returns The number of started instances from the given date
   */
  public countOrigins(applicationName: string, startDate: Dayjs): Observable<ListMetric> {
    return this.instanceClient
      .countOrigins(applicationName, startDate)
      .pipe(
        map(originsNumbers =>
          ListMetric.build('admin.performance.list-metrics.origins-numbers', originsNumbers)
        )
      );
  }

  // -- PRIVATE --

  /**
   * Build a Numeric Metric for leads number
   */
  private buildCountLeadsMetric(leadsNumber: number): NumericMetric {
    return {
      labelKey: 'admin.performance.numeric-metrics.contacts-number',
      score: leadsNumber
    } as NumericMetric;
  }
}
