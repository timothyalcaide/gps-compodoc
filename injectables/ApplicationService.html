import { DOCUMENT } from '@angular/common';
import { Inject, Injectable } from '@angular/core';
import { Application } from '@clients/adventure/bean/application';
import { ApplicationClient } from '@core/services';
import { ApplicationName } from '@models/context/application-name';
import { Product } from '@models/context/product';
import { LeadContact } from '@models/lead/lead-contact';
import { LeadContactSchool } from '@models/lead/lead-contact-school';
import { User } from '@models/user/user';
import { environment } from 'environments/environment';
import { Observable } from 'rxjs';

@Injectable()
export class ApplicationService {
  constructor(@Inject(DOCUMENT) private document, private applicationClient: ApplicationClient) {}

  // -- APPLICATION --

  /**
   * Get an application by the application property
   *
   * @param application The application name
   * @returns An observable emitting the matching application
   */
  public getApplication(application: string): Observable<Application> {
    return this.applicationClient.getApplication(application);
  }

  /**
   * Get an application by the Id of a school
   *
   * @param schoolId The Id of the school
   * @returns An observable emitting the matching application
   */
  public getSchoolApplication(schoolId: string): Observable<Application> {
    return this.applicationClient.getSchoolApplication(schoolId);
  }

  /**
   * Get the Url of an application
   *
   * @param product The product
   * @param applicationName The application name
   * @returns The application Url
   */
  public getApplicationUrl(product: Product, applicationName: string): string {
    switch (applicationName) {
      case ApplicationName.MyRoad:
      case ApplicationName.CmqBtpNumerique:
        return `${environment.protocol}${environment.hostSuffix[applicationName]}`;
      default:
        return `${environment.protocol}${applicationName}.${environment.hostSuffix[product]}`;
    }
  }

  // -- CONTACT --

  /**
   * Check if a lead for contact admission exists
   *
   * @param userEmail The user's email
   */
  public hasContactAdmission(userEmail: string): Observable<boolean> {
    return this.applicationClient.hasContactAdmission(userEmail);
  }

  /**
   * Create a lead for a contact admission action & email the school
   *
   * @param applicationName The application on which the lead is linked
   * @param email The user email
   * @param waypointUuid The waypoint's uuid at which the user is interested
   * @param instanceId The Instance's uuid to retrieve the context of the adventure
   */
  public contactAdmission(
    applicationName: string,
    email: string,
    waypointUuid: string,
    instanceId: string
  ): Observable<LeadContact> {
    return this.applicationClient.contactAdmission(
      LeadContact.toLeadContact(applicationName, email, waypointUuid, instanceId)
    );
  }

  /**
   * Check if a lead for contact admission exists
   *
   * @param userEmail The user's email
   * @param schoolUuid The school's uuid
   */
  public hasContactSchool(userEmail: string, schoolUuid: string): Observable<boolean> {
    return this.applicationClient.hasContactSchool(userEmail, schoolUuid);
  }

  /**
   * Create a lead for a contact school action & send an email from the user to the school
   *
   * @param applicationName The application on which the lead is linked
   * @param email The user email
   * @param user The user
   * @param waypointUuid The waypoint's uuid at which the user is interested
   * @param instanceId The Instance's id to retrieve the context of the adventure
   * @param schoolUuid The contacted school's uuid
   * @param campusUuid The contacted campus's uuid
   * @param subject The subject of the email to send
   * @param body The body of the email to send
   */
  public contactSchool(
    applicationName: string,
    email: string,
    user: User,
    waypointUuid: string,
    instanceId: string,
    schoolUuid: string,
    campusUuid: string,
    subject: string,
    body: string
  ): Observable<LeadContactSchool> {
    return this.applicationClient.contactSchool(
      LeadContactSchool.toLeadContactSchool(
        applicationName,
        email,
        user,
        waypointUuid,
        instanceId,
        schoolUuid,
        campusUuid,
        subject,
        body,
        this.document.defaultView.location.origin
      )
    );
  }
}
