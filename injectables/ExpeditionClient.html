import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { environment } from 'environments/environment';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';
import { ExpeditionInstance } from '../models';
import { ApiExpedition, ApiExpeditionInstance, ExpeditionStep } from '../models/expedition.model';

@Injectable({
  providedIn: 'root'
})
export class ExpeditionClient {
  private readonly EXPEDITION_INSTANCE_URI = `${environment.adventureUri}/expedition-instances`;
  private readonly EXPEDITION_URI = `${environment.adventureUri}/expeditions`;

  constructor(private http: HttpClient) {}

  /**
   * Get all expedition instances for the authenticated user
   *
   * @returns An observable emitting the list of expedition instances
   */
  public getExpeditionInstances(): Observable<ApiExpeditionInstance[]> {
    return this.http.get<ApiExpeditionInstance[]>(this.EXPEDITION_INSTANCE_URI);
  }

  public getExpeditions(): Observable<ApiExpedition[]> {
    return this.http.get<ApiExpedition[]>(this.EXPEDITION_URI);
  }

  /**
   * Get an expedition instance from its id
   *
   * @param id The id of the expedition instance
   * @returns An observable emitting the expedition instance
   */
  public getExpeditionInstance(id: string): Observable<ApiExpeditionInstance> {
    return this.http.get<ApiExpeditionInstance>(`${this.EXPEDITION_INSTANCE_URI}/${id}`);
  }

  public getExpedition(id: string): Observable<ApiExpedition> {
    return this.http.get<ApiExpedition>(`${this.EXPEDITION_URI}/${id}`);
  }

  /**
   * Create new expedition instance from expedition Uuid
   *
   * @param expeditionUuid The expedition uuid the user wants to start
   * @returns The created expedition instance
   */
  public createExpeditionInstance(expeditionUuid: string): Observable<ExpeditionInstance> {
    const params = new HttpParams().set('expeditionUuid', expeditionUuid);

    return this.http
      .post<ApiExpeditionInstance>(this.EXPEDITION_INSTANCE_URI, {}, { params })
      .pipe(map(apiExpeditionInstance => ExpeditionInstance.build(apiExpeditionInstance)));
  }

  /**
   * Get a challenge from an expedition instance id and its id
   *
   * @param expeditionInstanceId The expedition instance id
   * @param challengeId The challenge id
   * @returns An observable emitting the challenge
   */
  public getStepsFromExpeditionInstance(
    expeditionInstanceId: string,
    challengeId: string
  ): Observable<ExpeditionStep> {
    return this.http.get<ExpeditionStep>(
      `${this.CHALLENGE_URI(expeditionInstanceId)}/${challengeId}`
    );
  }

  /**
   * Add a challenge synthesis to an expedition instance
   *
   * @param expeditionInstanceId The expedition instance id
   * @param challengeId The challenge id
   * @param synthesis The challenge synthesis
   * @returns An observable emitting the updated expedition instance
   */
  public addChallengeSynthesis(
    expeditionInstanceId: string,
    challengeId: string,
    synthesis: string
  ): Observable<ApiExpeditionInstance> {
    return this.http.put<ApiExpeditionInstance>(
      `${this.CHALLENGE_URI(expeditionInstanceId)}/${challengeId}/synthesis`,
      { text: synthesis }
    );
  }

  private CHALLENGE_URI(expeditionInstanceId: string): string {
    return `${this.EXPEDITION_INSTANCE_URI}/${expeditionInstanceId}/expeditionSteps`;
  }
}
