import { Injectable } from '@angular/core';
import { Expand, Waypoint } from '@clients/adventure/bean/waypoint';
import { Nomenclature } from '@clients/poi/poi';
import { AuthService, InstanceClient } from '@core/services';
import { AdventureService } from '@core/services/adventure/adventure.service';
import { Product } from '@models/context/product';
import { Filters } from '@models/filters/filters';
import { Gender } from '@models/gender/gender';
import { Instance } from '@models/instance/instance';
import { LeadRoadmap } from '@models/lead/lead-roadmap';
import { HumanPoi } from '@models/poi/human-poi/human-poi';
import { WaypointType } from '@models/waypoint/waypoint-type';
import { Observable, of } from 'rxjs';
import { defaultIfEmpty, map, mergeMap } from 'rxjs/operators';

@Injectable()
export class InstanceService {
  constructor(
    private adventureService: AdventureService,
    private authService: AuthService,
    private client: InstanceClient
  ) {}

  public getInstance(id: string): Observable<Instance> {
    return this.client.instance(id);
  }

  /**
   * Get the list of instances of the current user
   *
   * @returns An observable emitting the list of instances
   */
  public instances(): Observable<Instance[]> {
    return this.authService.isAuthenticated
      ? this.client.instances().pipe(defaultIfEmpty([]))
      : of([]);
  }

  /* -- MAP -- */

  public horizons(waypoints: Waypoint[], stage: number, gender: Gender): Observable<Waypoint[]> {
    return this.client.horizons(this.adventureService.instance.id, stage, waypoints, gender);
  }

  public siblings(
    waypoints: Waypoint[],
    stage: number,
    gender: Gender,
    filters: Filters
  ): Observable<Waypoint[]> {
    return this.client.siblings(
      this.adventureService.instance.id,
      stage,
      waypoints,
      gender,
      filters
    );
  }

  /**
   * Get the list of domains and waypoints for the next step in the adventure
   *
   * @param waypoints The list of waypoints in the instance since the previous village
   * @param stage The stage in the instance
   * @param gender The gender of the user
   * @param filters The list of filters
   * @param type The type of requested waypoints
   * @returns An observable emitting an expand, listing the next diplomas and/or jobs
   */
  public expand(
    waypoints: Waypoint[],
    stage: number,
    gender: Gender,
    filters?: Filters,
    type?: WaypointType
  ): Observable<Expand> {
    return this.client.expand(
      this.adventureService.instance.id,
      stage,
      waypoints,
      gender,
      filters,
      type
    );
  }

  /* -- HUMAN -- */

  public humanPois(
    waypoints: Waypoint[],
    stage: number,
    student: boolean,
    instance: Instance
  ): Observable<HumanPoi[]> {
    return this.alumniPois(waypoints, stage, instance).pipe(
      mergeMap(alumni =>
        student
          ? this.studentPois(waypoints, stage, instance).pipe(
              map(students => alumni.concat(students))
            )
          : of<HumanPoi[]>(alumni)
      )
    );
  }

  public humanPoi(instanceId: string, humanId: string): Observable<HumanPoi> {
    return this.client.humanPoi(instanceId, humanId);
  }

  public alumniPois(
    waypoints: Waypoint[],
    stage: number,
    instance: Instance
  ): Observable<HumanPoi[]> {
    if (!waypoints)
      // Roadmap case
      return this.client.humanPois(instance.id, Nomenclature.alumni);

    return this.client.humanPois(instance.id, Nomenclature.alumni, stage, waypoints);
  }

  public studentPois(
    waypoints: Waypoint[],
    stage: number,
    instance: Instance
  ): Observable<HumanPoi[]> {
    if (!waypoints)
      // Roadmap case
      return this.client.humanPois(instance.id, Nomenclature.student);

    return this.client.humanPois(instance.id, Nomenclature.student, stage, waypoints);
  }

  /**
   * Create a lead for a save roadmap action & send the roadmap by email to the user attached to the instance
   *
   * @param applicationName The application on which the lead is linked
   * @param email The user email
   * @param instanceId The Instance's id to retrieve the context of the adventure
   * @param url The coupon code if the reward is of the coupon code type
   */
  public sendRoadmap(
    applicationName: string,
    email: string,
    instanceId: string,
    url: string
  ): Observable<LeadRoadmap> {
    return this.client.sendRoadmap(
      LeadRoadmap.toLeadRoadmap(
        applicationName,
        email,
        instanceId,
        url.replace(Product.Show, Product.Explore)
      )
    );
  }
}
