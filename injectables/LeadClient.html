import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { LeadBackOffice } from '@models/lead/lead-back-office';
import { Dayjs } from 'dayjs';
import { environment } from 'environments/environment';
import { Observable } from 'rxjs';

@Injectable()
export class LeadClient {
  private readonly LEAD_URI = `${environment.adventureUri}/leads`;

  constructor(private http: HttpClient) {}

  // -- ADMIN --

  /**
   * List all the leads for a given application
   *
   * @param applicationName The application name
   * @param startDate The date from when the list starts
   * @returns The leads list
   */
  public listLeadsForApplication(
    applicationName: string,
    startDate: Dayjs
  ): Observable<LeadBackOffice[]> {
    const params = new HttpParams().set('a', applicationName).set('d', `${startDate.unix()}`);

    return this.http.get<LeadBackOffice[]>(`${this.LEAD_URI}/list-app`, { params });
  }

  /**
   * List all the leads for a given school
   *
   * @param schoolUuid The school uuid
   * @param startDate The date from when the list starts
   * @returns The leads list
   */
  public listLeadsForSchool(schoolUuid: string, startDate: Dayjs): Observable<LeadBackOffice[]> {
    const params = new HttpParams().set('s', schoolUuid).set('d', `${startDate.unix()}`);

    return this.http.get<LeadBackOffice[]>(`${this.LEAD_URI}/list-school`, { params });
  }

  /**
   * Get the number of leads received by an application
   *
   * @param applicationName The application name
   * @param startDate The date from when the count starts
   * @returns The number of leads received by an application from the given date
   */
  public countLeadsForApplication(applicationName: string, startDate: Dayjs): Observable<number> {
    const params = new HttpParams().set('a', applicationName).set('d', `${startDate.unix()}`);

    return this.http.get<number>(`${this.LEAD_URI}/count-app`, { params });
  }

  /**
   * Get the number of leads received by a school
   *
   * @param schoolUuid The school uuid
   * @param startDate The date from when the count starts
   * @returns The number of leads received by a school from the given date
   */
  public countLeadsForSchool(schoolUuid: string, startDate: Dayjs): Observable<number> {
    const params = new HttpParams().set('s', schoolUuid).set('d', `${startDate.unix()}`);

    return this.http.get<number>(`${this.LEAD_URI}/count-school`, { params });
  }
}
