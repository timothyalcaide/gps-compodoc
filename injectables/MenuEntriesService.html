import { Injectable } from '@angular/core';
import { situationIcon } from '@clients/adventure/bean/situation.helper';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { RouteIds } from '@core/routing/route-ids';
import { ApplicationContext, SituationContext } from '@core/services';
import { Instance } from '@models/instance/instance';
import { MenuEntry } from '@models/menu-entry/menu-entry';
import { Situation } from '@models/situation/situation';
import { INTENTIONS_ROUTE } from '@pages/intentions/intentions.routing';
import { ACCOUNT_ROUTE, EXPEDITION_ROUTE, ROADMAPS_ROUTE, SCHOOL_ROUTE } from 'app/app.routing';
import { combineLatest, Observable } from 'rxjs';
import { map, share } from 'rxjs/operators';
import { AdventureService } from '../adventure/adventure.service';

@Injectable({
  providedIn: 'root'
})
export class MenuEntriesService {
  public menuEntries$: Observable<MenuEntry[]>;

  private instance$ = this.adventureService.instance$;
  private situation$ = this.situationContext.situation$;

  constructor(
    private adventureService: AdventureService,
    private context: ApplicationContext,
    private situationContext: SituationContext
  ) {
    this.menuEntries$ = combineLatest([this.instance$, this.situation$]).pipe(
      map(([instance, situation]) => this.getMenuEntries(instance, situation)),
      share()
    );
  }

  private getMenuEntries(instance: Instance, situation: Situation): MenuEntry[] {
    const menuEntries: MenuEntry[] = [];

    const { applicationName, product } = this.context;

    if (
      !INTENTIONS_ROUTE.data.forbiddenProducts?.includes(product) &&
      !INTENTIONS_ROUTE.data.forbiddenApplications?.includes(applicationName)
    ) {
      menuEntries.push(this.intentionsMenuEntry(instance));
    }

    if (
      !ROADMAPS_ROUTE.data.forbiddenProducts?.includes(product) &&
      !ROADMAPS_ROUTE.data.forbiddenApplications?.includes(applicationName)
    )
      menuEntries.push(this.roadmapsMenuEntry());

    if (
      !SCHOOL_ROUTE.data.forbiddenProducts?.includes(product) &&
      !SCHOOL_ROUTE.data.forbiddenApplications?.includes(applicationName)
    )
      menuEntries.push(this.schoolMenuEntry());

    if (
      !EXPEDITION_ROUTE.data.forbiddenProducts?.includes(product) &&
      !EXPEDITION_ROUTE.data.forbiddenApplications?.includes(applicationName)
    )
      menuEntries.push(this.exploreMenuEntry());

    if (
      !ACCOUNT_ROUTE.data.forbiddenProducts?.includes(product) &&
      !ACCOUNT_ROUTE.data.forbiddenApplications?.includes(applicationName)
    )
      menuEntries.push(this.profileMenuEntry(situation));

    return menuEntries;
  }

  private intentionsMenuEntry(instance: Instance): MenuEntry {
    return {
      labelKey: 'navigation.discover',
      svgIcon: 'pin-village-destination',
      routeId: instance ? RouteIds.ADVENTURE_MAP : RouteIds.INTENTIONS,
      link: instance ? `/${AppRouteUrls.ADVENTURE_MAP(instance.id)}` : `/${AppRouteUrls.INTENTIONS}`
    };
  }

  private roadmapsMenuEntry(): MenuEntry {
    return {
      labelKey: 'roadmap.roadmaps',
      svgIcon: 'roadmap',
      routeId: RouteIds.ROADMAPS,
      link: `/${AppRouteUrls.ROADMAPS}`
    };
  }

  private schoolMenuEntry(): MenuEntry {
    return {
      labelKey: this.context?.application?.name,
      svgIcon: 'school',
      routeId: RouteIds.SCHOOL,
      link: `/${AppRouteUrls.SCHOOL}`,
      svgIconStroke: true
    };
  }

  private exploreMenuEntry(): MenuEntry {
    return {
      labelKey: 'timeline.headline',
      svgIcon: 'explore',
      routeId: RouteIds.EXPEDITION,
      link: `/${AppRouteUrls.EXPEDITION}`,
      materialIcon: true
    };
  }

  private profileMenuEntry(situation: Situation): MenuEntry {
    return {
      labelKey: 'auth.my-profile',
      svgIcon: situation ? situationIcon(situation.status, situation.gender) : 'profile',
      routeId: RouteIds.PROFILE,
      link: `/${AppRouteUrls.PROFILE}`
    };
  }
}
