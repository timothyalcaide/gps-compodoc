import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { ApplicationContext } from '@core/services';
import { Filters } from '@models/filters/filters';
import { Offer } from '@models/offer/offer';
import { EventPoi } from '@models/poi/event/event-poi';
import { ApiEvent } from '@models/poi/event/event-poi.api';
import { Mooc } from '@models/poi/mooc/mooc';
import { ParcoursPoi } from '@models/poi/parcours-poi/parcours-poi';
import { PoiType } from '@models/poi/poi-type';
import { PositionablePoi } from '@models/poi/positionable-poi/positionable-poi';
import { WaypointType } from '@models/waypoint/waypoint-type';
import { environment } from 'environments/environment';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';
import { Poi } from './poi';

@Injectable()
export class PoiClient {
  private readonly BASE_URL = `${environment.poiUri}`;
  private readonly EVENT_URI = `${environment.poiUri}/events`;
  private readonly MOOC_URI = `${environment.poiUri}/moocs`;
  private readonly VIDEO_URI = `${environment.poiUri}/videos`;

  constructor(private context: ApplicationContext, private http: HttpClient) {}

  // -- POI --

  public pois<P extends Poi>(provider: string, applicationUuid: string): Observable<P[]> {
    const params = new HttpParams().set('app', applicationUuid);

    return this.http.get<P[]>(`${this.BASE_URL}/${provider}`, { params });
  }

  /**
   * Get a POI
   *
   * @param provider The provider of the POI
   * @param poiId The id of the POI
   * @returns The matching POI
   */
  public poi<P extends Poi>(provider: string, poiId: string): Observable<P> {
    return this.http.get<P>(`${this.BASE_URL}/${provider}/${poiId}`);
  }

  public save<P extends Poi>(provider: string, poi: P): Observable<P> {
    return this.http.post<P>(`${this.BASE_URL}/${provider}`, poi);
  }

  public update<P extends Poi>(provider: string, poi: P): Observable<P> {
    return this.http.put<P>(`${this.BASE_URL}/${provider}`, poi);
  }

  public delete<P extends Poi>(provider: string, poi: P): Observable<void> {
    return this.http.delete<void>(`${this.BASE_URL}/${provider}/${poi.id}`);
  }

  // -- SPECIFIC POI --

  /**
   * Retrieve all positionable pois by the resume node uuid
   *
   * @param   locationUuid The Uuid of the Application/Location
   * @param   resumeNodeUuid The Uuid of the resume node
   * @param   resumeNodeType The type of the resume node
   * @param   poiFilters The list of filters
   */
  public positionablePois(
    locationUuid: string,
    resumeNodeUuid: string,
    resumeNodeType: WaypointType,
    poiFilters?: Filters
  ): Observable<PositionablePoi[]> {
    let params = new HttpParams().set('node', resumeNodeUuid).set('label', resumeNodeType);

    if (locationUuid) params = params.set('app', locationUuid);

    return this.http
      .post<PositionablePoi[]>(
        `${this.BASE_URL}/positionable-pois`,
        poiFilters ? poiFilters.toApi() : {},
        {
          params
        }
      )
      .pipe(
        map(positionablePois =>
          positionablePois.map(positionablePoi => {
            if (positionablePoi.type === PoiType.Event)
              return EventPoi.build(positionablePoi as ApiEvent);

            return positionablePoi;
          })
        )
      );
  }

  // -- EVENTS --

  /**
   * Save Event poi for a location
   *
   * @param event The Event object
   * @param locationUuid The Uuid of the Application/Location
   */
  public saveEvent(event: EventPoi, locationUuid: string): Observable<EventPoi> {
    const params = new HttpParams()
      .set('app', this.context.applicationName)
      .set('locationUuid', locationUuid);

    return this.http
      .post<ApiEvent>(`${this.EVENT_URI}`, event.toApi(), { params })
      .pipe(map((apiEvent: ApiEvent) => EventPoi.build(apiEvent)));
  }

  /**
   * Retrieve all Event pois by the location uuid
   *
   * @param locationUuid The Uuid of the Application/Location
   */
  public getEvents(locationUuid: string): Observable<EventPoi[]> {
    const params = new HttpParams().set('locationUuid', locationUuid);

    return this.http
      .get<ApiEvent[]>(`${this.EVENT_URI}`, { params })
      .pipe(map((apiEvents: ApiEvent[]) => apiEvents.map(apiEvent => EventPoi.build(apiEvent))));
  }

  /**
   * Update existing Event poi for a location
   *
   * @param eventPoi The Event object
   * @param locationUuid The Uuid of the Application/Location
   */
  public updateEvent(eventPoi: EventPoi, locationUuid: string): Observable<EventPoi> {
    const params = new HttpParams()
      .set('app', this.context.applicationName)
      .set('locationUuid', locationUuid);

    return this.http
      .put<ApiEvent>(`${this.EVENT_URI}`, eventPoi.toApi(), { params })
      .pipe(map((apiEvent: ApiEvent) => EventPoi.build(apiEvent)));
  }

  /**
   * Delete existing Event poi
   *
   * @param eventPoi The Event object
   */
  public deleteEvent(eventPoi: EventPoi): Observable<void> {
    const params = new HttpParams().set('app', this.context.applicationName);

    return this.http.delete<void>(`${this.EVENT_URI}/${eventPoi.id}`, { params });
  }

  /**
   * Retrieve all Event pois by the resume node uuid
   *
   * @param   resumeNodeUuid The Uuid of the resume node
   * @param   locationUuid The Uuid of the Application/Location
   * @param   eventFilters The list of filters
   */
  public events(
    locationUuid: string,
    resumeNodeUuid: string,
    eventFilters?: Filters
  ): Observable<EventPoi[]> {
    let params = new HttpParams().set('resumeNodeUuid', resumeNodeUuid);

    if (locationUuid) params = params.set('locationUuid', locationUuid);

    return this.http
      .post<ApiEvent[]>(`${this.EVENT_URI}/filters`, eventFilters ? eventFilters.toApi() : {}, {
        params
      })
      .pipe(map((apiEvents: ApiEvent[]) => apiEvents.map(apiEvent => EventPoi.build(apiEvent))));
  }

  public cursusPois(applicationId: string, nodeId?: string): Observable<ParcoursPoi[]> {
    let params = new HttpParams().set('app', applicationId);

    if (nodeId) params = params.set('node', nodeId);

    return this.http.get<ParcoursPoi[]>(`${this.BASE_URL}/type/cursuses`, {
      params
    });
  }

  // -- MOOCS --

  /**
   * Retrieve all Mooc pois by the resume node uuid
   *
   * @param   resumeNodeUuid The Uuid of the resume node
   * @param   label The label of the resume node
   */
  public moocs(resumeNodeUuid: string, label?: string): Observable<Mooc[]> {
    let params = new HttpParams().set('resumeNodeUuid', resumeNodeUuid);

    if (label) params = params.set('label', label);

    return this.http.get<Mooc[]>(`${this.MOOC_URI}`, {
      params
    });
  }

  // -- VIDEOS --

  /**
   * Retrieve all Videos by the resume node uuid
   *
   * @param   resumeNodeUuid The Uuid of the resume node
   */
  public videos(resumeNodeUuid: string): Observable<string[]> {
    const params = new HttpParams().set('resumeNodeUuid', resumeNodeUuid);

    return this.http
      .get<{ url: string }[]>(`${this.VIDEO_URI}`, {
        params
      })
      .pipe(map(videos => videos.map(video => video.url)));
  }

  // -- GENERATE POI --

  public createPoisFromOffers<P extends Poi>(
    provider: string,
    applicationUuid,
    offers: Offer[]
  ): Observable<P[]> {
    const params = new HttpParams().set('app', applicationUuid);

    return this.http.post<P[]>(
      `${this.BASE_URL}/${provider}/fromoffers`,
      offers.map(offer => offer.toApi()),
      { params }
    );
  }
}
