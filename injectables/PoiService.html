import { Injectable } from '@angular/core';
import { Score } from '@clients/adventure/bean/instance-score';
import { isDiploma, Waypoint } from '@clients/adventure/bean/waypoint';
import { Poi } from '@clients/poi/poi';
import { PoiClient } from '@clients/poi/poi.client';
import { ApplicationContext, HroadsService } from '@core/services';
import { Utils } from '@features/map/utils';
import { Filters } from '@models/filters/filters';
import { Gender } from '@models/gender/gender';
import { KpiItem } from '@models/kpi.model';
import { EventPoi } from '@models/poi/event/event-poi';
import { Mooc } from '@models/poi/mooc/mooc';
import { ParcoursPoi } from '@models/poi/parcours-poi/parcours-poi';
import { PoiProvider } from '@models/poi/poi-provider';
import { PositionablePoi } from '@models/poi/positionable-poi/positionable-poi';
import { TeamPoi } from '@models/poi/team-poi/team-poi';
import { Skill } from '@models/skill/skill';
import { TranslateService } from '@ngx-translate/core';
import { forkJoin, Observable } from 'rxjs';
import { filter, map, switchMap } from 'rxjs/operators';
import { InstanceService } from '../instance.service';

@Injectable()
export class PoiService {
  constructor(
    private client: PoiClient,
    private context: ApplicationContext,
    private hroads: HroadsService,
    private instanceService: InstanceService,
    private translateService: TranslateService
  ) {}

  // POIS

  /**
   * Get a POI
   *
   * @param provider The provider of the POI
   * @param poiId The id of the POI
   * @param instanceId The id of the instance
   * @returns The matching POI
   */
  public poi(provider: string, poiId: string, instanceId: string): Observable<Poi> {
    if (provider === PoiProvider.human) return this.instanceService.humanPoi(instanceId, poiId);

    return this.client.poi(provider, poiId);
  }

  // PARCOURS POI

  public positionablePois(waypoint: Waypoint, poiFilters?: Filters): Observable<PositionablePoi[]> {
    return this.client
      .positionablePois(this.context.application.nodeId, waypoint.id, waypoint.type, poiFilters)
      .pipe(filter(pois => !!pois && !!pois.length));
  }

  /**
   * Retrieve all pois of Event type from a diploma and locations
   */
  public events(waypoint: Waypoint, eventFilters?: Filters): Observable<EventPoi[]> {
    return this.client.events(this.context.application.nodeId, waypoint.id, eventFilters);
  }

  /**
   * Retrieve all Pois except type = "event"
   */
  public cursusPois(waypoint: Waypoint): Observable<ParcoursPoi[]> {
    return this.client.cursusPois(this.context.application.nodeId, waypoint.id);
  }

  /**
   * Retrieve all videos from a waypoint
   */
  public videos(waypoint: Waypoint): Observable<string[]> {
    return this.client.videos(waypoint.id);
  }

  /**
   * Retrieve all moocs
   */
  public moocs(waypoint: Waypoint): Observable<Mooc[]> {
    return this.client.moocs(waypoint.id);
  }

  /**
   * Get the team POI
   *
   * @param humanPoiId The id of one of the human POIs of the village
   * @param waypoint The current waypoint
   * @returns The team POI
   */
  public teamPoi(humanPoiId: string, waypoint: Waypoint): TeamPoi {
    return new TeamPoi(
      humanPoiId,
      this.translateService.instant('team.poi.name'),
      this.translateService.instant(
        isDiploma(waypoint.type) ? 'team.poi.description-diploma' : 'team.poi.description-job'
      ),
      Utils.randomBetween(0, 1) ? Gender.man : Gender.lady
    );
  }

  /**
   * Compute the score details from winner POIs of an instance
   *
   * @param winners The winner POIS array from the instance score
   * @param instanceId The Id of the instance
   * @returns An observable emitting the score details associating each POI type to an array of POIs
   */
  public computeScoreDetails(winners: Score[], instanceId: string): Observable<Map<string, Poi[]>> {
    return forkJoin(winners.map(score => this.poi(score.metadata, score.id, instanceId))).pipe(
      map(pois =>
        pois
          .filter(poi => !!poi)
          .reduce((res, poi) => {
            (res[poi.type] = res[poi.type] || []).push(poi);

            return res;
          }, {} as Map<string, Poi[]>)
      )
    );
  }

  /**
   * KPIs skills for a waypoint
   */
  public skills(waypoint: Waypoint): Observable<Skill[]> {
    return this.hroads.kpi(waypoint, 'skills', HroadsService.SKILLS_LIMIT).pipe(
      switchMap((kpiItems: KpiItem[]) =>
        forkJoin(
          kpiItems.map(item =>
            this.moocsForKpi(item, 'Skill').pipe(
              map(moocs => ({
                title: item.name,
                percentage: Math.floor((item.nb_resumes / item.count_u) * 100),
                moocs
              }))
            )
          )
        )
      )
    );
  }

  /**
   * Retrieve all moocs based on Skills
   */
  private moocsForKpi(kpiItems: KpiItem, label: string): Observable<Mooc[]> {
    return this.client.moocs(kpiItems.uuid, label);
  }
}
