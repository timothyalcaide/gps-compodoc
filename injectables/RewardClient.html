import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { InstanceScore, Score } from '@clients/adventure/bean/instance-score';
import { UserScore } from '@clients/adventure/bean/user-score';
import { ApplicationContext } from '@core/services';
import { InstanceDtoReward } from '@models/instance/instance-dto-reward';
import { ApiLeadReward, LeadReward } from '@models/lead/lead-reward';
import { Reward } from '@models/reward/reward';
import { RewardType } from '@models/reward/reward-type';
import { environment } from 'environments/environment';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';

@Injectable({ providedIn: 'root' })
export class RewardClient {
  private readonly REWARDS_URI = `${environment.adventureUri}/rewards`;
  private readonly USER_SCORE_URI = `${environment.adventureUri}/user-scores`;
  private readonly INSTANCE_SCORE_URI = `${environment.adventureUri}/instance-scores`;

  constructor(private context: ApplicationContext, private http: HttpClient) {}

  /* -- REWARD -- */

  public rewards(
    enabled?: boolean,
    type?: RewardType,
    applicationName?: string
  ): Observable<Reward[]> {
    let params = new HttpParams().set('a', applicationName || this.context.applicationName);

    if (enabled) params = params.set('e', `${enabled}`);

    if (type) params = params.set('t', `${type}`);

    return this.http.get<Reward[]>(this.REWARDS_URI, { params });
  }

  public reward(id: string): Observable<Reward> {
    const params = new HttpParams();

    return this.http.get<Reward>(`${this.REWARDS_URI}/${id}`, { params });
  }

  public saveReward(reward: Reward): Observable<Reward> {
    return this.http.post<Reward>(this.REWARDS_URI, reward);
  }

  public updateReward(reward: Reward): Observable<Reward> {
    return this.http.put<Reward>(this.REWARDS_URI, reward);
  }

  public deleteReward(reward: Reward): Observable<void> {
    return this.http.delete<void>(`${this.REWARDS_URI}/${reward.id}`);
  }

  /* -- USER SCORE -- */

  public userScore(email: string): Observable<UserScore> {
    const params = new HttpParams().set('a', this.context.applicationName).set('e', email);

    return this.http.get<UserScore>(this.USER_SCORE_URI, { params });
  }

  public saveUserScore(userScore: UserScore): Observable<UserScore> {
    return this.http.post<UserScore>(this.USER_SCORE_URI, userScore);
  }

  public updateUserScore(userScore: UserScore): Observable<UserScore> {
    return this.http.put<UserScore>(this.USER_SCORE_URI, userScore);
  }

  /* -- SCORE -- */

  public score(instanceScoreId: string, score: Score): Observable<Score> {
    return this.http.post<Score>(`${this.INSTANCE_SCORE_URI}/${instanceScoreId}/scores`, score);
  }

  /* -- INSTANCE SCORE -- */
  /**
   * Create instance score
   *
   * @param instanceDto The instanceScore
   */
  public create(instanceDto: InstanceDtoReward): Observable<InstanceScore> {
    return this.http.post<InstanceScore>(`${this.INSTANCE_SCORE_URI}`, instanceDto.toApi());
  }

  /**
   * Get instanceScore
   *
   * @param id of instance score
   */
  public instanceScore(id: string): Observable<InstanceScore> {
    return this.http.get<InstanceScore>(`${this.INSTANCE_SCORE_URI}/${id}`);
  }

  // -- CLAIM --

  /**
   * Create a lead for claim reward & email the reward's owner
   *
   * @param leadReward The lead reward
   */
  public claimReward(leadReward: LeadReward): Observable<LeadReward> {
    return this.http
      .post<ApiLeadReward>(`${this.REWARDS_URI}/claim-reward`, leadReward)
      .pipe(map(lead => LeadReward.build(lead)));
  }

  /**
   * Check if a lead for claim reward exists
   *
   * @param userEmail The user email
   * @param rewardId The reward ID
   */
  public getClaimReward(userEmail: string, rewardId: string): Observable<LeadReward> {
    const params = new HttpParams().set('e', userEmail).set('rid', rewardId);

    return this.http.get<LeadReward>(`${this.REWARDS_URI}/claim-reward`, { params });
  }

  /**
   * Get all the rewards claimed by the authenticated user, enabled AND disabled
   *
   * @returns An observable emitting the list of rewards claimed by the authenticated user
   */
  public claimedRewards(): Observable<Reward[]> {
    return this.http.get<Reward[]>(`${this.REWARDS_URI}/claimed-reward`);
  }
}
