import { Injectable, OnDestroy } from '@angular/core';
import { InstanceScore, Score, ScoreType } from '@clients/adventure/bean/instance-score';
import { UserScore } from '@clients/adventure/bean/user-score';
import { Poi } from '@clients/poi/poi';
import { GAMIFICATION_ENABLED_CONFIG } from '@configs/gamification-enabled.config';
import { ApplicationContext, AuthService } from '@core/services';
import { InstanceDtoReward } from '@models/instance/instance-dto-reward';
import { LeadReward } from '@models/lead/lead-reward';
import { Reward } from '@models/reward/reward';
import { RewardType } from '@models/reward/reward-type';
import { BehaviorSubject, Observable, of, Subject } from 'rxjs';
import { filter, map, mapTo, mergeMap, switchMap, switchMapTo, tap } from 'rxjs/operators';
import { RewardClient } from '../clients';

@Injectable({ providedIn: 'root' })
export class RewardService implements OnDestroy {
  /**
   * The list of enabled rewards of the current application
   */
  public enabledRewards$ = new BehaviorSubject<Reward[]>([]); // TODO : Move to context service

  private instanceScore$ = new BehaviorSubject<InstanceScore>(null); // TODO : Move to context service
  private onDestroySubject$: Subject<void> = new Subject<void>();

  // TODO : Move to context service

  public set userScore(userScore: UserScore) {
    localStorage.setItem('userScore', JSON.stringify(userScore));
  }

  public get userScore(): UserScore {
    return JSON.parse(localStorage.getItem('userScore'));
  }

  public get instanceScore(): InstanceScore {
    return this.instanceScore$.value;
  }

  private get EMPTY_USER_SCORE(): UserScore {
    return {
      application: this.context.applicationName,
      email: this.authService.email,
      points: 0
    } as UserScore;
  }

  constructor(
    private authService: AuthService,
    private client: RewardClient,
    private context: ApplicationContext
  ) {}

  public ngOnDestroy(): void {
    this.onDestroySubject$.next();
    this.onDestroySubject$.complete();
  }

  public init(): Observable<void> {
    return new Observable(subscriber => {
      subscriber.next(
        GAMIFICATION_ENABLED_CONFIG(this.context.product, this.context.applicationName)
      );

      subscriber.complete();
    }).pipe(
      filter(gamificationEnabled => !!gamificationEnabled),
      switchMapTo(this.getUserScore()),
      switchMapTo(this.enabledRewards()),
      map(() => {})
    );
  }

  /**
   * Get the list of rewards that the user have not won
   *
   * @returns The list of rewards that the user have not won
   */
  public rewardsFuture(): Observable<Reward[]> {
    return this.enabledRewards$.pipe(
      map(enabledRewards =>
        enabledRewards.filter(enabledReward => this.userScore.points < enabledReward.points)
      )
    );
  }

  /**
   * Get the next reward that the user can win
   *
   * @returns The next reward that the user can win
   */
  public futureReward(): Observable<Reward> {
    return this.rewardsFuture().pipe(
      map(futureRewards => (futureRewards.length ? futureRewards[0] : null))
    );
  }

  // -- USER SCORE --

  public addScore(s: Score): void {
    const us = this.userScore;

    us.points += s.points;
    this.userScore = us;
  }

  public scoringEe(): Observable<Score> {
    return this.scoring(this.instanceScore.id, {
      id: null,
      type: ScoreType.EE
    });
  }

  public scoringPoi(poi: Poi): Observable<Score> {
    return this.scoring(this.instanceScore.id, {
      id: poi.id,
      type: ScoreType.POI,
      metadata: poi.provider
    });
  }

  // -- INSTANCE SCORE --

  public showEasterEgg(): boolean {
    if (!this.instanceScore) return false;

    return (
      !this.instanceScore.easterEgg &&
      this.instanceScore.winners.length >= this.instanceScore.configuration.unlockEE
    );
  }

  /**
   * Create Instance Score based on InstanceDtoReward
   *
   * @param instanceDtoReward Instance object with minimal properties
   */
  public createInstanceScore(instanceDtoReward: InstanceDtoReward): Observable<InstanceScore> {
    return this.client.create(instanceDtoReward).pipe(
      tap(instanceScore => {
        this.instanceScore$.next(instanceScore);
      })
    );
  }

  /**
   * Get Instance score
   *
   * @param id of Instance Score
   */
  public getInstanceScore(id: string): Observable<InstanceScore> {
    return this.client.instanceScore(id).pipe(
      tap(instanceScore => {
        this.instanceScore$.next(instanceScore);
      })
    );
  }

  // -- REWARD --

  /**
   * Retrieve a reward by id
   */
  public reward(id: string): Observable<Reward> {
    return this.client.reward(id);
  }

  /**
   * Retrieve the reward for the default interview student if it's enable
   */
  public studentInterviewReward(): Observable<Reward> {
    return this.enabledRewards(RewardType.STUDENT_INTERVIEW).pipe(
      // Get only student interview rewards that are enabled and have a contact email
      filter((rewards: Array<Reward>) => rewards.length > 0),
      map((rewards: Array<Reward>) => rewards[0]),
      filter((reward: Reward) => !!reward && reward.enabled && !!reward.email)
    );
  }

  /**
   * Get the list of rewards that the user has won
   *
   * @returns The list of rewards that the user has won
   */
  public rewardsWon(): Observable<Reward[]> {
    return this.enabledRewards$.pipe(
      map(enabledRewards => enabledRewards.filter(r => this.userScore.points >= r.points))
    );
  }

  public rewardsWonNb(): Observable<number> {
    return this.rewardsWon().pipe(map(wonRewards => wonRewards.length));
  }

  /**
   * Get the list of points of application rewards
   *
   * @param claimedRewards The list of rewards of the application
   * @returns An observable of the list of points of application rewards
   */
  public rewardPoints(claimedRewards?: Reward[]): Observable<number[]> {
    return this.enabledRewards$.pipe(
      map(enabledRewards =>
        [...enabledRewards, ...claimedRewards].map(r => r.points).sort((a, b) => a - b)
      )
    );
  }

  // -- CLAIM --

  /**
   * Check if a lead for reward exists
   *
   * @param rewardId The reward ID
   */
  public hasClaimReward(rewardId: string): Observable<boolean> {
    return this.client
      .getClaimReward(this.authService.email, rewardId)
      .pipe(map((leadReward: LeadReward) => !!leadReward));
  }

  /**
   * Create a lead for claim reward & email the reward's owner
   *
   * @param rewardId The id of the linked Reward
   */
  public claimReward(rewardId: string): Observable<LeadReward> {
    return this.client.claimReward(
      LeadReward.toLeadReward(this.context.applicationName, this.authService.email, rewardId)
    );
  }

  /**
   * Check if a lead for claim reward exists
   *
   * @param rewardId The reward ID
   */
  public getClaimReward(rewardId: string): Observable<LeadReward> {
    return this.client.getClaimReward(this.authService.email, rewardId);
  }

  /**
   * Get all the rewards claimed by the authenticated user, enabled AND disabled
   *
   * @returns An observable emitting the list of rewards claimed by the authenticated user
   */
  public claimedRewards(): Observable<Reward[]> {
    return this.client.claimedRewards();
  }

  // -- USER SCORE

  public saveUserScore(): Observable<UserScore> {
    localStorage.setItem('userScore', JSON.stringify(this.userScore));

    if (!this.userScore.application || !this.userScore.email) return of(this.userScore);

    if (!this.userScore.id) return this.client.saveUserScore(this.userScore);

    return this.client.updateUserScore(this.userScore);
  }

  /**
   * Get the enabled rewards of a school
   *
   * @param applicationName The name of the application of the school
   * @returns The lis of enabled rewards of the school
   */
  public schoolEnabledRewards(applicationName: string): Observable<Reward[]> {
    return this.client.rewards(true, null, applicationName).pipe(
      map(rewards => rewards || []),
      map(rewards => rewards.sort((a, b) => a.points - b.points))
    );
  }

  /**
   * Retrieve enabled rewards
   * Can be filter by type
   */
  private enabledRewards(type?: RewardType): Observable<Reward[]> {
    return this.client.rewards(true, type).pipe(
      map(rewards => rewards.sort((a, b) => a.points - b.points)),
      tap(rewards => {
        this.enabledRewards$.next(rewards);
      })
    );
  }

  private scoring(instanceScoreId: string, score: Score): Observable<Score> {
    return this.client.score(instanceScoreId, score).pipe(
      tap(s => {
        this.addScore(s);
      }),
      mergeMap(s => this.saveUserScore().pipe(mapTo(s)))
    );
  }

  private getUserScore(): Observable<UserScore> {
    return of(this.userScore).pipe(
      switchMap((localStorageUserScore: UserScore) => {
        if (localStorageUserScore?.application === this.context.applicationName) {
          if (this.authService.email) {
            if (!localStorageUserScore.email) {
              /**
               * The user is authenticated and the user score in the local storage is no one's so it get
               * the current user's user score and merge it with the user score in the local storage
               */
              return this.client.userScore(this.authService.email).pipe(
                map(us => us || this.EMPTY_USER_SCORE),
                mergeMap(us => {
                  if (!us.id) return this.client.saveUserScore(us);

                  return this.client.updateUserScore({
                    ...us,
                    points: us.points + localStorageUserScore.points
                  } as UserScore);
                })
              );
            }

            /**
             * The user is authenticated and there is a user score in the local storage that does not
             * belong to the user so it get the current user's user score
             */
            return this.client.userScore(this.authService.email);
          }

          if (localStorageUserScore.email) {
            /**
             * The user is not authenticated and an associated user score is in the local storage so the
             * current user has no user score
             */
            return of(this.EMPTY_USER_SCORE);
          }

          /**
           * The user is not authenticated and a user score not associated is in the local storage so
           * it becomes the current user's user score
           */
          return of(localStorageUserScore);
        }

        if (this.authService.email) {
          /**
           * The user is authenticated and there is no user score for the application in the local
           * storage so it get the current user's user score
           */
          return this.client.userScore(this.authService.email);
        }

        /**
         * The user is not authenticated and the local storage is empty so the current user has no
         * user score
         */
        return of(this.EMPTY_USER_SCORE);
      }),
      map(us => us || this.EMPTY_USER_SCORE),
      tap(userScore => {
        // Saves the new user score to the local storage
        this.userScore = userScore;
      })
    );
  }
}
