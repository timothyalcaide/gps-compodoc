import { Injectable } from '@angular/core';
import { Status } from '@clients/adventure/bean/situation';
import { SituationClient } from '@clients/adventure/situation.client';
import { ApplicationContext } from '@core/services';
import { BacSpecialtiesGroup } from '@models/bac-specialties-group/bac-specialties-group';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';

@Injectable()
export class SituationService {
  /* -- STATUS -- */

  // Status not bac
  private readonly STATUSES = [
    Status.middleschool,
    Status.highschool,
    Status.student,
    Status.pro,
    Status.unemployed
  ];

  private readonly BACS = [
    Status.Bac,
    Status.Bac2,
    Status.Bac3,
    Status.Bac4,
    Status.Bac5,
    Status.Bac6,
    Status.Bac8
  ];

  constructor(private client: SituationClient, private context: ApplicationContext) {}

  public statusesGlobal(): Observable<Status[]> {
    return this.statuses().pipe(
      map(statuses => this.handleBacStatus(statuses)),
      map(statuses =>
        statuses
          .filter(s => this.STATUSES.includes(s))
          .sort((a, b) => this.STATUSES.indexOf(a) - this.STATUSES.indexOf(b))
      )
    );
  }

  public statusesBac(): Observable<Status[]> {
    return this.statuses().pipe(
      map(statuses =>
        statuses
          .filter(s => this.BACS.includes(s))
          .sort((a, b) => this.BACS.indexOf(a) - this.BACS.indexOf(b))
      )
    );
  }

  /**
   * Get recommended bac specialties of a waypoint
   *
   * @param waypointId The Id of the waypoint
   * @returns The list of recommended bac specialties
   */
  public recommendedBacSpecialties(waypointId: string): Observable<BacSpecialtiesGroup[]> {
    return this.client.recommendedBacSpecialties(waypointId);
  }

  // -- PRIVATE --

  private statuses(): Observable<Status[]> {
    return this.client.statuses();
  }

  /**
   * Replace Status.Bac* by Status.Student
   */
  private handleBacStatus(statuses: Status[]): Status[] {
    let flag = false;

    statuses.forEach(s => {
      if (s.startsWith(Status.Bac.toString())) flag = true;
    });

    if (flag) {
      if (!statuses.includes(Status.student)) statuses.push(Status.student);

      return statuses.filter(s => !this.BACS.includes(s));
    }

    return statuses;
  }
}
