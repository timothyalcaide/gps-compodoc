import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { AuthService } from '@core/services';
import { User } from '@models/user/user';
import { ApiUser } from '@models/user/user.api';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';

const UAA_URI = '/apis/uaa';
const USER_URI = `${UAA_URI}/users`;

/**
 * Client responsible of user management
 */
@Injectable()
export class UaaClient {
  constructor(private http: HttpClient, private authService: AuthService) {}

  /* -- USERS -- */

  /**
   * Check if a user exists
   *
   * @param email The user email
   * @returns True if the user exists; false otherwise
   */
  public exists(email: string): Observable<boolean> {
    return this.http.get<boolean>(`${USER_URI}/${email}/exists`);
  }

  /**
   * Create a user without setting a password and send an email with a link to initialise password
   *
   * @param newUser The user to register without password
   */
  public passwordless(newUser: User): Observable<void> {
    return this.http.post<void>(`${USER_URI}/passwordless`, newUser.toApi(), {
      params: {
        clientId: this.authService.clientId,
        redirectUri: this.authService.redirectUri
      }
    });
  }

  /**
   * Send an email to the user to reinitialize their password
   *
   * @param email The user email
   */
  public updatePassword(email: string): Observable<void> {
    return this.http.post<void>(`${USER_URI}/update-password`, email, {
      params: {
        clientId: this.authService.clientId,
        redirectUri: this.authService.redirectUri
      }
    });
  }

  // -- ADMIN --

  /**
   * Get the user details from its email
   *
   * @param email The user email
   * @returns The full user information
   */
  public getUser(email: string): Observable<User> {
    return this.http.get<ApiUser>(`${USER_URI}/${email}`).pipe(map(apiUser => User.build(apiUser)));
  }
}
