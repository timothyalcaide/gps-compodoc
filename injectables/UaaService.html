import { Injectable } from '@angular/core';
import { UaaClient } from '@clients/uaa/uaa.client';
import { Partner } from '@models/partner/partner';
import { PartnerId } from '@models/partner/partner-id';
import { User } from '@models/user/user';
import { Observable, of } from 'rxjs';
import { mapTo, switchMap } from 'rxjs/operators';

/**
 * Service responsible of user management
 */
@Injectable()
export class UaaService {
  constructor(private client: UaaClient) {}

  /* -- USERS -- */

  /**
   * Check if a user exists
   *
   * @param email The user email
   * @returns True if the user exists; false otherwise
   */
  public exists(email: string): Observable<boolean> {
    return this.client.exists(email);
  }

  /**
   * Create a user without setting a password and send an email with a link to initialise password
   *
   * @param newUser The user to register without password
   */
  public passwordless(newUser: User): Observable<void> {
    return this.client.passwordless(newUser);
  }

  /**
   * Send an email to the user to reinitialize their password
   *
   * @param email The user email
   */
  public updatePassword(email: string): Observable<void> {
    return this.client.updatePassword(email);
  }

  /**
   * Check if the user exists. Return the user if it exists, create it and return it otherwise
   *
   * @param email The user email
   * @param firstName The user first name
   * @param lastName The user last name
   * @param lEtudiantOptIn If the user opts in for L'Ã‰tudiant. Only for MyRoad
   * @returns An observable emitting the user object
   */
  public checkIfUserExistsAndGetOrCreate(
    email,
    firstName,
    lastName,
    lEtudiantOptIn: boolean | null
  ): Observable<User> {
    return this.exists(email).pipe(
      switchMap(exists => {
        const newUser = new User(
          email,
          firstName,
          lastName,
          [],
          [],
          '',
          '',
          lEtudiantOptIn !== null ? [new Partner(lEtudiantOptIn, PartnerId.LEtudiant)] : []
        );

        if (exists) return of(newUser);

        // If the user does not exists, create an account without password
        return this.passwordless(newUser).pipe(mapTo(newUser));
      })
    );
  }
}
