  constructor(
    private authService: AuthService,
    private client: VideoAskClient,
    private context: ApplicationContext
  ) {}

  public getMockedHumanPoisWithVideoAsks(
    humanPois: HumanPoi[],
    waypointId: string,
    isAlumni?: boolean
  ): HumanPoi[] {
    if (
      this.context.applicationName === ApplicationName.CmqBtpNumerique &&
      isAlumni &&
      (waypointId === this.DIPLOME_INGENIEUR_ID ||
        waypointId === this.DIRECTEUR_DE_FILLIALE_ID ||
        waypointId === this.ELECTRICIEN_ID)
    )
      return [this.LIONEL_HUMAN_POI, ...humanPois.slice(0, -1)];

    return humanPois;
  }

  /**
   * Get the URL of a VideoAsk
   *
   * @param videoAskId The id of the VideoAsk
   * @returns The URL of the VideoAsk
   */
  public getVideoAskUrl(videoAskId: string): string {
    let videoAskVariables = '';
    const user = this.authService.userSessionStorage;

    if (user) {
      videoAskVariables = `#${this.VIDEO_ASK_CONTACT_EMAIL_PARAM_KEY}=${encodeURIComponent(
        user.email
      )}&${this.VIDEO_ASK_CONTACT_NAME_PARAM_KEY}=${user.firstName} ${user.lastName}`;
    }

    return `${this.VIDEO_ASK_HOST}${videoAskId}${videoAskVariables}`;
  }

  /**
   * Get the URL of the thumbnail preview of a VideoAsk
   *
   * @param videoAskId The id of the VideoAsk
   * @returns An observable emitting the URL of the thumbnail preview (the video and the GIF as fallback)
   */
  public getThumbnailPreviewUrl(videoAskId: string): Observable<{ video: string; gif: string }> {
    return this.client.getSharingInfo(videoAskId).pipe(
      map(videoAskSharing => ({
        video: videoAskSharing.questions[0].media_url.replace('video.mp4', 'embed.mp4'),
        gif: videoAskSharing.questions[0].gif
      }))
    );
  }
}
