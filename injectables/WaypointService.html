import { Injectable } from '@angular/core';
import { Application } from '@clients/adventure/bean/application';
import { Waypoint } from '@clients/adventure/bean/waypoint';
import { Nomenclature } from '@clients/poi/poi';
import { ApplicationContext, WaypointClient } from '@core/services';
import { Company } from '@models/company/company';
import { Filters } from '@models/filters/filters';
import { Gender } from '@models/gender/gender';
import { Instance } from '@models/instance/instance';
import { PartnerName } from '@models/partner/partner-name';
import { HumanPoi } from '@models/poi/human-poi/human-poi';
import { School } from '@models/school/school';
import { Structure } from '@models/structure/structure';
import { OriginAndDestination } from '@models/waypoint/origin-with-destination';
import { WaypointType } from '@models/waypoint/waypoint-type';
import { Observable } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class WaypointService {
  constructor(private client: WaypointClient, public ctx: ApplicationContext) {}

  /**
   * Get a waypoint from an id
   *
   * @param waypointId The waypoint id
   * @param gender The user gender
   * @returns The matching waypoint
   */
  public waypoint(waypointId: string, gender: Gender): Observable<Waypoint> {
    return this.client.waypoint(waypointId, gender);
  }

  /* -- SCHOOL -- */

  /**
   * Retrieve schools associated to a given waypoint within a given application, optionally using
   * filters
   *
   * @param waypoint The waypoint
   * @param application The application
   * @param schoolFilters Optional filters
   * @param limit The number of results allowed, default 100
   * @returns The matching schools
   */
  public schools(
    waypoint: Waypoint,
    application: Application,
    schoolFilters?: Filters,
    limit: number = 100
  ): Observable<School[]> {
    return this.client.schools(
      waypoint.id,
      application.nodeId,
      waypoint.type,
      schoolFilters,
      limit
    );
  }

  /**
   * Retrieve a school associated to a given waypoint
   *
   * @param waypoint The waypoint
   * @param application The application
   * @param schoolId The school Id
   * @returns The matching school
   */
  public school(
    waypoint: Waypoint,
    application: Application,
    schoolId: string
  ): Observable<School> {
    return this.client.school(waypoint.id, application.nodeId, schoolId);
  }

  /* -- STRUCTURE -- */

  /**
   * Retrieve structure within a given application, optionally using
   * filters
   *
   * @param applicationNodeId The application node id
   * @param structureFilters Optional filters
   * @param limit The number of results allowed, default 100
   * @returns The matching structures
   */
  public structures(
    applicationNodeId: string,
    structureFilters?: Filters,
    limit: number = 100
  ): Observable<Structure[]> {
    return this.client.structures(applicationNodeId, structureFilters, limit);
  }

  /* -- ORIGINS -- */

  /**
   * Get origin waypoints for a destination and a user gender
   *
   * @param destinationId The destination Id
   * @param originFilters Optional filters
   */
  public origins(destinationId: string, originFilters?: Filters): Observable<Waypoint[]> {
    return this.client.origins(destinationId, originFilters);
  }

  /**
   * Get an origin and/or a destination from a partner reference
   *
   * @param partnerName The name of the partner
   * @param id The reference id to the partner item
   * @param type The type of waypoint
   * @returns An observable of the waypoints matching the partner reference
   */
  public fromPartner(
    partnerName: PartnerName,
    id: string,
    type?: WaypointType
  ): Observable<OriginAndDestination> {
    return this.client.fromPartner(partnerName, id, type);
  }

  /* -- MAP -- */

  /**
   * Retrieve Waypoint backward routes
   */
  public backwards(waypoint: Waypoint, instance: Instance): Observable<Waypoint[]> {
    return this.client.backward(
      waypoint,
      instance.definition.application,
      instance.situation.gender
    );
  }

  /* -- RESUME NODE -- */

  /**
   * Find resume nodes (jobs or diplomas) matching a given search text based on the user status
   *
   * @param applicationNodeId The application node Id
   * @param resumeNodeType The resume node type (or waypoint type). Can be Job or Diploma
   * @param searchText The search text
   * @param trendsFilters The filters containing the region used to get the job trends
   * @returns The waypoint results
   */
  public resumeNodes(
    applicationNodeId: string,
    resumeNodeType: WaypointType,
    searchText: string,
    trendsFilters: Filters
  ): Observable<Waypoint[]> {
    return this.client.resumeNodes(applicationNodeId, resumeNodeType, searchText, trendsFilters);
  }

  /* -- COMPANY -- */

  /**
   * Retrieves companies associated to a given waypoint within a given application, optionally using
   * filters
   *
   * @param waypoint The waypoint
   * @param application The application
   * @param schoolFilters Optional filters
   * @param limit The number of results allowed, default 100
   * @returns The matching companies
   */
  public companies(
    waypoint: Waypoint,
    application: Application,
    schoolFilters?: Filters,
    limit: number = 100
  ): Observable<Company[]> {
    return this.client.companies(
      waypoint.id,
      application.nodeId,
      waypoint.type,
      schoolFilters,
      limit
    );
  }

  /* -- HUMAN -- */

  /**
   * Get Humans POIs for a waypoint
   *
   * @param waypointId The waypoint ID
   * @param nomenclature The Human POIs nomenclature. Can be either alumni or student
   * @param filters The waypoint filters, used to get the application
   * @returns An array of HumanPois for this waypoint
   */
  public humanPois(
    waypointId: string,
    nomenclature: Nomenclature,
    filters: Filters
  ): Observable<HumanPoi[]> {
    return this.client.humanPois(waypointId, nomenclature, filters);
  }
}
