import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { ApplicationContext } from '@core/services';
import { MinDateValidator } from '@directives/custom-validators/min-date-validator.directive';
import { EventPoi } from '@models/poi/event/event-poi';
import { ParcoursPoi } from '@models/poi/parcours-poi/parcours-poi';
import { PoiProvider } from '@models/poi/poi-provider';
import dayjs from 'dayjs';

/**
 * The dialog data interface
 */
interface AdminPoiDialogData {
  /**
   * The POI to edit; is null if the dialog is opened for creating a new POI
   */
  poi: EventPoi | ParcoursPoi;
  /**
   * The POI provider (event or parcours)
   */
  provider: PoiProvider;
}

@Component({
  selector: 'app-admin-poi-dialog',
  templateUrl: './admin-poi-dialog.component.html'
})
export class AdminPoiDialogComponent implements OnInit {
  public form: FormGroup;
  public poi: EventPoi | ParcoursPoi = null;
  public isUpdate = false;

  private provider: PoiProvider;

  public get isEventPoi(): boolean {
    return this.provider === PoiProvider.neo4j;
  }

  constructor(
    @Inject(MAT_DIALOG_DATA) public data: AdminPoiDialogData,
    private context: ApplicationContext,
    private dialog: MatDialogRef<AdminPoiDialogComponent>,
    private formBuilder: FormBuilder
  ) {
    this.poi = data.poi;
    this.isUpdate = !!data.poi?.id;
    this.provider = data.provider;
  }

  /**
   * Initialise dialog form according to POI type (event or parcours)
   */
  public ngOnInit(): void {
    if (this.isEventPoi) {
      // Initialise dialog form with event POI fields
      this.groupEventPoiForm();
    } else {
      // Initialise dialog form with parcours POI fields
      this.groupParcoursPoiForm();
    }
  }

  public onSave(): void {
    if (this.poi instanceof EventPoi) {
      this.dialog.close(
        new EventPoi(
          this.form.controls.name.value,
          dayjs(this.form.controls.start.value),
          dayjs(this.form.controls.end.value),
          this.form.controls.description.value,
          this.form.controls.url.value,
          this.poi?.id,
          this.poi?.displayName,
          this.poi?.school,
          this.poi?.schoolUuid,
          this.poi?.city,
          this.poi?.distance,
          this.poi?.placement,
          this.poi?.position
        )
      );
    } else {
      this.dialog.close({
        ...this.poi,
        ...(this.form.getRawValue() as ParcoursPoi),
        provider: this.provider
      });
    }
  }

  /** PRIVATE */

  private groupEventPoiForm(): void {
    const eventPoi: EventPoi = this.poi as EventPoi;

    this.form = this.formBuilder.group({
      name: [eventPoi?.name, Validators.required],
      description: eventPoi?.description,
      url: eventPoi?.url,
      start: [eventPoi?.start, [Validators.required, MinDateValidator(new Date().getTime())]],
      end: [eventPoi?.end, [Validators.required, MinDateValidator(new Date().getTime())]]
    });
  }

  private groupParcoursPoiForm(): void {
    const parcoursPoi: ParcoursPoi = this.poi as ParcoursPoi;

    this.form = this.formBuilder.group({
      name: [parcoursPoi?.name, Validators.required],
      description: parcoursPoi?.description,
      type: [parcoursPoi?.type, Validators.required],
      nodeId: [parcoursPoi?.nodeId],
      applicationId: parcoursPoi?.applicationId || this.context.application.nodeId,
      placement: parcoursPoi?.placement
    });

    // Lock POI name on edit
    if (this.poi?.id) this.form.get('name').disable();
  }
}
