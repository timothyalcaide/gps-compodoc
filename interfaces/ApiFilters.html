export interface ApiFilters {
  /**
   * True if results have to have sandwich training options, false otherwise
   */
  offer_is_sandwich_training?: boolean;

  /**
   * True if results have to be public, false otherwise
   */
  location_is_public?: boolean;

  /**
   * The filter geolocation
   */
  near?: { latitude: number; longitude: number };

  /**
   * The regional filter
   */
  location_regions?: string[];

  /**
   * The max cost filter
   */
  max_cost?: number;

  /**
   * True if results can have exams
   */
  application_offer?: boolean;

  /**
   * The list of required exams
   */
  application_offers?: string[];

  /**
   * Number of waypoints expected by an expand for a domain
   */
  limit_nodes?: number;

  /**
   * Number of domains expected by an expand
   */
  limit_domains?: number;

  /**
   * Number of domains to skip in the full domain list
   */
  skip_domains?: number;

  /**
   * True if results have to be internships
   */
  is_internship?: boolean;

  /**
   * The required status
   */
  status?: string;

  /**
   * The list of required bac specialties
   */
  // TODO: Fix the british and american english between back and front
  bac_specialities?: string[];

  /**
   * The application Uuid
   */
  applicationUuid?: string;

  /**
   * The region used to get job trends
   */
  trends_region?: string;
}
