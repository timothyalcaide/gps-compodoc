import { Waypoint } from '@clients/adventure/bean/waypoint';
import { ApiDefinition } from '@models/definition/definition.api';
import { ApiSituation } from '@models/situation/situation.api';

/**
 * An instance as it is represented on API
 */
export interface ApiInstance {
  /**
   * The instance id
   */
  id: string;

  /**
   * The creation date
   */
  date: string;

  /**
   * The waypoints through which the user has gone
   */
  waypoints: Waypoint[];

  /**
   * The real life duration of the adventure
   */
  endYear: number;

  /**
   * The user situation
   */
  situation: ApiSituation;

  /**
   * The definition from which the instance is created
   */
  definition: ApiDefinition;

  /**
   * The user email to which the instance is attached
   */
  userEmail: string;

  /**
   * Whether the instance has been saved by the attached user
   */
  saved: boolean;
}
