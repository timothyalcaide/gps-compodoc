import { CouponCode } from '@models/coupon-code/coupon-code';
import { ApiLead, Lead, Type } from '@models/lead/lead';
import { User } from '@models/user/user';

/**
 * Represents a lead who claim a reward for a specific application
 * Only one exists for each application, email and rewardId
 */
export class LeadReward extends Lead {
  /**
   * @param id The lead id
   * @param application The application on which the lead is linked
   * @param email The user email
   * @param user The user
   * @param type The type of the lead
   * @param rewardId The id of the linked Reward
   * @param couponCode The coupon code if the reward is of the coupon code type
   * @protected
   */
  constructor(
    public id: string,
    public application: string,
    public email: string,
    public user: User,
    public type: Type,
    public rewardId: string,
    public couponCode: CouponCode
  ) {
    super(id, application, email, user, type);
  }

  public static toLeadReward(application: string, email: string, rewardId: string): LeadReward {
    return new LeadReward(null, application, email, null, Type.REWARD, rewardId, null);
  }

  /**
   * Build the Lead Reward object from the Api Lead Reward object
   *
   * @param apiLeadReward The Api object Lead Reward
   * @returns The new Lead Reward object
   */
  public static build(apiLeadReward: ApiLeadReward): LeadReward {
    if (!apiLeadReward) return null;

    return new LeadReward(
      apiLeadReward.id,
      apiLeadReward.application,
      apiLeadReward.email,
      apiLeadReward.apiUser ? User.build(apiLeadReward.apiUser) : null,
      apiLeadReward.type,
      apiLeadReward.rewardId,
      apiLeadReward.couponCode
    );
  }

  /**
   * Get the Api Lead Reward object from the Lead Reward object
   *
   * @returns The corresponding Api Lead Reward object
   */
  public toApi(): ApiLeadReward {
    return {
      id: this.id,
      application: this.application,
      email: this.email,
      apiUser: this.user?.toApi(),
      type: this.type,
      rewardId: this.rewardId,
      couponCode: this.couponCode
    } as ApiLeadReward;
  }
}

export interface ApiLeadReward extends ApiLead {
  /**
   * The id of the linked Reward
   */
  rewardId: string;

  /**
   * The coupon code if the reward is of the coupon code type
   */
  couponCode?: CouponCode;
}
