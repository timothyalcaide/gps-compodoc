import { Campus } from '@models/campus/campus';

/**
 * Location model that describes a school or a company for a particular waypoint, as it is received from API
 */
export interface ApiLocation {
  id: string;

  name: string;

  /**
   * The number of users who have been in the location
   * For schools, it is the number of users who are studying at or have graduated by this school
   * For companies, it is the number of user who are or have been working in the company
   */
  userCount?: number;

  /**
   * The name of the city where the location is located.
   * Used if the location is located in only one city
   */
  city?: string;

  /**
   * A list of cities where the location is located.
   * Used if the location has numerous cities
   */
  campuses?: Campus[];

  /**
   * The user (or filter) distance from the location
   */
  distance?: number;

  /**
   * (Optional) The location application if the location is a Millionroads client
   */
  application?: string;

  /**
   * The country where the location is located
   */
  country?: string;

  /**
   * The code of the country where the location is located
   */
  countryCode?: string;
}
