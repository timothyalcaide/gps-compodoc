import { Waypoint } from '@clients/adventure/bean/waypoint';
import { PoiProvider } from '@models/poi/poi-provider';
import { PoiType } from '@models/poi/poi-type';
import { StepLocation } from '@models/step-location/step-location';
import { WaypointType } from '@models/waypoint/waypoint-type';

export interface Poi {
  id?: string;
  name: string;
  description?: string;
  type: PoiType;
  provider: PoiProvider;

  /**
   * The POI Rect in the window when it is placed in the map
   */
  rect?: DOMRect;
}

export interface Step {
  id: string;
  name: string;
  type: WaypointType;

  /**
   * The timestamp when the step started
   */
  start: number;

  /**
   * The timestamp when the step ended
   * If the step is ongoing, the end date does not exists
   */
  end?: number;

  isInternship: boolean; // Is stage ?
  isSandwichTraining: boolean; // Is alternance
  // double cursus -> step: diploma && substep: diploma
  location: StepLocation;
  testimony: Testimony; // Témoignages
  substeps: Step[]; // Stages, ect

  /**
   * The waypoint corresponding to the step
   */
  waypoint?: Waypoint;

  /**
   * The VideoAsk id
   */
  videoAskId?: string;
}

export interface Testimony {
  video_url: string;
  text: string;
}

export enum Nomenclature {
  alumni = 'alumni',
  student = 'student',
  user = 'user'
}

export enum Placement {
  start = 'start', // avant stain
  end = 'end', // après stain
  parallel = 'parallel', // left stain
  part_of = 'part_of' // right stain (bottom, top)
}
