import { ChangeDetectionStrategy, Component, Input, OnInit } from '@angular/core';
import { Step } from '@clients/poi/poi';

interface TroncateSteps {
  first: Step[];
  intermediate: Step[];
  last: Step[];
}

@Component({
  selector: 'app-human-steps',
  templateUrl: './human-steps.component.html',
  styleUrls: ['./human-steps.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class HumanStepsComponent implements OnInit {
  @Input() public steps: Step[];

  public truncateSteps: TroncateSteps;
  private readonly MAX_STEPS_DISPLAYED = 4;

  public ngOnInit(): void {
    if (this.steps.length > this.MAX_STEPS_DISPLAYED) {
      this.truncateSteps = this.troncSteps(this.steps, this.MAX_STEPS_DISPLAYED);
    }
  }

  public troncSteps(steps: Step[], num: number): TroncateSteps {
    const { length } = steps;

    const rest = num % 2;
    const split = (num - rest) / 2;

    return {
      first: steps.slice(0, split),
      intermediate: steps.slice(split, length - split),
      last: steps.slice(length - (split + rest), length)
    };
  }
}
