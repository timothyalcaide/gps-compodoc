import { CommonModule } from '@angular/common';
import { NgModule } from '@angular/core';
import { RouterModule, Routes } from '@angular/router';
import { RouteIds } from '@core/routing/route-ids';
import { MapSvgModule } from '@features/map/components/map-svg/map-svg.module';
import { MapService } from '@features/map/services';
import { CanExitMapGuard } from '@guards/can-exit-map.guard';
import { ConfirmationDialogModule } from '@organisms/confirmation-dialog/confirmation-dialog.module';
import { AdventureResolver } from '@resolvers/adventure.resolver';
import { InstanceService } from '@services/instance.service';
import { MapTemplateModule } from '@shared/components/templates/map-template/map-template.module';
import { AdventureMapPageComponent } from './adventure-map.component';

const routes: Routes = [
  {
    path: '',
    component: AdventureMapPageComponent,
    resolve: { instance: AdventureResolver },
    canDeactivate: [CanExitMapGuard],
    data: { title: 'map', routeId: RouteIds.ADVENTURE_MAP }
  }
];

@NgModule({
  declarations: [AdventureMapPageComponent],
  imports: [
    RouterModule.forChild(routes),
    CommonModule,
    ConfirmationDialogModule,
    MapSvgModule,
    MapTemplateModule
  ],
  providers: [AdventureResolver, CanExitMapGuard, InstanceService, MapService]
})
export class AdventureMapPageModule {}
