import { CommonModule } from '@angular/common';
import { NgModule } from '@angular/core';
import { MatIconModule } from '@angular/material/icon';
import { MatTabsModule } from '@angular/material/tabs';
import { RouterModule, Routes } from '@angular/router';
import { NavigationTemplateModule } from '@shared/components/templates/navigation-template/navigation-template.module';
import { SimpleTemplateModule } from '@shared/components/templates/simple-template/simple-template.module';
import { ExpeditionStepDescriptionModule } from './components';
import { ExpeditionInstanceCardModule } from './components/expedition-instance-card/expedition-instance-card.module';
import { ExpeditionPageComponent } from './expedition.component';
import { SharedExpeditionModule } from './modules';

const routes: Routes = [
  {
    path: '', // expeditions
    component: ExpeditionPageComponent,
    data: { title: 'expedition.headline' }
  },
  {
    path: 'library',
    loadChildren: (): Promise<unknown> =>
      "import('./modules/library/library.module').then(m => m.LibraryPageModule)"
  },
  {
    path: ':id',
    loadChildren: (): Promise<unknown> =>
      "import('./modules/expedition-instance-details/expedition-instance-details.module').then(\
        m => m.ExpeditionInstanceDetailsPageModule\
      )"
  },

  {
    path: ':id/challenges/:challengeId',
    loadChildren: (): Promise<unknown> =>
      "import('./modules/expedition-step/expedition-step.module').then(\
        m => m.ExpeditionStepPageModule\
      )"
  }
];

@NgModule({
  declarations: [ExpeditionPageComponent],
  imports: [
    CommonModule,
    ExpeditionInstanceCardModule,
    ExpeditionStepDescriptionModule,
    MatIconModule,
    MatTabsModule,
    NavigationTemplateModule,
    RouterModule.forChild(routes),
    SharedExpeditionModule,
    SimpleTemplateModule
  ]
})
export class ExpeditionPageModule {}
