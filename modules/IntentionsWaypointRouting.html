import { NgModule } from '@angular/core';
import { Route, RouterModule, Routes } from '@angular/router';
import { AppRouteUrls } from '@core/routing/app/route-urls';
import { SituationRedirectionGuard } from '@guards/situation-redirection.guard';

export const INTENTIONS_WAYPOINT_ROUTE: Route = {
  path: AppRouteUrls.INTENTIONS_WAYPOINT(':type'),
  pathMatch: 'full',
  loadChildren: () =>
    import('@pages/intentions/intentions-waypoint/intentions-waypoint.module').then(
      m => m.IntentionsWaypointPageModule
    ),
  canActivate: [SituationRedirectionGuard]
};

export const INTENTIONS_WAYPOINT_DETAILS_ROUTE: Route = {
  path: AppRouteUrls.INTENTIONS_WAYPOINT_DETAILS(':type', ':waypointId'),
  loadChildren: () =>
    import(
      '@pages/intentions/intentions-waypoint/intentions-waypoint-details/intentions-waypoint-details.module'
    ).then(m => m.IntentionsWaypointDetailsPageModule),
  canActivate: [SituationRedirectionGuard]
};

export const INTENTIONS_WAYPOINT_SEARCH_ROUTE: Route = {
  path: AppRouteUrls.INTENTIONS_WAYPOINT_SEARCH(':type'),
  pathMatch: 'full',
  loadChildren: () =>
    import(
      '@pages/intentions/intentions-waypoint/intentions-waypoint-search/intentions-waypoint-search.module'
    ).then(m => m.IntentionsWaypointSearchPageModule),
  canActivate: [SituationRedirectionGuard]
};

export const INTENTIONS_JOB_DEFINITIONS_ROUTE: Route = {
  path: AppRouteUrls.INTENTIONS_WAYPOINT_DEFINITIONS(':type'),
  pathMatch: 'full',
  loadChildren: () =>
    import(
      '@pages/intentions/intentions-waypoint/intentions-waypoint-definitions/intentions-waypoint-definitions.module'
    ).then(m => m.IntentionsWaypointDefinitionsPageModule),
  canActivate: [SituationRedirectionGuard]
};

export const INTENTIONS_JOB_ORIGINS_ROUTE: Route = {
  path: AppRouteUrls.INTENTIONS_JOB_ORIGINS(':jobId'),
  loadChildren: () =>
    import(
      '@pages/intentions/intentions-waypoint/intentions-job-origins/intentions-job-origins.module'
    ).then(m => m.IntentionsJobOriginsPageModule),
  canActivate: [SituationRedirectionGuard]
};

const routes: Routes = [
  INTENTIONS_WAYPOINT_ROUTE,
  INTENTIONS_WAYPOINT_SEARCH_ROUTE,
  INTENTIONS_JOB_DEFINITIONS_ROUTE,
  INTENTIONS_WAYPOINT_DETAILS_ROUTE,
  INTENTIONS_JOB_ORIGINS_ROUTE
];

@NgModule({
  imports: [RouterModule.forChild(routes)],
  exports: [RouterModule],
  providers: [SituationRedirectionGuard]
})
export class IntentionsWaypointRouting {}
