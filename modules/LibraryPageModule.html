import { CommonModule } from '@angular/common';
import { NgModule } from '@angular/core';
import { MatProgressSpinnerModule } from '@angular/material/progress-spinner';
import { RouterModule, Routes } from '@angular/router';
import { RouteIds } from '@core/routing/route-ids';
import { CreateExpeditionModule } from '@features/expeditions/components/create-expedition/create-expedition.module';
import { ExpeditionCardModule } from '@features/expeditions/components/expedition-card/expedition-card.module';
import { FiltersModule } from '@molecules/filters/filters.module';
import { SimpleTemplateModule } from '@shared/components/templates/simple-template/simple-template.module';
import { SharedExpeditionModule } from '../expedition-shared.module';
import { LibraryPageComponent } from './library.component';

const routes: Routes = [
  {
    path: '',
    component: LibraryPageComponent,
    data: { title: 'expedition.new-expedition', routeId: RouteIds.EXPEDITION_LIBRARY }
  },
  {
    path: ':id',
    loadChildren: (): Promise<unknown> =>
      "import('../expedition-instance-details/expedition-instance-details.module').then(\
        m => m.ExpeditionInstanceDetailsPageModule\
      )"
  },
  {
    path: ':id/challenges/:challengeId',
    loadChildren: (): Promise<unknown> =>
      "import('../expedition-step-static/expedition-step-static.module').then(\
        m => m.ExpeditionStepStaticPageModule\
      )"
  }
];

@NgModule({
  declarations: [LibraryPageComponent],
  imports: [
    CommonModule,
    CreateExpeditionModule,
    ExpeditionCardModule,
    FiltersModule,
    MatProgressSpinnerModule,
    RouterModule.forChild(routes),
    SharedExpeditionModule,
    SimpleTemplateModule
  ]
})
export class LibraryPageModule {}
