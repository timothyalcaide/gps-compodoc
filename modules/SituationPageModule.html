import { CommonModule } from '@angular/common';
import { NgModule } from '@angular/core';
import { RouterModule, Routes } from '@angular/router';
import { RouteIds } from '@core/routing/route-ids';
import { SituationFormModule } from '@organisms/situation-form/situation-form.module';
import { TestingMyRoadHomeModule } from '@organisms/testing-my-road-home/testing-my-road-home.module';
import { PartnerResolver } from '@resolvers/partner.resolver';
import { NavigationTemplateModule } from '@shared/components/templates/navigation-template/navigation-template.module';
import { SituationPageComponent } from './situation.component';

const routes: Routes = [
  {
    path: '',
    component: SituationPageComponent,
    resolve: {
      originAndDestinationFromPartner: PartnerResolver
    },
    data: { title: 'navigation.situation', routeId: RouteIds.SITUATION }
  }
];

@NgModule({
  declarations: [SituationPageComponent],
  imports: [
    CommonModule,
    NavigationTemplateModule,
    SituationFormModule,
    RouterModule.forChild(routes),
    TestingMyRoadHomeModule
  ],
  providers: [PartnerResolver]
})
export class SituationPageModule {}
